 144          $continueEnc = false;

 146            $tasEncPos = strpos($Values, $taStartEnc);

 149              $tVal  .= substr($Values, 0, $tasEncPos) . $ImpliedAND;

 151              $continueEnc = true;
 152            }

 165            $continueEnc = false;
 166            if ($eEncPos != strlen($Values) - strlen($EndEnc)) {
 167              $taeEncPos = strpos($Values, $taEndEnc);
 168              $toeEncPos = strpos($Values, $toEndEnc);
 169              if (($taeEncPos !== false) && ($toeEncPos === false || $toeEncPos > $taeEncPos)) {
 170                $encVal      = substr($Values, 0, $taeEncPos);

 289    function CreateComparison($Column,$Value,$Comparison,$Separator,$CompType,$dbtype) {
 290      $theValue="$Value";
 291      if ($Separator=="'") {
 292        $theValue=str_replace("'", "''", $theValue);
 293      }
 294      if ($CompType == 1) {
 295          //numeric

 296       $theValue = "".floatval($theValue);
 297      }
 298     else if ($CompType==2) {
 299          //date or checkbox value

 300          $theValue = WA_DBS_clearOutSQLKeywords($theValue);
 301      }
 302      $theComparison="=";
 303      $theSeparator=$Separator;
 304  
 305      if ($Comparison=="=" || $Comparison=="<>" || $Comparison=="<" || $Comparison=="<=" || $Comparison==">=" || $Comparison==">" || $Comparison=="IS" || $Comparison=="IS NOT" || $Comparison=="LIKE")    {
 306        $theComparison=$Comparison;
 307      }
 308      if ($Comparison=="IS" || $Comparison=="IS NOT")     {
 309        $theSeparator="";
 310      }
 311      if (isset($dbtype)) {
 312        if ($dbtype == "ORACLE") {
 313          if ($CompType == 2)     {
 314            return "(".WA_DBS_cleanUpColumnName($Column)." ".$theComparison." ".$theSeparator.$theValue."', 'MM/DD/YY'))";
 315          }
 316          else     {
 317            return "(".WA_DBS_cleanUpColumnName($Column)." ".$theComparison." ".$theSeparator.$theValue.$theSeparator.")";
 318          }
 319        }
 320        if ($dbtype == "MYSQL")     {
 321          if ($CompType == 2)      {
 322            return "(".WA_DBS_cleanUpColumnName($Column)." ".$theComparison." ".$theSeparator.getMySQLDateORDIE($theValue).$theSeparator.")";
 323          }
 324          else      {
 325            return "(".WA_DBS_cleanUpColumnName($Column)." ".$theComparison." ".$theSeparator.$theValue.$theSeparator.")";
 326          }
 327        }
 328      }
 329      return "(".WA_DBS_cleanUpColumnName($Column)." ".$theComparison." ".$theSeparator.$theValue.$theSeparator.")";

 379  function setQueryBuilderSource(&$RS_Source,$QB,$useOr)      {

 381    if ($QB->ObjectType == "WC") {
 382      $theClause = $QB->whereClause;

 384        return;
 385      }    
 386      $theConnector = ($useOr) ? "OR" : "AND";
 387      $theIndex = findSourceInsertIndex($RS_Source);
 388      if($theIndex[1]){
 389        $theClause = preg_replace("/^\s*WHERE\s{1,}/i", $theConnector." ", $theClause);
 390      }
 391      else{
 392        $theClause = "WHERE ".preg_replace("/^\s*(AND|OR)\s{1,}/i", " ", $theClause);
 393      }
 394    }
 395    else if ($QB->ObjectType == "OB") {
 396      $theIndex = array();
 397      $theIndex[0] = strlen($RS_Source);
 398      $theClause = $QB->getFinalClause($RS_Source);
 399    }
 400    if ($theClause != "") {
 401      $RS_Source = substr($RS_Source, 0, $theIndex[0])." ".$theClause." ".substr($RS_Source, $theIndex[0]);

 490  function findSourceInsertIndex($source)     {
 491    $returnIndex = 0;
 492    $indexOfLastFrom = (preg_match("/\s{1,}(FROM)\s{1,}/i",$source, $matches)>0) ? strpos($source,$matches[0])+1 : false;
 493    $sourceAfterLastFrom = substr($source, $indexOfLastFrom);
 494    $indexOfNextFrom = (preg_match("/\s{1,}(FROM)\s{1,}/i",$sourceAfterLastFrom, $matches)>0) ? strpos($sourceAfterLastFrom,$matches[0]) : false;
 495    while ($indexOfNextFrom > 0)  {
 496      $indexOfLastFrom+=$indexOfNextFrom+1;
 497      $sourceAfterLastFrom = substr($sourceAfterLastFrom, $indexOfNextFrom+1);
 498      $indexOfNextFrom = (preg_match("/\s{1,}(FROM)\s{1,}/i",$sourceAfterLastFrom, $matches)>0) ? strpos($sourceAfterLastFrom,$matches[0]) : false;
 499    }
 500    
 501    $indexOfLastON = (preg_match("/\s{1,}(ON)\s{1,}/i",$sourceAfterLastFrom, $matches)>0) ? strpos($sourceAfterLastFrom,$matches[0]) : false;
 502    while ($indexOfLastON !== false)  {
 503      $indexOfLastFrom+=$indexOfLastON+1;
 504      $sourceAfterLastFrom = substr($sourceAfterLastFrom, $indexOfLastON+1);
 505      $indexOfLastON = (preg_match("/\s{1,}(ON)\s{1,}/i",$sourceAfterLastFrom, $matches)>0) ? strpos($sourceAfterLastFrom,$matches[0]) : false;
 506    }
 507  
 508  
 509    $indexOfOrderBy = (preg_match("/\s{1,}(ORDER BY)\s{1,}/i",$sourceAfterLastFrom, $matches)>0) ? strpos($sourceAfterLastFrom,$matches[0]) : false;
 510    $indexOfGroupBy = (preg_match("/\s{1,}(GROUP BY)\s{1,}/i",$sourceAfterLastFrom, $matches)>0) ? strpos($sourceAfterLastFrom,$matches[0]) : false;
 511  
 512    if ($indexOfOrderBy > 0)
 513      $returnIndex = $indexOfLastFrom+$indexOfOrderBy;
 514    if ($indexOfGroupBy > 0)
 515      $returnIndex = $indexOfLastFrom+$indexOfGroupBy;
 516    if ($returnIndex == 0){
 517          $returnIndex = (preg_match("/(;\s*)$/",$source, $matches)>0) ? strpos($source,$matches[0]) : strlen($source);
 518    }
 519    return Array($returnIndex,(strpos(strtoupper($sourceAfterLastFrom), "WHERE") || strpos(strtoupper($sourceAfterLastFrom), "WHERE")===0));
 520  }
