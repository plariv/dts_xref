   7  /* Horizontal Tabbed Panels

   8   *

   9   * The default style for a TabbedPanels widget places all tab buttons

  10   * (left aligned) above the content panel.

  11   */
  12  
  13  /* This is the selector for the main TabbedPanels container. For our

  14   * default style, this container does not contribute anything visually,

  15   * but it is floated left to make sure that any floating or clearing done

  16   * with any of its child elements are contained completely within the

  17   * TabbedPanels container, to minimize any impact or undesireable

  18   * interaction with other floated elements on the page that may be used

  19   * for layout.

  20   *

  21   * If you want to constrain the width of the TabbedPanels widget, set a

  22   * width on the TabbedPanels container. By default, the TabbedPanels widget

  23   * expands horizontally to fill up available space.

  24   *

  25   * The name of the class ("TabbedPanels") used in this selector is not

  26   * necessary to make the widget function. You can use any class name you

  27   * want to style the TabbedPanels container.

  28   */
  29  .TabbedPanels {
  30      padding: 0px;
  31      float: left;
  32      clear: none;
  33      margin-top: 4px;
  34      margin-right: 0px;
  35      margin-bottom: 4px;
  36      margin-left: 0px;
  37  }
  38  
  39  /* This is the selector for the TabGroup. The TabGroup container houses

  40   * all of the tab buttons for each tabbed panel in the widget. This container

  41   * does not contribute anything visually to the look of the widget for our

  42   * default style.

  43   *

  44   * The name of the class ("TabbedPanelsTabGroup") used in this selector is not

  45   * necessary to make the widget function. You can use any class name you

  46   * want to style the TabGroup container.

  47   */
  48  .TabbedPanelsTabGroup {
  49      margin: 0px;
  50      padding: 0px;
  51  }
  52  
  53  /* This is the selector for the TabbedPanelsTab. This container houses

  54   * the title for the panel. This is also the tab "button" that the user clicks

  55   * on to activate the corresponding content panel so that it appears on top

  56   * of the other tabbed panels contained in the widget.

  57   *

  58   * For our default style, each tab is positioned relatively 1 pixel down from

  59   * where it wold normally render. This allows each tab to overlap the content

  60   * panel that renders below it. Each tab is rendered with a 1 pixel bottom

  61   * border that has a color that matches the top border of the current content

  62   * panel. This gives the appearance that the tab is being drawn behind the

  63   * content panel.

  64   *

  65   * The name of the class ("TabbedPanelsTab") used in this selector is not

  66   * necessary to make the widget function. You can use any class name you want

  67   * to style this tab container.

  68   */
  69  .TabbedPanelsTab {
  70      position: relative;
  71      top: 0px;
  72      float: left;
  73      padding: 1px 8px 1px 8px;
  74      margin: 0px 1px 0px 0px;
  75      background-color: #C8C8C8;
  76      list-style: none;
  77      border-left: solid 1px #CCC;
  78      border-bottom: solid 0px #999;
  79      border-top: solid 1px #999;
  80      border-right: solid 1px #999;
  81      -moz-user-select: none;
  82      -khtml-user-select: none;
  83      cursor: pointer;
  84      font-family: Verdana, Arial, Helvetica, sans-serif;
  85      font-size: 12px;
  86      color: #5E625E;
  87  }
  88  
  89  /* This selector is an example of how to change the appearnce of a tab button

  90   * container as the mouse enters it. The class "TabbedPanelsTabHover" is

  91   * programatically added and removed from the tab element as the mouse enters

  92   * and exits the container.

  93   */
  94  .TabbedPanelsTabHover {
  95      background-color: #8D8C7C;
  96      color: #FFFFFF;
  97  }
  98  
  99  /* This selector is an example of how to change the appearance of a tab button

 100   * container after the user has clicked on it to activate a content panel.

 101   * The class "TabbedPanelsTabSelected" is programatically added and removed

 102   * from the tab element as the user clicks on the tab button containers in

 103   * the widget.

 104   *

 105   * As mentioned above, for our default style, tab buttons are positioned

 106   * 1 pixel down from where it would normally render. When the tab button is

 107   * selected, we change its bottom border to match the background color of the

 108   * content panel so that it looks like the tab is part of the content panel.

 109   */
 110  .TabbedPanelsTabSelected {
 111      background-color: #78756F;
 112      border-bottom: 1px solid #EEE;
 113      color: #FFFFFF;
 114  }
 115  
 116  /* This selector is an example of how to make a link inside of a tab button

 117   * look like normal text. Users may want to use links inside of a tab button

 118   * so that when it gets focus, the text *inside* the tab button gets a focus

 119   * ring around it, instead of the focus ring around the entire tab.

 120   */
 121  .TabbedPanelsTab a {
 122      color: black;
 123      text-decoration: none;
 124  }

 126  /* This is the selector for the ContentGroup. The ContentGroup container houses

 127   * all of the content panels for each tabbed panel in the widget. For our

 128   * default style, this container provides the background color and borders that

 129   * surround the content.

 130   *

 131   * The name of the class ("TabbedPanelsContentGroup") used in this selector is

 132   * not necessary to make the widget function. You can use any class name you

 133   * want to style the ContentGroup container.

 134   */
 135  .TabbedPanelsContentGroup {
 136      clear: both;
 137      border-left: solid 0px #999;
 138      border-bottom: solid 1px #CCC;
 139      border-top: solid 1px #999;
 140      border-right: solid 0px #CCC;
 141  }
 142  
 143  /* This is the selector for the Content panel. The Content panel holds the

 144   * content for a single tabbed panel. For our default style, this container

 145   * provides some padding, so that the content is not pushed up against the

 146   * widget borders.

 147   *

 148   * The name of the class ("TabbedPanelsContent") used in this selector is

 149   * not necessary to make the widget function. You can use any class name you

 150   * want to style the Content container.

 151   */
 152  .TabbedPanelsContent {
 153      padding: 0px;
 154      font-size: 12px;
 155      line-height: 18px;
 156      margin-top: 6px;
 157      color: #333333;
 158      margin-left: 0px;
 159  }

 161  /* This selector is an example of how to change the appearnce of the currently

 162   * active container panel. The class "TabbedPanelsContentVisible" is

 163   * programatically added and removed from the content element as the panel

 164   * is activated/deactivated.

 165   */
 166  .TabbedPanelsContentVisible {
 167  }

 169  /* Vertical Tabbed Panels

 170   *

 171   * The following rules override some of the default rules above so that the

 172   * TabbedPanels widget renders with its tab buttons along the left side of

 173   * the currently active content panel.

 174   *

 175   * With the rules defined below, the only change that will have to be made

 176   * to switch a horizontal tabbed panels widget to a vertical tabbed panels

 177   * widget, is to use the "VTabbedPanels" class on the top-level widget

 178   * container element, instead of "TabbedPanels".

 179   */
 180  
 181  /* This selector floats the TabGroup so that the tab buttons it contains

 182   * render to the left of the active content panel. A border is drawn around

 183   * the group container to make it look like a list container.

 184   */
 185  .VTabbedPanels .TabbedPanelsTabGroup {
 186      float: left;
 187      width: auto;
 188      background-color: #EEE;
 189      position: relative;
 190      border-top: solid 1px #999;
 191      border-right: solid 1px #999;
 192      border-left: solid 1px #CCC;
 193      border-bottom: solid 1px #CCC;
 194  }
 195  
 196  /* This selector disables the float property that is placed on each tab button

 197   * by the default TabbedPanelsTab selector rule above. It also draws a bottom

 198   * border for the tab. The tab button will get its left and right border from

 199   * the TabGroup, and its top border from the TabGroup or tab button above it.

 200   */
 201  .VTabbedPanels .TabbedPanelsTab {
 202      float: none;
 203      margin: 0px;
 204      border-top: none;
 205      border-left: none;
 206      border-right: none;
 207  }
 208  
 209  /* This selector disables the float property that is placed on each tab button

 210   * by the default TabbedPanelsTab selector rule above. It also draws a bottom

 211   * border for the tab. The tab button will get its left and right border from

 212   * the TabGroup, and its top border from the TabGroup or tab button above it.

 213   */
 214  .VTabbedPanels .TabbedPanelsTabSelected {
 215      background-color: #EEE;
 216      border-bottom: solid 1px #999;
 217  }
 218  
 219  /* This selector floats the content panels for the widget so that they

 220   * render to the right of the tabbed buttons.

 221   */
 222  .VTabbedPanels .TabbedPanelsContentGroup {
 223      clear: none;
 224      float: left;
 225      padding: 0px;
 226      width: auto;
 227      height: inherit;
 228  }
