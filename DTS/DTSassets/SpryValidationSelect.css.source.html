   7  /* These are the classes applied on the messages

   8   * (required message and invalid state message)

  15  /* These selectors change the way messages look when the widget is in one of the error states (required, invalid).

  16   * These classes set a default red border and color for the error text.

  17   * The state class (.selectRequiredState or .selectInvalidState) is applied on the top-level container for the widget,

  18   * and this way only the specific error message can be shown by setting the display property to "inline".

  19   */
  20  .selectRequiredState .selectRequiredMsg,
  21  .selectInvalidState .selectInvalidMsg {
  22      display: inline;
  23      color: #641615;
  24      }
  25  
  26  /* The next three group selectors control the way the core element (SELECT) looks like when the widget is in one of the states: 

  27   * focus, required / invalid, valid 

  28   * There are two selectors for each state, to cover the two main usecases for the widget:

  29   * - the widget id is placed on the top level container for the SELECT

  30   * - the widget id is placed on the SELECT element itself (there are no error messages)

  31   */
  32   
  33  /* When the widget is in the valid state the SELECT has a green background applied on it. */

  34  .selectValidState select, select.selectValidState {
  35      background-color: #FDFFEE;
  36      }
  37  
  38  /* When the widget is in an invalid state the SELECT has a red background applied on it. */

  39  select.selectRequiredState, .selectRequiredState select,
  40  select.selectInvalidState, .selectInvalidState select {
  41      background-color: #DCCDCA;
  42      }
  43  
  44  /* When the widget has received focus, the SELECT has a yellow background applied on it. */
