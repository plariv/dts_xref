 524  2.2.1. Preferred

 525  x:x:x:x:x:x:x:x, where the 'x's are the hexadecimal values of the eight 16-bit pieces of the address.

 526  Examples:

 527      FEDC:BA98:7654:3210:FEDC:BA98:7654:3210

 528      1080:0:0:0:8:800:200C:417A

 529  Note that it is not necessary to write the leading zeros in an

 530  individual field, but there must be at least one numeral in every

 531  field (except for the case described in 2.2.2.).

 532  

 533  2.2.2. Compressed

 534  The use of "::" indicates multiple groups of 16-bits of zeros.

 535  The "::" can only appear once in an address.  The "::" can also be

 536  used to compress the leading and/or trailing zeros in an address.

 537      1080:0:0:0:8:800:200C:417A --> 1080::8:800:200C:417A

 538      FF01:0:0:0:0:0:0:101 --> FF01::101

 553        IPv6prefix  = hexpart "/" 1*2DIGIT

 554  

 555        hexpart = hexseq | hexseq "::" [ hexseq ] | "::" [ hexseq ]

 556        hexseq  = hex4 *( ":" hex4)

 557        hex4    = 1*4HEXDIG

 558  */
 559  Spry.Widget.ValidationTextField.validateIP = function (value, format)
 560  {
 561      var validIPv6Addresses = [
 562          //preferred

 563          /^(?:[a-f0-9]{1,4}:){7}[a-f0-9]{1,4}(?:\/\d{1,3})?$/i,
 564  
 565          //various compressed

 566          /^[a-f0-9]{0,4}::(?:\/\d{1,3})?$/i,
 567          /^:(?::[a-f0-9]{1,4}){1,6}(?:\/\d{1,3})?$/i,
 568          /^(?:[a-f0-9]{1,4}:){1,6}:(?:\/\d{1,3})?$/i,
 569          /^(?:[a-f0-9]{1,4}:)(?::[a-f0-9]{1,4}){1,6}(?:\/\d{1,3})?$/i,
 570          /^(?:[a-f0-9]{1,4}:){2}(?::[a-f0-9]{1,4}){1,5}(?:\/\d{1,3})?$/i,
 571          /^(?:[a-f0-9]{1,4}:){3}(?::[a-f0-9]{1,4}){1,4}(?:\/\d{1,3})?$/i,
 572          /^(?:[a-f0-9]{1,4}:){4}(?::[a-f0-9]{1,4}){1,3}(?:\/\d{1,3})?$/i,
 573          /^(?:[a-f0-9]{1,4}:){5}(?::[a-f0-9]{1,4}){1,2}(?:\/\d{1,3})?$/i,
 574          /^(?:[a-f0-9]{1,4}:){6}(?::[a-f0-9]{1,4})(?:\/\d{1,3})?$/i,
 575  
 576  
 577          //IPv6 mixes with IPv4

 578          /^(?:[a-f0-9]{1,4}:){6}(?:\d{1,3}\.){3}\d{1,3}(?:\/\d{1,3})?$/i,
 579          /^:(?::[a-f0-9]{1,4}){0,4}:(?:\d{1,3}\.){3}\d{1,3}(?:\/\d{1,3})?$/i,
 580          /^(?:[a-f0-9]{1,4}:){1,5}:(?:\d{1,3}\.){3}\d{1,3}(?:\/\d{1,3})?$/i,
 581          /^(?:[a-f0-9]{1,4}:)(?::[a-f0-9]{1,4}){1,4}:(?:\d{1,3}\.){3}\d{1,3}(?:\/\d{1,3})?$/i,
 582          /^(?:[a-f0-9]{1,4}:){2}(?::[a-f0-9]{1,4}){1,3}:(?:\d{1,3}\.){3}\d{1,3}(?:\/\d{1,3})?$/i,    
 583          /^(?:[a-f0-9]{1,4}:){3}(?::[a-f0-9]{1,4}){1,2}:(?:\d{1,3}\.){3}\d{1,3}(?:\/\d{1,3})?$/i,
 584          /^(?:[a-f0-9]{1,4}:){4}(?::[a-f0-9]{1,4}):(?:\d{1,3}\.){3}\d{1,3}(?:\/\d{1,3})?$/i
 585      ];
 586      var validIPv4Addresses = [
 587          //IPv4

 588          /^(\d{1,3}\.){3}\d{1,3}$/i
 589      ];
 590      var validAddresses = [];
 591      if (format == 'ipv6' || format == 'ipv6_ipv4') {
 592          validAddresses = validAddresses.concat(validIPv6Addresses);
 593      }
 594      if (format == 'ipv4' || format == 'ipv6_ipv4') {
 595          validAddresses = validAddresses.concat(validIPv4Addresses);
 596      }
 597  
 598      var ret = false;
 599      for (var i=0; i<validAddresses.length; i++) {
 600          if (validAddresses[i].test(value)) {
 601              ret = true;
 602              break;
 603          }
 604      }
 605  
 606      if (ret && value.indexOf(".") != -1) {
 607          //if address contains IPv4 fragment, it must be valid; all 4 groups must be less than 256

 608          var ipv4 = value.match(/:?(?:\d{1,3}\.){3}\d{1,3}/i);
 609          if(!ipv4) {
 610              return false;
 611          }
 612          ipv4 = ipv4[0].replace(/^:/, '');
 613          var pieces = ipv4.split('.');
 614          if (pieces.length != 4) {
 615              return false;
 616          }
 617          var regExp = /^[\-\+]?\d*$/;
 618          for (var i=0; i< pieces.length; i++) {
 619              if (pieces[i] == '') {
 620                  return false;
 621              }
 622              var piece = parseInt(pieces[i], 10);
 623              if (isNaN(piece) || piece > 255 || !regExp.test(pieces[i]) || pieces[i].length>3 || /^0{2,3}$/.test(pieces[i])) {
 624                  return false;
 625              }
 626          }
 627      }
 628      if (ret && value.indexOf("/") != -1) {
 629          // if prefix-length is specified must be in [1-128]

 630          var prefLen = value.match(/\/\d{1,3}$/);
 631          if (!prefLen) return false;
 632          var prefLenVal = parseInt(prefLen[0].replace(/^\//,''), 10);
 633          if (isNaN(prefLenVal) || prefLenVal > 128 || prefLenVal < 1) {
 634              return false;
 635          }
 636      }
 637      return ret;
 638  };

 929  Spry.Widget.ValidationTextField.prototype.doValidations = function(testValue, revertValue)
 930  {
 931      if (this.isDisabled()) return false;
 932  
 933      if (this.flags.locked) {
 934          return false;
 935      }
 936  
 937      if (testValue.length == 0 && !this.isRequired) {
 938          this.errors = 0;
 939          return false;
 940      }
 941      this.flags.locked = true;
 942  
 943      var mustRevert = false;
 944      var continueValidations = true;
 945      if (!this.options.isRequired && testValue.length == 0) {
 946          continueValidations = false;
 947      }
 948  
 949      var errors = 0;
 950      var fixedValue = testValue;
 951  
 952      //characterMasking - test if all characters are valid with the characterMasking (keyboard filter)

 953      if (this.useCharacterMasking && this.characterMasking) {
 954          for(var i=0; i<testValue.length; i++) {
 955              if (!this.characterMasking.test(testValue.charAt(i))) {
 956                  errors = errors | Spry.Widget.ValidationTextField.ERROR_FORMAT;
 957                  fixedValue = revertValue;
 958                  mustRevert = true;
 959                  break;
 960              }
 961          }
 962      }
 963  
 964      //regExpFilter - character mask positioning (additional mask to restrict some characters only in some position)

 965      if (!mustRevert && this.useCharacterMasking && this.regExpFilter) {
 966          if (!this.regExpFilter.test(fixedValue)) {
 967              errors = errors | Spry.Widget.ValidationTextField.ERROR_FORMAT;
 968              mustRevert = true;
 969          }
 970      }
 971  
 972      //pattern - testValue matches the pattern so far

 973      if (!mustRevert && this.pattern) {
 974          var currentRegExp = this.patternToRegExp(testValue.length);
 975          if (!currentRegExp.test(testValue)) {
 976              errors = errors | Spry.Widget.ValidationTextField.ERROR_FORMAT;
 977              mustRevert = true;
 978          } else if (this.patternLength != testValue.length) {
 979              //testValue matches pattern so far, but it's not ok if it does not have the proper length

 980              //do not revert, but should show the error

 981              errors = errors | Spry.Widget.ValidationTextField.ERROR_FORMAT;
 982          }
 983      }
 984  
 985      if (fixedValue == '') {
 986          errors = errors | Spry.Widget.ValidationTextField.ERROR_REQUIRED;
 987      }
 988  
 989      if (!mustRevert && this.pattern && this.useCharacterMasking) {
 990          var n = this.getAutoComplete(testValue.length);
 991          if (n) {
 992              fixedValue += n;
 993          }
 994      }
 995  
 996      if(!mustRevert && this.minChars !== null  && continueValidations) {
 997          if (testValue.length < this.minChars) {
 998              errors = errors | Spry.Widget.ValidationTextField.ERROR_CHARS_MIN;
 999              continueValidations = false;
1000          }
1001      }
1002  
1003      if(!mustRevert && this.maxChars !== null && continueValidations) {
1004          if (testValue.length > this.maxChars) {
1005              errors = errors | Spry.Widget.ValidationTextField.ERROR_CHARS_MAX;
1006              continueValidations = false;
1007          }
1008      }
1009  
1010      //validation - testValue passes widget validation function

1011      if (!mustRevert && this.validation && continueValidations) {
1012          var value = this.validation(fixedValue, this.options);
1013          if (false === value) {
1014              errors = errors | Spry.Widget.ValidationTextField.ERROR_FORMAT;
1015              continueValidations = false;
1016          } else {
1017              this.typedValue = value;
1018          }
1019      }
1020  
1021      if(!mustRevert && this.validation && this.minValue !== null && continueValidations) {
1022          var minValue = this.validation(this.minValue.toString(), this.options);
1023          if (minValue !== false) {
1024              if (this.typedValue < minValue) {
1025                  errors = errors | Spry.Widget.ValidationTextField.ERROR_RANGE_MIN;
1026                  continueValidations = false;
1027              }
1028          }
1029      }
1030  
1031      if(!mustRevert && this.validation && this.maxValue !== null && continueValidations) {
1032          var maxValue = this.validation(this.maxValue.toString(), this.options);
1033          if (maxValue !== false) {
1034              if( this.typedValue > maxValue) {
1035                  errors = errors | Spry.Widget.ValidationTextField.ERROR_RANGE_MAX;
1036                  continueValidations = false;
1037              }
1038          }
1039      }
1040  
1041      //an invalid value was tested; must make sure it does not get inside the input

1042      if (this.useCharacterMasking && mustRevert) {
1043          this.revertState(revertValue);
1044      }
1045  
1046      this.errors = errors;
1047      this.fixedValue = fixedValue;
1048  
1049      this.flags.locked = false;
1050  
1051      return mustRevert;
1052  };

1054  Spry.Widget.ValidationTextField.prototype.onChange = function(e)
1055  {
1056      if (Spry.is.opera && this.flags.operaRevertOnKeyUp) {
1057          return true;
1058      }
1059      if (Spry.is.ie && e && e.propertyName != 'value') {
1060          return true;
1061      }
1062
