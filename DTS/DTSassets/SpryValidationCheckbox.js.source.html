  34  Spry.Widget.ValidationCheckbox = function(element, opts)
  35  {
  36      this.init(element);
  37      
  38      Spry.Widget.Utils.setOptions(this, opts);
  39  
  40      // set validateOn flags

  41      var validateOn = ['submit'].concat(this.validateOn || []);
  42      validateOn = validateOn.join(",");
  43      this.validateOn = 0 | (validateOn.indexOf('submit') != -1 ? Spry.Widget.ValidationCheckbox.ONSUBMIT : 0);
  44      this.validateOn = this.validateOn | (validateOn.indexOf('blur') != -1 ? Spry.Widget.ValidationCheckbox.ONBLUR : 0);
  45      this.validateOn = this.validateOn | (validateOn.indexOf('change') != -1 ? Spry.Widget.ValidationCheckbox.ONCHANGE : 0);
  46      
  47      // sanity checks

  48      if (!isNaN(this.minSelections)) {
  49          this.minSelections = (this.minSelections > 0)? parseInt(this.minSelections, 10): null;
  50      }
  51      if (!isNaN(this.maxSelections)) {
  52          this.maxSelections = (this.maxSelections > 0)? parseInt(this.maxSelections, 10): null;
  53      }
  54  
  55      if (this.additionalError)
  56          this.additionalError = this.getElement(this.additionalError);
  57      // Unfortunately in some browsers like Safari, the Stylesheets our

  58      // page depends on may not have been loaded at the time we are called.

  59      // This means we have to defer attaching our behaviors until after the

  60      // onload event fires, since some of our behaviors rely on dimensions

  61      // specified in the CSS.

  62  
  63      if (Spry.Widget.ValidationCheckbox.onloadDidFire)
  64          this.attachBehaviors();
  65      else 
  66          Spry.Widget.ValidationCheckbox.loadQueue.push(this);
  67  };

  99      if (this.event_handlers)
 100          for (var i=0; i<this.event_handlers.length; i++)
 101          {
 102              Spry.Widget.Utils.removeEventListener(this.event_handlers[i][0], this.event_handlers[i][1], this.event_handlers[i][2], false);
 103          }
 104      try { delete this.element; } catch(err) {}
 105      if (this.checkboxElements)

 108              try { delete this.checkboxElements[i];} catch(err) {}

 115      var qlen = q.length;
 116      for (var i = 0; i < qlen; i++) {
 117          if (q[i] == this) {
 118              q.splice(i, 1);

 158  Spry.Widget.ValidationCheckbox.prototype.attachBehaviors = function()
 159  {
 160      if (!this.element)
 161          return;
 162      // find the INPUT type="checkbox" element(s) inside current container

 163      if (this.element.nodeName == "INPUT") {
 164          this.checkboxElements = [this.element];
 165      } else {
 166          this.checkboxElements = this.getCheckboxes();
 167      }
 168      if (this.checkboxElements) {
 169          var self = this;
 170          this.event_handlers = [];
 171  
 172          var qlen = this.checkboxElements.length;
 173          for (var i = 0; i < qlen; i++) {
 174              // focus

 175              this.event_handlers.push([this.checkboxElements[i], "focus", function(e) { return self.onFocus(e); }]);
 176              // blur

 177              this.event_handlers.push([this.checkboxElements[i], "blur", function(e) { return self.onBlur(e); }]);
 178              // add click instead of onChange

 179              if (this.validateOn & Spry.Widget.ValidationCheckbox.ONCHANGE) {
 180                  this.event_handlers.push([this.checkboxElements[i], "click", function(e) { return self.onClick(e); }]);
 181              }
 182          }
 183  
 184          for (var i=0; i<this.event_handlers.length; i++) {
 185              Spry.Widget.Utils.addEventListener(this.event_handlers[i][0], this.event_handlers[i][1], this.event_handlers[i][2], false);
 186          }
 187  
 188          // submit

 189          this.form = Spry.Widget.Utils.getFirstParentWithNodeName(this.element, "FORM");
 190          if (this.form) {
 191              // if no "onSubmit" handler has been attached to the current form, attach one

 192              if (!this.form.attachedSubmitHandler && !this.form.onsubmit) {
 193                  this.form.onsubmit = function(e) { e = e || event; return Spry.Widget.Form.onSubmit(e, e.srcElement || e.currentTarget) };
 194                  this.form.attachedSubmitHandler = true;                 
 195              }
 196              if (!this.form.attachedResetHandler) {
 197                  Spry.Widget.Utils.addEventListener(this.form, "reset", function(e) { e = e || event; return Spry.Widget.Form.onReset(e, e.srcElement || e.currentTarget) }, false);
 198                  this.form.attachedResetHandler = true;                 
 199              }
 200              // add the currrent widget to the "onSubmit" check queue;

 201              Spry.Widget.Form.onSubmitWidgetQueue.push(this);
 202          }
 203      }
 204      
 205  };

 207  Spry.Widget.ValidationCheckbox.prototype.getCheckboxes = function() {
 208      var arrCheckboxes;
 209      var elements  = this.element.getElementsByTagName("INPUT");
 210      if (elements.length) {
 211          arrCheckboxes = [];
 212          var qlen = elements.length;
 213          for (var i = 0; i < qlen; i++) {
 214              if (elements[i].type == "checkbox") {
 215                  arrCheckboxes.push(elements[i]);
 216              }
 217          }
 218          return arrCheckboxes;
 219      }
 220      return null;

 249  Spry.Widget.ValidationCheckbox.prototype.onBlur = function(e)
 250  {
 251      var eventCheckbox = (e.srcElement != null) ? e.srcElement : e.target;
 252      if (eventCheckbox.disabled) return;
 253      
 254      this.hasFocus = false;
 255      var doValidation = false;
 256      if (this.validateOn & Spry.Widget.ValidationCheckbox.ONBLUR)
 257          doValidation = true;
 258      if (doValidation)
 259          this.validate();
 260      this.removeClassName(this.element, this.focusClass);
 261      this.removeClassName(this.additionalError, this.focusClass);
 262      
 263  };

 283  Spry.Widget.ValidationCheckbox.prototype.validate = function() {
 284      this.reset();
 285  
 286      var nochecked = 0;
 287      if (this.checkboxElements) {
 288          var qlen = this.checkboxElements.length;
 289          for (var i = 0; i < qlen; i++) {
 290              if (!this.checkboxElements[i].disabled && this.checkboxElements[i].checked) {
 291                  nochecked++;
 292              }
 293          }
 294      }
 295  
 296      // check isRequired

 297      if (this.isRequired) {
 298          if (nochecked == 0) {
 299              this.addClassName(this.element, this.requiredClass);
 300              this.addClassName(this.additionalError, this.requiredClass);
 301              return false;
 302          }
 303      }
 304      if (this.minSelections) {
 305          if (this.minSelections > nochecked) {
 306              this.addClassName(this.element, this.minSelectionsClass);
 307              this.addClassName(this.additionalError, this.minSelectionsClass);
 308              return false;
 309          }
 310      }
 311      if (this.maxSelections) {
 312          if (this.maxSelections < nochecked) {
 313              this.addClassName(this.element, this.maxSelectionsClass);
 314              this.addClassName(this.additionalError, this.maxSelectionsClass);
 315              return false;
 316          }
 317      }
 318      this.addClassName(this.element, this.validClass);
 319      this.addClassName(this.additionalError, this.validClass);
 320      return true;
 321  };

 324      var ret = true;

 326          var qlen = this.checkboxElements.length;

 433  Spry.Widget.Utils.getFirstParentWithNodeName = function(node, nodeName)
 434  {
 435      while (node.parentNode
 436              && node.parentNode.nodeName.toLowerCase() != nodeName.toLowerCase()
 437              && node.parentNode.nodeName != 'BODY') {
 438          node = node.parentNode;
 439      }
 440  
 441      if (node.parentNode && node.parentNode.nodeName.toLowerCase() == nodeName.toLowerCase()) {
 442          return node.parentNode;
 443      } else {
 444          return null;
 445      }

 463  Spry.Widget.Utils.contains = function (who, what)
 464  {
 465      if (typeof who.contains == 'object') {
 466          return what && who && (who == what || who.contains(what));
 467      } else {
 468          var el = what;
 469          while(el) {
 470              if (el == who) {
 471                  return true;
 472              }
 473              el = el.parentNode;
 474          }
 475          return false;
 476      }
 477  };
