  39  Spry.Effect.Transitions = {
  40      linearTransition: function(time, begin, change, duration)
  41      {
  42          if (time > duration) return change+begin;
  43          return begin + (time / duration) * change;
  44      },
  45      sinusoidalTransition: function(time, begin, change, duration)
  46      {
  47          if (time > duration) return change+begin;
  48          return begin + ((-Math.cos((time/duration)*Math.PI)/2) + 0.5) * change;
  49      },
  50      squareTransition: function(time, begin, change, duration)
  51      {
  52          if (time > duration) return change+begin;
  53          return begin + Math.pow(time/duration, 2) * change;
  54      },
  55      squarerootTransition: function(time, begin, change, duration)
  56      {
  57          if (time > duration) return change+begin;
  58          return begin + Math.sqrt(time/duration) * change;
  59      },
  60      fifthTransition: function(time, begin, change, duration)
  61      {
  62          if (time > duration) return change+begin;
  63          return begin + Math.sqrt((-Math.cos((time/duration)*Math.PI)/2) + 0.5) * change;
  64      },
  65      circleTransition: function(time, begin, change, duration)
  66      {
  67          if (time > duration) return change+begin;
  68          var pos = time/duration;
  69          return begin + Math.sqrt(1 - Math.pow((pos-1), 2))* change;
  70      },
  71      pulsateTransition: function(time, begin, change, duration)
  72      {
  73          if (time > duration) return change+begin;
  74          return begin + (0.5 + Math.sin(17*time/duration)/2) * change;
  75      },
  76      growSpecificTransition: function(time, begin, change, duration)
  77      {
  78          if (time > duration) return change+begin;
  79          var pos = time/duration;
  80          return begin + (5 * Math.pow(pos, 3) - 6.4 * Math.pow(pos, 2) + 2 * pos) * change;
  81      }

 124  Spry.Effect.Registry.prototype.effectsAreTheSame = function(effectA, effectB)
 125  {
 126      if (effectA.element != effectB.element)
 127          return false;
 128  
 129      var compare = Spry.Effect.Utils.optionsAreIdentical(effectA.options, effectB.options);
 130      // reset finish and setup functions

 131      if (compare)
 132      {
 133          if (typeof effectB.options.setup == 'function')
 134              effectA.options.setup = effectB.options.setup;
 135  
 136          if (typeof effectB.options.finish == 'function')
 137              effectA.options.finish = effectB.options.finish;
 138      }        
 139  
 140      return compare;
 141  };

 210  Spry.Effect.Utils.camelize = function(stringToCamelize)
 211  {
 212      if (stringToCamelize.indexOf('-') == -1){
 213          return stringToCamelize;    
 214      }
 215      var oStringList = stringToCamelize.split('-');
 216      var isFirstEntry = true;
 217      var camelizedString = '';
 218  
 219      for(var i=0; i < oStringList.length; i++)
 220      {
 221          if(oStringList[i].length>0)
 222          {
 223              if(isFirstEntry)
 224              {
 225                  camelizedString = oStringList[i];
 226                  isFirstEntry = false;
 227              }
 228              else
 229              {
 230                  var s = oStringList[i];
 231                  camelizedString += s.charAt(0).toUpperCase() + s.substring(1);
 232              }
 233          }
 234      }
 235  
 236      return camelizedString;

 309  Spry.Effect.Utils.optionsAreIdentical = function(optionsA, optionsB)
 310  {
 311      if(optionsA == null && optionsB == null)
 312          return true;
 313  
 314      if(optionsA != null && optionsB != null)
 315      {
 316          var objectCountA = 0;
 317          var objectCountB = 0;
 318  
 319          for (var propA in optionsA) objectCountA++;
 320          for (var propB in optionsB) objectCountB++;
 321  
 322          if(objectCountA != objectCountB)
 323              return false;
 324  
 325          for (var prop in optionsA)
 326          {
 327              var typeA = typeof optionsA[prop];
 328              var typeB = typeof optionsB[prop];
 329              if ( typeA != typeB || (typeA != 'undefined' && optionsA[prop] != optionsB[prop]))
 330                  return false;
 331          }
 332  
 333          return true;
 334      }
 335  
 336      return false;
 337  };

 367  Spry.Utils.Notifier.prototype.addObserver = function(observer)
 368  {
 369      if (!observer)
 370          return;
 371  
 372      // Make sure the observer isn't already on the list.

 373  
 374      var len = this.observers.length;
 375      for (var i = 0; i < len; i++)
 376          if (this.observers[i] == observer) return;
 377  
 378      this.observers[len] = observer;
 379  };
 380  
 381  Spry.Utils.Notifier.prototype.removeObserver = function(observer)
 382  {
 383      if (!observer)
 384          return;
 385  
 386      for (var i = 0; i < this.observers.length; i++)
 387      {
 388          if (this.observers[i] == observer)
 389          {
 390              this.observers.splice(i, 1);
 391              break;
 392          }
 393      }
 394  };

 396  Spry.Utils.Notifier.prototype.notifyObservers = function(methodName, data)
 397  {
 398      if (!methodName)
 399          return;
 400  
 401      if (!this.suppressNotifications)
 402      {
 403          var len = this.observers.length;
 404          for (var i = 0; i < len; i++)
 405          {
 406              var obs = this.observers[i];
 407              if (obs)
 408              {
 409                  if (typeof obs == "function")
 410                      obs(methodName, this, data);
 411                  else if (obs[methodName])
 412                      obs[methodName](this, data);
 413              }
 414          }
 415      }
 416  };

 451  {
 452      var value;
 453      var camelized = Spry.Effect.Utils.camelize(prop);
 454      try
 455      {
 456          if (element.style)
 457              value = element.style[camelized];
 458

 591  {
 592      var dimensions = new Spry.Effect.Utils.Rectangle;
 593      var computedStyle = null;
 594  
 595      if (element.style.width && /px/i.test(element.style.width))
 596          dimensions.width = parseInt(element.style.width, 10); // without padding

 597      else
 598      {
 599          computedStyle = Spry.Effect.getComputedStyle(element);
 600          var tryComputedStyle = computedStyle && computedStyle.width && /px/i.test(computedStyle.width);
 601  
 602          if (tryComputedStyle)
 603              dimensions.width = parseInt(computedStyle.width, 10); // without padding, includes css

 604  
 605          if (!tryComputedStyle || dimensions.width == 0) // otherwise we might run into problems on safari and opera (mac only)
 606              dimensions.width = element.offsetWidth;   // includes padding

 607      }
 608  
 609      if (element.style.height && /px/i.test(element.style.height))
 610          dimensions.height = parseInt(element.style.height, 10); // without padding

 611      else
 612      {
 613          if (!computedStyle)
 614              computedStyle = Spry.Effect.getComputedStyle(element);
 615  
 616          var tryComputedStyle = computedStyle && computedStyle.height && /px/i.test(computedStyle.height);
 617  
 618          if (tryComputedStyle)
 619              dimensions.height = parseInt(computedStyle.height, 10); // without padding, includes css

 620  
 621          if(!tryComputedStyle || dimensions.height == 0) // otherwise we might run into problems on safari and opera (mac only)
 622              dimensions.height = element.offsetHeight;   // includes padding

 623      }
 624      return dimensions;

1455  Spry.Effect.Cluster.prototype.drawEffect = function()
1456  {
1457      var isRunning = true;
1458      var allEffectsDidRun = false;
1459      var baseEffectIsStillRunning = false;
1460      var evalNextEffectsRunning = false;
1461  
1462      if ((this.currIdx == -1 && this.direction == Spry.forwards) || (this.currIdx == this.effectsArray.length && this.direction == Spry.backwards))
1463          this.initNextEffectsRunning();
1464  
1465      var start = this.direction == Spry.forwards ? 0 : this.effectsArray.length-1;
1466      var stop = this.direction == Spry.forwards ? this.effectsArray.length : -1;
1467      var step = this.direction == Spry.forwards ? 1 : -1;
1468      for (var i = start; i != stop; i+=step)
1469      {
1470          if (this.effectsArray[i].isRunning == true)
1471          {
1472              baseEffectIsStillRunning = this.effectsArray[i].effect.drawEffect();
1473              if (baseEffectIsStillRunning == false && i == this.currIdx)
1474              {
1475                  this.effectsArray[i].isRunning = false;
1476                  evalNextEffectsRunning = true;
1477              }
1478          }
1479      }
1480  
1481      if (evalNextEffectsRunning == true)
1482          allEffectsDidRun = this.initNextEffectsRunning();
1483  
1484      if (allEffectsDidRun == true) {
1485          this.stop();
1486          isRunning = false;
1487          for (var i = 0; i < this.effectsArray.length; i++)
1488              this.effectsArray[i].isRunning = false;
1489  
1490          this.currIdx = this.direction == Spry.forwards ? this.effectsArray.length: -1;
1491      }
1492      return isRunning;
