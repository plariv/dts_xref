  34  Spry.Widget.BrowserSniff = function()
  35  {
  36      var b = navigator.appName.toString();
  37      var up = navigator.platform.toString();
  38      var ua = navigator.userAgent.toString();
  39  
  40      this.mozilla = this.ie = this.opera = this.safari = false;
  41      var re_opera = /Opera.([0-9\.]*)/i;
  42      var re_msie = /MSIE.([0-9\.]*)/i;
  43      var re_gecko = /gecko/i;
  44      var re_safari = /(applewebkit|safari)\/([\d\.]*)/i;
  45      var r = false;
  46  
  47      if ( (r = ua.match(re_opera))) {
  48          this.opera = true;
  49          this.version = parseFloat(r[1]);
  50      } else if ( (r = ua.match(re_msie))) {
  51          this.ie = true;
  52          this.version = parseFloat(r[1]);
  53      } else if ( (r = ua.match(re_safari))) {
  54          this.safari = true;
  55          this.version = parseFloat(r[2]);
  56      } else if (ua.match(re_gecko)) {
  57          var re_gecko_version = /rv:\s*([0-9\.]+)/i;
  58          r = ua.match(re_gecko_version);
  59          this.mozilla = true;
  60          this.version = parseFloat(r[1]);
  61      }
  62      this.windows = this.mac = this.linux = false;
  63  
  64      this.Platform = ua.match(/windows/i) ? "windows" :
  65                      (ua.match(/linux/i) ? "linux" :
  66                      (ua.match(/mac/i) ? "mac" :
  67                      ua.match(/unix/i)? "unix" : "unknown"));
  68      this[this.Platform] = true;
  69      this.v = this.version;
  70  
  71      if (this.safari && this.mac && this.mozilla) {
  72          this.mozilla = false;
  73      }
  74  };

  80      
  81      options = Spry.Widget.Utils.firstValid(options, {});
  82      this.flags = {locked: false};
  83      this.options = {};
  84      this.element = element;
  85      this.init(element);
  86  
  87      if (!this.isBrowserSupported()){
  88          return;    
  89      }
  90  
  91      options.useCharacterMasking = Spry.Widget.Utils.firstValid(options.useCharacterMasking, true);
  92      options.hint = Spry.Widget.Utils.firstValid(options.hint, '');
  93      options.isRequired = Spry.Widget.Utils.firstValid(options.isRequired, true);
  94      options.additionalError = Spry.Widget.Utils.firstValid(options.additionalError, false);
  95  
  96      Spry.Widget.Utils.setOptions(this, options);
  97      Spry.Widget.Utils.setOptions(this.options, options);
  98  
  99      if (options.additionalError)
 100          this.additionalError = this.getElement(options.additionalError);
 101  
 102      //make sure we validate at least on submit

 103      var validateOn = ['submit'].concat(Spry.Widget.Utils.firstValid(this.options.validateOn, []));
 104      validateOn = validateOn.join(",");
 105      this.validateOn = 0;
 106      this.validateOn = this.validateOn | (validateOn.indexOf('submit') != -1 ? Spry.Widget.ValidationTextarea.ONSUBMIT : 0);
 107      this.validateOn = this.validateOn | (validateOn.indexOf('blur') != -1 ? Spry.Widget.ValidationTextarea.ONBLUR : 0);
 108      this.validateOn = this.validateOn | (validateOn.indexOf('change') != -1 ? Spry.Widget.ValidationTextarea.ONCHANGE : 0);
 109  
 110      if (Spry.Widget.ValidationTextarea.onloadDidFire){
 111          this.attachBehaviors();
 112      }else{
 113          Spry.Widget.ValidationTextarea.loadQueue.push(this);
 114      }

 227  Spry.Widget.ValidationTextarea.prototype.attachBehaviors = function()
 228  {
 229      if (this.element){
 230          if (this.element.nodeName == "TEXTAREA") {
 231              this.input = this.element;
 232          } else {
 233              this.input = Spry.Widget.Utils.getFirstChildWithNodeNameAtAnyLevel(this.element, "TEXTAREA");
 234          }
 235      }
 236      if (this.options && this.options.counterType && (this.options.counterType == 'chars_count' || this.options.counterType == 'chars_remaining')){
 237              this.counterEl = document.getElementById(this.options.counterId);
 238              this.counterChar();
 239      }
 240  
 241      if (this.input) {
 242          this.input.setAttribute("AutoComplete", "off");
 243          this.putHint();
 244          this.cursorPosition = new Spry.Widget.SelectionDescriptor(this.input);
 245  
 246          var self = this;
 247          this.event_handlers = [];
 248  
 249          //attach the pattern related event handlers (to stop invalid keys) 

 250          if (this.useCharacterMasking) {
 251              if (Spry.is.ie){
 252                  this.event_handlers.push([this.input, "propertychange", function(e) { return self.onKeyEvent(e || event); }]);
 253                  this.event_handlers.push([this.input, "drop", function(e) { return self.onDrop (e || event); }]);
 254                  this.event_handlers.push([this.input, "keypress", function(e) { return self.onKeyPress(e || event); }]);
 255              } else{
 256                  this.event_handlers.push([this.input, "keydown", function(e) { return self.onKeyDown(e); }]);
 257                  this.event_handlers.push([this.input, "keypress", function(e) { return self.safariKeyPress(e); }]);
 258                  this.event_handlers.push([this.input, "keyup", function(e) { return self.safariValidate(e); }]);
 259                  if (Spry.is.safari){
 260                      this.event_handlers.push([this.input, "mouseup", function(e) { return self.safariMouseUp(e); }]);
 261                      this.event_handlers.push([this.input, "mousedown", function(e) { return self.safariMouseDown(e); }]);
 262                  } else {
 263                      //Firefox bug: 355219

 264                      //this.event_handlers.push([this.input, "input", function(e) { self.onKeyEvent(e); return true;}]);

 265                      this.event_handlers.push([this.input, "dragdrop", function(e) { return self.onKeyEvent(e); }]);
 266                      this.event_handlers.push([this.input, "dragenter", function(e) { self.removeHint(); return self.onKeyDown(e); }]);
 267                      this.event_handlers.push([this.input, "dragexit", function(e) { return self.putHint(); }]);
 268                  }
 269              }
 270              // we need to save an initial state in case of invalid input

 271              this.event_handlers.push([this.input, "keydown", function(e) {return self.onKeyDown(e || event); }]);
 272          }
 273  
 274          this.event_handlers.push([this.input, "focus", function(e) { return self.onFocus(e || event); }]);
 275          this.event_handlers.push([this.input, "mousedown", function(e) { return self.onMouseDown(e || event); }]);
 276          this.event_handlers.push([this.input, "blur", function(e) { return self.onBlur(e || event); }]);
 277  
 278          if (this.validateOn & Spry.Widget.ValidationTextarea.ONCHANGE){
 279                  if (Spry.is.ie){
 280                          this.event_handlers.push([this.input, "propertychange", function(e) { return self.onChange(e || event); }]);
 281                          this.event_handlers.push([this.input, "drop", function(e) { return self.onChange(e || event); }]);
 282                  } else{
 283                          this.event_handlers.push([this.input, "keydown", function(e) { return self.onKeyDown(e); }]);
 284                          this.event_handlers.push([this.input, "keypress", function(e) { return self.safariChangeKeyPress(e); }]);
 285                          this.event_handlers.push([this.input, "keyup", function(e) { return self.safariChangeValidate(e); }]);
 286                          if (Spry.is.safari){
 287                              this.event_handlers.push([this.input, "mouseup", function(e) { return self.safariChangeMouseUp(e); }]);
 288                              this.event_handlers.push([this.input, "mousedown", function(e) { return self.safariMouseDown(e); }]);
 289                          } else {
 290                              // Firefox bug: 355219

 291                              //this.event_handlers.push([this.input, "input", function(e) { return self.onChange(e); }]);

 292                              this.event_handlers.push([this.input, "dragdrop", function(e) {return self.onChange(e); }]);
 293                              this.event_handlers.push([this.input, "dragenter", function(e) { self.removeHint(); return self.onKeyDown(e); }]);
 294                              this.event_handlers.push([this.input, "dragexit", function(e) { return self.putHint(); }]);
 295                          }
 296                  }
 297          }
 298          // The counter should be called directly when no enforcement or change restrictions exists

 299          if (! (this.validateOn & Spry.Widget.ValidationTextarea.ONCHANGE) && !this.useCharacterMasking){
 300                  if (Spry.is.ie){
 301                          this.event_handlers.push([this.input, "propertychange", function(e) { return self.counterChar(); }]);
 302                          this.event_handlers.push([this.input, "drop", function(e) { return self.counterChar(); }]);
 303                  } else{
 304                          this.event_handlers.push([this.input, "keypress", function(e) { return self.counterChar(); }]);
 305                          this.event_handlers.push([this.input, "keyup", function(e) { return self.counterChar(); }]);
 306                          if (Spry.is.safari){
 307                              this.event_handlers.push([this.input, "mouseup", function(e) { return self.counterChar(); }]);
 308                          } else {
 309                              // Firefox bug: 355219

 310                              //this.event_handlers.push([this.input, "input", function(e) { return self.onChange(e); }]);

 311                              this.event_handlers.push([this.input, "dragdrop", function(e) {return self.counterChar(); }]);
 312                          }
 313                  }
 314          }
 315  
 316          for (var i=0; i<this.event_handlers.length; i++) {
 317              Spry.Widget.Utils.addEventListener(this.event_handlers[i][0], this.event_handlers[i][1], this.event_handlers[i][2], false);
 318          }
 319  
 320          this.form = Spry.Widget.Utils.getFirstParentWithNodeName(this.input, "FORM");
 321          if (this.form) {
 322              if (!this.form.attachedSubmitHandler && !this.form.onsubmit) {
 323                  this.form.onsubmit = function(e) { e = e || event; return Spry.Widget.Form.onSubmit(e, e.srcElement || e.currentTarget) };
 324                  this.form.attachedSubmitHandler = true;                 
 325              }
 326              if (!this.form.attachedResetHandler) {
 327                  Spry.Widget.Utils.addEventListener(this.form, "reset", function(e) { e = e || event; return Spry.Widget.Form.onReset(e, e.srcElement || e.currentTarget) }, false);
 328                  this.form.attachedResetHandler = true;                 
 329              }
 330              // add the currrent widget to the "onSubmit" check queue;

 331              Spry.Widget.Form.onSubmitWidgetQueue.push(this);
 332          }
 333      }
 334      this.saveState();
 335  };

 337  Spry.Widget.ValidationTextarea.prototype.onTyping = function(e){
 338      if (this.input.disabled == true || this.input.readOnly == true){
 339              return;    
 340      }
 341  
 342      if (!this.initialCursor){
 343          this.initialCursor = this.cursorPosition;    
 344      }
 345      // on IE a stack overflow appears

 346      if (this.flags.locked){
 347              return true;
 348      }
 349  
 350      var val = this.input.value;
 351  
 352      var ret = true;
 353      
 354      if (this.flags.hintOn){
 355          return true;
 356      }
 357       if (e && this.input && this.options && this.options.maxChars > 0 && ret){
 358          if ( val.length > this.options.maxChars  && 
 359                              ((!Spry.Widget.Utils.isSpecialKey(e) && this.cursorPosition.start == this.cursorPosition.end) ||
 360                                (Spry.Widget.Utils.isSpecialKey(e) && val != this.initialValue) ||
 361                                   this.cursorPosition.start != this.cursorPosition.end)
 362               ){
 363                      // cut the extra chars and display error

 364                      this.flags.locked = true;
 365                      var initial = this.initialValue;
 366                      var start = this.initialCursor.start;
 367                      var end = this.initialCursor.end;
 368                      if (initial.length && this.initialCursor.end < initial.length) {
 369                              // we try to behave more like maxlength textfield

 370                              var tmp = end - start + this.options.maxChars - initial.length;
 371                              var newValue = initial.substring(0, start) + val.substring(start, start+tmp) + initial.substring(end, initial.length < this.options.maxChars ? initial.length:this.options.maxChars);
 372                              end = start + tmp;
 373                      }else{
 374                              var newValue = val.substring(0, this.options.maxChars);
 375                              end = start = this.options.maxChars;
 376                      }
 377                      if (Spry.is.ie) {
 378                          this.input.innerText = newValue;
 379                      } else {
 380                          this.input.value = newValue;
 381                      }
 382                      this.redTextFlash();
 383                      this.cursorPosition.moveTo(end, end);
 384                      this.flags.locked = false;
 385                      ret = false;
 386              } else{
 387                      this.setState(Spry.Widget.ValidationTextarea.VALID);
 388                      this.isMaxInvalid = false;
 389              }
 390      }
 391      this.counterChar();
 392      return ret;
 393  };

 395  Spry.Widget.ValidationTextarea.prototype.validateMinRequired = function(val){
 396      var oldInvalid = false;
 397      if (typeof this.notFireMinYet == 'undefined'){
 398          this.notFireMinYet = false;
 399      }else{
 400          oldInvalid = true;
 401          this.notFireMinYet = true;
 402      }
 403      if (this.onBlurOn){
 404          this.notFireMinYet = true;
 405      }else if (!this.onKeyEventOn){
 406          this.notFireMinYet = true;
 407      }
 408  
 409      if (this.input && this.options && this.options.isRequired){
 410              if (val.length > 0 && this.isRequireInvalid && (!this.hint || (this.hint && !this.flags.hintOn) || (this.hint && val != this.hint))){
 411                          this.switchClassName(this.validClass);
 412                          this.setState(Spry.Widget.ValidationTextarea.VALID);
 413                          this.isRequireInvalid = false;
 414              }else if ((val.length == 0 || !(!this.hint || (this.hint && !this.flags.hintOn) || (this.hint && val != this.hint))) && (!this.isRequireInvalid || oldInvalid)){
 415                          if (this.notFireMinYet || Spry.is.ie){
 416                              this.switchClassName(this.requiredClass);
 417                              this.setState(Spry.Widget.ValidationTextarea.REQUIRED);
 418                          }
 419                          this.isRequireInvalid = true;
 420                          this.isMinInvalid = false;
 421              }
 422      }
 423      if (this.input && this.options && this.options.minChars > 0 && !this.isRequireInvalid){
 424              if (val.length >= this.options.minChars && (!this.hint || (this.hint && !this.flags.hintOn) || (this.hint && val != this.hint)) && this.isMinInvalid){
 425                          this.switchClassName(this.validClass);
 426                          this.setState(Spry.Widget.ValidationTextarea.VALID);
 427                          this.isMinInvalid = false;
 428              }else if ( (val.length < this.options.minChars || (this.hint && val == this.hint && this.flags.hintOn)) && !this.isMinInvalid){
 429                          this.switchClassName(this.invalidCharsMinClass);
 430                          this.setState(Spry.Widget.ValidationTextarea.MINIMUM);
 431                          this.isMinInvalid = true;
 432              }
 433      }

 435  Spry.Widget.ValidationTextarea.prototype.counterChar = function(){
 436      if (!this.counterEl || !this.options || !this.options.counterType || (this.options.counterType != 'chars_remaining' && this.options.counterType != 'chars_count')){
 437          return;    
 438      }
 439  
 440      if (this.options.counterType == 'chars_remaining') {
 441          if (this.options.maxChars > 0){
 442              if (this.flags.hintOn){
 443                  this.setCounterElementValue(this.options.maxChars);
 444              } else {
 445                  if (this.options.maxChars > this.input.value.length){
 446                      this.setCounterElementValue(this.options.maxChars - this.input.value.length);
 447                  }else{
 448                      this.setCounterElementValue(0);
 449                  }
 450              }
 451          }
 452      } else {
 453          if (this.flags.hintOn){
 454              this.setCounterElementValue(0);
 455          } else {
 456              if (this.useCharacterMasking && typeof this.options.maxChars != 'undefined' && this.options.maxChars < this.input.value.length){
 457                  this.setCounterElementValue(this.options.maxChars);
 458              } else {
 459                  this.setCounterElementValue(this.input.value.length);
 460              }
 461          }
 462      }
 463  };

 484  Spry.Widget.ValidationTextarea.prototype.validate = function(){
 485      if (this.input.disabled == true || this.input.readOnly == true){
 486              return true;    
 487      }
 488  
 489    if (this.validateOn & Spry.Widget.ValidationTextarea.ONSUBMIT) {
 490      this.removeHint();
 491    }
 492    
 493      var val = this.input.value;
 494      this.validateMinRequired(val);
 495  
 496      var ret = !this.isMinInvalid && !this.isRequireInvalid;
 497  
 498      if (ret && this.options.maxChars > 0 && !this.useCharacterMasking){
 499              if (val.length <= this.options.maxChars || (this.hint && this.hint == val && this.flags.hintOn))    {
 500                      this.switchClassName(this.validClass);
 501                      this.setState(Spry.Widget.ValidationTextarea.VALID);
 502                    this.isMaxInvalid = false;
 503              }else{
 504                      this.switchClassName(this.invalidCharsMaxClass);
 505                      this.setState(Spry.Widget.ValidationTextarea.MAXIMUM);
 506                      this.isMaxInvalid = true;    
 507              }
 508      }
 509      ret = ret && !this.isMaxInvalid;
 510      if (ret) {
 511          this.switchClassName(this.validClass);
 512      }
 513      this.counterChar();    
 514      return ret;
 515  };

 823          if (this.element.nodeName == "TEXTAREA") {

 837                          range_all.moveStart('character', 1);

 842                      this.text = range.text;
