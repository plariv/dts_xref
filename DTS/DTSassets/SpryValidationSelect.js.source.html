  34  Spry.Widget.ValidationSelect = function(element, opts)
  35  {
  36      this.init(element);
  37  
  38      Spry.Widget.Utils.setOptions(this, opts);
  39  
  40      // set validateOn flags

  41      var validateOn = ['submit'].concat(this.validateOn || []);
  42      validateOn = validateOn.join(",");
  43      this.validateOn = 0 | (validateOn.indexOf('submit') != -1 ? Spry.Widget.ValidationSelect.ONSUBMIT : 0);
  44      this.validateOn = this.validateOn | (validateOn.indexOf('blur') != -1 ? Spry.Widget.ValidationSelect.ONBLUR : 0);
  45      this.validateOn = this.validateOn | (validateOn.indexOf('change') != -1 ? Spry.Widget.ValidationSelect.ONCHANGE : 0);
  46  
  47      if (this.additionalError)
  48          this.additionalError = this.getElement(this.additionalError);
  49  
  50      // Unfortunately in some browsers like Safari, the Stylesheets our

  51      // page depends on may not have been loaded at the time we are called.

  52      // This means we have to defer attaching our behaviors until after the

  53      // onload event fires, since some of our behaviors rely on dimensions

  54      // specified in the CSS.

  55  
  56      if (Spry.Widget.ValidationSelect.onloadDidFire)
  57          this.attachBehaviors();
  58      else 
  59          Spry.Widget.ValidationSelect.loadQueue.push(this);
  60  };

  88      // onChange and onBlur events are used to trigger validation

  93      if (this.event_handlers)
  94          for (var i=0; i<this.event_handlers.length; i++) {
  95              Spry.Widget.Utils.removeEventListener(this.event_handlers[i][0], this.event_handlers[i][1], this.event_handlers[i][2], false);

 146  Spry.Widget.ValidationSelect.prototype.attachBehaviors = function()
 147  {
 148      // find the SELECT element inside current container

 149      if (this.element.nodeName == "SELECT") {
 150          this.selectElement = this.element;
 151      } else {
 152          this.selectElement = Spry.Widget.Utils.getFirstChildWithNodeNameAtAnyLevel(this.element, "SELECT");
 153      }
 154  
 155      if (this.selectElement) {
 156          var self = this;
 157          this.event_handlers = [];
 158          // focus

 159          // attach on beforeactivate instead of focus for

 160          //      - IE 6 (to overcome this bug: setting a class name onfocus does not affect the open dropdown)

 161          //      - IE 7 (to overcome this bug: setting a class name, closes the select)

 162          var focusEventName = "focus";
 163          var ua = navigator.userAgent.match(/msie (\d+)\./i);
 164          if (ua) {
 165              ua = parseInt(ua[1], 10);
 166              if (ua >= 6) {
 167                  focusEventName = "beforeactivate";
 168              }
 169          }
 170          this.event_handlers.push([this.selectElement, focusEventName, function(e) { if (self.isDisabled()) return true; return self.onFocus(e); }]);
 171          // blur

 172          this.event_handlers.push([this.selectElement, "blur", function(e) { if (self.isDisabled()) return true; return self.onBlur(e); }]);
 173          // change

 174          if (this.validateOn & Spry.Widget.ValidationSelect.ONCHANGE) {
 175              this.event_handlers.push([this.selectElement, "change", function(e) { if (self.isDisabled()) return true; return self.onChange(e); }]);
 176              this.event_handlers.push([this.selectElement, "keypress", function(e) { if (self.isDisabled()) return true; return self.onChange(e); }]);
 177          }
 178  
 179          for (var i=0; i<this.event_handlers.length; i++) {
 180              Spry.Widget.Utils.addEventListener(this.event_handlers[i][0], this.event_handlers[i][1], this.event_handlers[i][2], false);
 181          }
 182  
 183          // submit

 184          this.form = Spry.Widget.Utils.getFirstParentWithNodeName(this.selectElement, "FORM");
 185          if (this.form) {
 186              // if no "onSubmit" handler has been attached to the current form, attach one

 187              if (!this.form.attachedSubmitHandler && !this.form.onsubmit) {
 188                  this.form.onsubmit = function(e) { e = e || event; return Spry.Widget.Form.onSubmit(e, e.srcElement || e.currentTarget) };
 189                  this.form.attachedSubmitHandler = true;                 
 190              }
 191              if (!this.form.attachedResetHandler) {
 192                  Spry.Widget.Utils.addEventListener(this.form, "reset", function(e) { e = e || event; return Spry.Widget.Form.onReset(e, e.srcElement || e.currentTarget) }, false);
 193                  this.form.attachedResetHandler = true;                 
 194              }
 195              // add the currrent widget to the "onSubmit" check queue;

 196              Spry.Widget.Form.onSubmitWidgetQueue.push(this);
 197          }
 198      }
 199  };

 254  Spry.Widget.ValidationSelect.prototype.validate = function() {
 255      this.reset();
 256      // check isRequired

 257      if (this.isRequired) {
 258          // there are no options, or no option has been selected

 259          if (this.selectElement.options.length == 0 || this.selectElement.selectedIndex == -1) {
 260              this.addClassName(this.element, this.requiredClass);
 261              this.addClassName(this.additionalError, this.requiredClass);
 262              return false;
 263          }
 264          // the current selected option has no "value" attribute

 265          // when no value is set, browsers implement different behaviour for the value property

 266          // IE: value = blank string ("")

 267          // FF, Opera: value = option text

 268          if (this.selectElement.options[this.selectElement.selectedIndex].getAttribute("value") == null) {
 269              this.addClassName(this.element, this.requiredClass);
 270              this.addClassName(this.additionalError, this.requiredClass);
 271              return false;
 272          }
 273          // the current selected option has blank string ("") value

 274          if (this.selectElement.options[this.selectElement.selectedIndex].value == this.emptyValue) {
 275              this.addClassName(this.element, this.requiredClass);
 276              this.addClassName(this.additionalError, this.requiredClass);
 277              return false;
 278          }
 279          // the current selected option has "disabled" attribute

 280          // IE 6 allows to select such options

 281          if (this.selectElement.options[this.selectElement.selectedIndex].disabled) {
 282              this.addClassName(this.element, this.requiredClass);
 283              this.addClassName(this.additionalError, this.requiredClass);
 284              return false;
 285          }
 286      }
 287      if (this.invalidValue) {
 288          if (this.selectElement.options.length > 0 && 
 289              this.selectElement.selectedIndex != -1 &&
 290              this.selectElement.options[this.selectElement.selectedIndex].value == this.invalidValue) {
 291              this.addClassName(this.element, this.invalidClass);
 292              this.addClassName(this.additionalError, this.invalidClass);
 293              return false;
 294          }
 295      }
 296      this.addClassName(this.element, this.validClass);
 297      this.addClassName(this.additionalError, this.validClass);
 298      return true;
 299  };

 410  Spry.Widget.Utils.getFirstParentWithNodeName = function(node, nodeName)
 411  {
 412      while (node.parentNode
 413              && node.parentNode.nodeName.toLowerCase() != nodeName.toLowerCase()
 414              && node.parentNode.nodeName != 'BODY') {
 415          node = node.parentNode;
 416      }
 417  
 418      if (node.parentNode && node.parentNode.nodeName.toLowerCase() == nodeName.toLowerCase()) {
 419          return node.parentNode;
 420      } else {
 421          return null;
 422      }

 440  Spry.Widget.Utils.contains = function (who, what)
 441  {
 442      if (typeof who.contains == 'object') {
 443          return what && who && (who == what || who.contains(what));
 444      } else {
 445          var el = what;
 446          while(el) {
 447              if (el == who) {
 448                  return true;
 449              }
 450              el = el.parentNode;
 451          }
 452          return false;
 453      }
 454  };
