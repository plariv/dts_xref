 172      // escape commas (,)

 173  	function esc( $text )
 174      {
 175          // we replace the comma so it is not URL escaped

 176          // if it is, flash just thinks it is a comma

 177          // which is no good if we are splitting the

 178          // string on commas.

 179          $tmp = str_replace( ',', '#comma#', $text );
 180          //$tmp = utf8_encode( $tmp );

 181          // now we urlescape all dodgy characters (like & % $ etc..)

 182          return urlencode( $tmp );
 183      }
