  58  function formatTime( sval:String )
  59  {
  60      var val:Number = parseFloat(sval);
  61      var hours:Number = Math.floor(val);
  62      var minutes:Number = Math.round((val - hours) * 60);
  63      var rval:String = "" + hours;
  64      if (rval.length < 2) {
  65          rval = "0" + rval;
  66      }
  67      if (minutes < 10) {
  68          rval = rval + ":0" + minutes;
  69      } else {
  70          rval = rval + ":" + minutes;
  71      }
  72      return rval;
  73  }

 255  function show_tip( owner:Object, x:Number, y:Number, tip_obj:Object )
 256  {
 257      if( ( _root.tooltip != undefined ) )
 258      {
 259          if(_root.tooltip._owner==owner)
 260              return;    // <-- it's our tooltip and it is showing

 261          else
 262              removeMovieClip("tooltip");    // <-- it is someone elses tootlip - remove it

 263      }
 264      
 265      var tmp:String;
 266      var lines:Array = [];
 267      //

 268      // Dirty hack. Takes a tool_tip_wrapper, and replaces the #val# with the

 269      // tool_tip text, so noew you can do: "My Val = $#val#%", which turns into:

 270      // "My Val = $12.00%"

 271      //

 272      if( _root.tool_tip_wrapper != undefined )
 273      {
 274          tmp = _root.tool_tip_wrapper.replace('#val#',tip_obj.value);
 275          tmp = tmp.replace('#key#',tip_obj.key);
 276          tmp = tmp.replace('#x_label#',tip_obj.x_label);
 277          tmp = tmp.replace('#val:time#',formatTime(tip_obj.value));
 278          
 279          if( _root._x_legend != undefined )
 280              tmp = tmp.replace('#x_legend#',_root._x_legend.get_legend());
 281      }
 282      else
 283      {
 284          if( tip_obj.x_label == undefined )
 285              tmp = tip_obj.value;
 286          else
 287              tmp = tip_obj.x_label+'<br>'+tip_obj.value;
 288      }
 289          
 290      lines = tmp.split( '<br>' );
 291      
 292      var tooltip = _root.createEmptyMovieClip( "tooltip", this.getNextHighestDepth() );
 293          
 294      // let the tooltip know who owns it, else we get weird race conditions where one

 295      // bar has onRollOver fired, then another has onRollOut and deletes the tooltip

 296      tooltip._owner = owner;
 297  
 298      tooltip.createTextField( "txt_title", tooltip.getNextHighestDepth(), 5, 5, 100, 100);
 299      if( lines.length > 1 )
 300          tooltip.txt_title.text = lines.shift();
 301  
 302      var fmt:TextFormat = new TextFormat();
 303      fmt.color = 0x0000F0;
 304      fmt.font = "Verdana";
 305      
 306      // this needs to be an option:

 307      fmt.bold = true;
 308      fmt.size = 12;
 309      fmt.align = "right";
 310      tooltip.txt_title.setTextFormat(fmt);
 311      tooltip.txt_title.autoSize="left";
 312      
 313      tooltip.createTextField( "txt", tooltip.getNextHighestDepth(), 5, tooltip.txt_title._height, 100, 100);
 314      
 315      tooltip.txt.text = lines.join( '\n' );
 316      
 317      var fmt2:TextFormat = new TextFormat();
 318      fmt2.color = 0x000000;
 319      fmt2.font = "Verdana";
 320      fmt2.size = 12;
 321      fmt2.align = "left";
 322      tooltip.txt.setTextFormat(fmt2);
 323      tooltip.txt.autoSize="left";
 324  
 325      var max_width:Number = Math.max( tooltip.txt_title._width, tooltip.txt._width );
 326      var y_pos:Number = y - tooltip.txt_title._height - tooltip.txt._height;
 327      
 328      if( y_pos < 0 )
 329      {
 330          // the tooltip has drifted off the top of the screen, move it down:

 331          y_pos = y + tooltip.txt_title._height + tooltip.txt._height;
 332      }
 333      
 334      var cstroke = {width:2, color:0x808080, alpha:100};
 335      var ccolor = {color:0xf0f0f0, alpha:100};
 336  
 337      ChartUtil.rrectangle(
 338          tooltip,
 339          max_width+10,
 340          tooltip.txt_title._height + tooltip.txt._height + 5,
 341          6,
 342          ((x+max_width+16) > Stage.width ) ? (Stage.width-max_width-16) : x,
 343          y_pos,
 344          cstroke,
 345          ccolor);
 346  
 347      // NetVicious, June, 2007

 348      // create shadow filter

 349      var dropShadow = new flash.filters.DropShadowFilter();
 350      dropShadow.blurX = 4;
 351      dropShadow.blurY = 4;
 352      dropShadow.distance = 4;
 353      dropShadow.angle = 45;
 354      dropShadow.quality = 2;
 355      dropShadow.alpha = 0.5;
 356      // apply shadow filter

 357      tooltip.filters = [dropShadow];
 358  
 359  }

 408  function mouse_move()
 409  {
 410      if( _root.chartValues == undefined )
 411          return;
 412      
 413      // is the mouse over the invisible layer?

 414      if( !_root._inv.hitTest(_root._xmouse, _root._ymouse) )
 415          return;
 416      
 417      //

 418      mouse_over( true );
 419      
 420      // get closest points from each data set

 421      var closest:Array = _root.get_closest();
 422  
 423      // find closest point along X axis

 424      var min:Number = Number.MAX_VALUE;
 425      for( var i:Number=0; i < closest.length; i++ )
 426          min = Math.min( min, closest[i].distance_x );
 427      
 428      //for( var i:Number=0; i < closest.length; i++ )

 429      //    trace( i +'  '+closest[i].distance_x +'  '+ closest[i].distance_y );

 430      
 431      // now select all points that are the same distance

 432      // along the X axis

 433      var xx:Object = {point:null, distance_x:Number.MAX_VALUE, distance_y:Number.MAX_VALUE };
 434      for( var i:Number=0; i < closest.length; i++ )
 435      {
 436          if( closest[i].distance_x == min )
 437          {
 438              // these share the same X position, so choose

 439              // the closest to the mouse in the Y

 440              if( closest[i].distance_y < xx.distance_y )
 441                  xx = closest[i];
 442          }
 443      }
 444  
 445      _root.tooltip_x.draw( xx.point );
 446      xx.point.is_tip = true;
 447      
 448      // make the line dot nice and big, does

 449      // nothing for bars

 450      for( var i:Number=0; i < _root.chartValues.styles.length; i++)
 451          _root.chartValues.styles[i].highlight_value();
 452  }
