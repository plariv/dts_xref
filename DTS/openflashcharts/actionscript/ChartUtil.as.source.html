  81  	static function Lighten( col:Number ) : Number
  82      {
  83          var rgb = col; //decimal value for a purple color

  84          var red = (rgb & 16711680) >> 16; //extacts the red channel

  85          var green = (rgb & 65280) >> 8; //extacts the green channel

  86          var blue = rgb & 255; //extacts the blue channel

  87          var p=2;
  88          red += red/p;
  89          if( red > 255 )
  90              red = 255;
  91              
  92          green += green/p;
  93          if( green > 255 )
  94              green = 255;
  95              
  96          blue += blue/p;
  97          if( blue > 255 )
  98              blue = 255;
  99              
 100          return red << 16 | green << 8 | blue;
 101      }

 117  	static function rrectangle( mc:MovieClip, w:Number, h:Number, rad:Number, x:Number, y:Number, stroke:Object, fill:Object) : Void {
 118          // added by JG on 30th May 07

 119          x = Math.round(x);
 120          y = Math.round(y);
 121          w = Math.round(w);
 122          h = Math.round(h);
 123          //

 124          mc.lineStyle(stroke.width, stroke.color, stroke.alpha);
 125          mc.beginFill(fill.color, fill.alpha);
 126          mc.moveTo(0+rad, 0);
 127          mc.lineTo(w-rad, 0);
 128          mc.curveTo(w, 0, w, rad);
 129          mc.lineTo(w, h-rad);
 130          mc.curveTo(w, h, w-rad, h);
 131          mc.lineTo(0+rad, h);
 132          mc.curveTo(0, h, 0, h-rad);
 133          mc.lineTo(0, 0+rad);
 134          mc.curveTo(0, 0, 0+rad, 0);
 135          mc.endFill();
 136          mc._x = x;
 137          mc._y = y;
 138      }
