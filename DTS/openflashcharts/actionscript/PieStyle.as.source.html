 155  	public function draw( top:Number ) : Void
 156      {
 157          this.clear_mcs( Stage.width/2, ((Stage.height-top)/2)+top );
 158          
 159          //radius for the pie

 160          //var rad:Number = (Stage.width<(Stage.height-top-60)) ? Stage.width/2 : (Stage.height-top-60)/2;

 161          var rad:Number = (Stage.width<Stage.height) ? Stage.width/2 : (Stage.height-top)/2;
 162          var labelLineSize:Number = rad+this.label_line;
 163          
 164          if( this.labels.length>0 && this.style.get( 'display' ) != 'none' )
 165          {
 166              this.init_labels();
 167              
 168              var tfs:Array = Array();
 169              
 170              // CSS style is NOT none, so create the text field objects

 171              for( var i:Number=0; i < this.ExPoints.length; i++ )
 172                  tfs.push( this.create_label( i, this.labels[i] ) );
 173                  
 174              //

 175              // start off with the radius at 100%, then keep shrinking it untill

 176              // all the labels fit into the Stage.

 177              
 178              // 2007-11-14 modified by veljac99:

 179              //  - impproved algorithm for finding radius which reduces number of itterations ( from 24 to 6  - using data-13.txt )

 180              
 181              
 182              var radMax:Number = rad;                      // maximum is 100%%

 183              var radMin:Number = rad * 0.1;                // minimum is  10%

 184              var radTest:Number = (radMax + radMin) * 0.7; // assume 70% (instead of usual 50%)

 185              labelLineSize = radTest+this.label_line;
 186              
 187              //tollerance - stop caclulations if we are inside acceptable tollerance (here is 2% with minimum 2 points)

 188              var tollerance:Number = radMax * 0.02;
 189              if (tollerance<2)
 190                  tollerance=2; //2 percent tollerance but minimum 2

 191              var iterations:Number=0;  // to avoid endless loop - allow only 30 itterations

 192              
 193              var outside:Boolean = false;
 194              do
 195              {
 196                  iterations +=1;
 197  
 198                  for( var i:Number=0; i < tfs.length; i++ )
 199                  {
 200                      var angle:Number = this.ExPoints[i].bar_bottom+this.ExPoints[i].bar_width/2;
 201                      outside = outside || this.move_label( tfs[i], labelLineSize, this.pie_mcs[i]._x, this.pie_mcs[i]._y, angle );
 202                  }
 203                  
 204                  //found? Great. Go out!

 205                  if ( (radMax - radMin)<= tollerance || iterations>30 /*30 = max itterationa*/ ){
 206                      rad = radTest;
 207                      trace( 'break' );
 208                      trace( "rad: " + rad + " iterations: " + iterations );
 209                      trace('--');
 210                      break;
 211                  }
 212                  
 213                  //not found - adopt and try again

 214                  //                trace ( "outside: " + outside + 

 215                  //                    " [" + radMin + " - " + radMax + "] radTest:" + radTest

 216                  //                    + "  iterations: " + iterations

 217                  //                    );

 218                  
 219                  
 220                  if (outside) {
 221                      radMax = radTest;
 222                  } else {
 223                      radMin = radTest;
 224                  }
 225                  radTest = (radMax + radMin)/2;
 226                  
 227                  labelLineSize = radTest+this.label_line;
 228                  outside = false;
 229                  
 230              }while( true );
 231              
 232          } //end if( this.labels.length>0 )

 233          
 234          this.draw_all(rad);

 309  	function move_label( tf:TextField, rad:Number, x:Number, y:Number, ang:Number )
 310      {
 311          //text field position

 312          var legend_x:Number = x+rad*Math.cos((ang)*3.6*TO_RADIANS);
 313          var legend_y:Number = y+rad*Math.sin((ang)*3.6*TO_RADIANS);
 314          
 315          //if legend stands to the right side of the pie

 316          if(legend_x<x)
 317              legend_x -= tf._width;
 318                  
 319          //if legend stands on upper half of the pie

 320          if(legend_y<y)
 321              legend_y -= tf._height;
 322          
 323          tf._x = legend_x;
 324          tf._y = legend_y;
 325          
 326          // is this label outside the stage?

 327          if( (tf._x>0) && (tf._y>0) && (tf._y+tf._height<Stage.height ) && (tf._x+tf._width<Stage.width) )
 328              return false;
 329          else
 330              return true;

 333  	function draw_slice( pieSlice:MovieClip, r1:Number, color:Number, slice_angle:Number )
 334      {
 335          //line from center to edge

 336          pieSlice.lineStyle( this.border_width, this.colour, 100 );
 337  
 338          //if the user selected the charts to be gradient filled do gradients

 339          if( this.gradientFill == 'true' )
 340          {
 341              //set gradient fill

 342              var colors:Array = [color, color];
 343              var alphas:Array = [100, 50];
 344              var ratios:Array = [100,255];
 345              var matrix:Object = {a:r1*2, b:0, c:50, d:0, e:r1*2, f:0, g:-3, h:3, i:1};
 346              pieSlice.beginGradientFill("radial", colors, alphas, ratios, matrix);
 347          }
 348          else
 349              pieSlice.beginFill(color, 100);
 350          
 351          pieSlice.moveTo(0, 0);
 352          pieSlice.lineTo(r1, 0);
 353      
 354          
 355          var angle:Number = 4;
 356          var a:Number = Math.tan((angle/2)*TO_RADIANS);
 357          
 358          var i:Number = 0;
 359          //draw curve segments spaced by angle 

 360          for( i=0; i+angle < slice_angle*3.6; i+=angle) {
 361              var endx:Number = r1*Math.cos((i+angle)*TO_RADIANS);
 362              var endy:Number = r1*Math.sin((i+angle)*TO_RADIANS);
 363              var ax:Number = endx+r1*a*Math.cos(((i+angle)-90)*TO_RADIANS);
 364              var ay:Number = endy+r1*a*Math.sin(((i+angle)-90)*TO_RADIANS);
 365              pieSlice.curveTo(ax, ay, endx, endy);    
 366          }
 367          
 368          //when aproaching end of slice, refine angle interval

 369          var angle:Number = 0.08;
 370          var a:Number = Math.tan((angle/2)*TO_RADIANS);
 371           
 372          for ( ; i+angle < slice_angle*3.6; i+=angle) {
 373              var endx:Number = r1*Math.cos((i+angle)*TO_RADIANS);
 374              var endy:Number = r1*Math.sin((i+angle)*TO_RADIANS);
 375              var ax:Number = endx+r1*a*Math.cos(((i+angle)-90)*TO_RADIANS);
 376              var ay:Number = endy+r1*a*Math.sin(((i+angle)-90)*TO_RADIANS);
 377              pieSlice.curveTo(ax, ay, endx, endy);    
 378          }
 379          
 380          //close slice

 381          pieSlice.endFill();
 382          pieSlice.lineTo(0,0);
 383      }
