  82  	function jiggle( left:Number, right:Number, x_label_width:Number, count:Number )
  83      {
  84          var r:Number = 0;
  85          
  86          if( x_label_width != 0 )
  87          {
  88              var item_width:Number = (right-left) / count;
  89              var r:Number = right-(item_width/2);
  90              var new_right:Number = right;
  91              
  92              // while the right most X label is off the edge of the

  93              // Stage, move the box.right - 1

  94              while( r+(x_label_width/2) > right )
  95              {
  96                  new_right -= 1;
  97                  // changing the right also changes the item_width:

  98                  item_width = (new_right-left) / count;
  99                  r = new_right-(item_width/2);
 100              }
 101              right = new_right;
 102          }
 103          
 104          return right;

 247      {
 248
