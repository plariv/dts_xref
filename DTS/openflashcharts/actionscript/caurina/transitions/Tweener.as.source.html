  73       *

  74       * @param        (first-n param)        Object                Object that should be tweened: a movieclip, textfield, etc.. OR an array of objects

  75       * @param        (last param)        Object                Object containing the specified parameters in any order, as well as the properties that should be tweened and their values

  76       * @param        .time                Number                Time in seconds or frames for the tweening to take (defaults 2)

  78       * @param        .useFrames            Boolean                Whether to use frames instead of seconds for time control (defaults false)

  79       * @param        .transition            String/Function        Type of transition equation... (defaults to "easeoutexpo")

  80       * @param        .onStart            Function            * Direct property, See the TweenListObj class

  81       * @param        .onUpdate            Function            * Direct property, See the TweenListObj class

  82       * @param        .onComplete            Function            * Direct property, See the TweenListObj class

  83       * @param        .onOverwrite        Function            * Direct property, See the TweenListObj class

  84       * @param        .onStartParams        Array                * Direct property, See the TweenListObj class

  85       * @param        .onUpdateParams        Array                * Direct property, See the TweenListObj class

  86       * @param        .onCompleteParams    Array                * Direct property, See the TweenListObj class

  88       * @param        .rounded            Boolean                * Direct property, See the TweenListObj class

  92  	public static function addTween ():Boolean {
  93          if (arguments.length < 2 || arguments[0] == undefined) return false;
  94  
  95          var rScopes:Array = new Array(); // List of objects to tween

  96          var i:Number, j:Number, istr:String, jstr:String;
  97  
  98          if (arguments[0] instanceof Array) {
  99              // The first argument is an array

 100              for (i = 0; i<arguments[0].length; i++) rScopes.push(arguments[0][i]);
 101          } else {
 102              // The first argument(s) is(are) object(s)

 103              for (i = 0; i<arguments.length-1; i++) rScopes.push(arguments[i]);
 104          }
 105          // rScopes = arguments.concat().splice(1); // Alternate (should be tested for speed later)

 106  
 107          // make properties chain ("inheritance")

 108          var p_obj:Object = TweenListObj.makePropertiesChain(arguments[arguments.length-1]);
 109  
 110          // Creates the main engine if it isn't active

 111          if (!_inited) init();
 112          if (!_engineExists || _root[getControllerName()] == undefined) startEngine(); // Quick fix for Flash not resetting the vars on double ctrl+enter...

 113  
 114          // Creates a "safer", more strict tweening object

 115          var rTime:Number = (isNaN(p_obj.time) ? 0 : p_obj.time); // Real time

 116          var rDelay:Number = (isNaN(p_obj.delay) ? 0 : p_obj.delay); // Real delay

 117  
 118          // Creates the property list; everything that isn't a hardcoded variable

 119          var rProperties:Object = new Object(); // Object containing objects [name:{.valueStart, .valueComplete }]

 120          var restrictedWords:Object = {time:true, delay:true, useFrames:true, skipUpdates:true, transition:true, onStart:true, onUpdate:true, onComplete:true, onOverwrite:true, rounded:true, onStartParams:true, onUpdateParams:true, onCompleteParams:true, onOverwriteParams:true, quickAdd:true};
 121          var modifiedProperties:Object = new Object();
 122          for (istr in p_obj) {
 123              if (!restrictedWords[istr]) {
 124                  // It's an additional pair, so adds

 125                  if (_specialPropertySplitterList[istr] != undefined) {
 126                      // Special property splitter

 127                      var splitProperties:Array = _specialPropertySplitterList[istr].splitValues(p_obj[istr], _specialPropertySplitterList[istr].parameters);
 128                      for (i = 0; i < splitProperties.length; i++) {
 129                          rProperties[splitProperties[i].name] = {valueStart:undefined, valueComplete:splitProperties[i].value};
 130                      }
 131                  } else if (_specialPropertyModifierList[istr] != undefined) {
 132                      // Special property modifier

 133                      var tempModifiedProperties:Array = _specialPropertyModifierList[istr].modifyValues(p_obj[istr]);
 134                      for (i = 0; i < tempModifiedProperties.length; i++) {
 135                          modifiedProperties[tempModifiedProperties[i].name] = {modifierParameters:tempModifiedProperties[i].parameters, modifierFunction:_specialPropertyModifierList[istr].getValue};
 136                      }
 137                  } else {
 138                      // Regular property or special property, just add the property normally

 139                      rProperties[istr] = {valueStart:undefined, valueComplete:p_obj[istr]};
 140                  }
 141              }
 142          }
 143  
 144          // Adds the modifiers to the list of properties

 145          for (istr in modifiedProperties) {
 146              if (rProperties[istr] != undefined) {
 147                  rProperties[istr].modifierParameters = modifiedProperties[istr].modifierParameters;
 148                  rProperties[istr].modifierFunction = modifiedProperties[istr].modifierFunction;
 149              }
 150              
 151          }
 152          
 153          var rTransition:Function; // Real transition

 154  
 155          if (typeof p_obj.transition == "string") {
 156              // String parameter, transition names

 157              var trans:String = p_obj.transition.toLowerCase();
 158              rTransition = _transitionList[trans];
 159          } else {
 160              // Proper transition function

 161              rTransition = p_obj.transition;
 162          }
 163          if (rTransition == undefined) rTransition = _transitionList["easeoutexpo"];
 164  
 165          var nProperties:Object;
 166          var nTween:TweenListObj;
 167          var myT:Number;
 168  
 169          for (i = 0; i < rScopes.length; i++) {
 170              // Makes a copy of the properties

 171              nProperties = new Object();
 172              for (istr in rProperties) {
 173                  nProperties[istr] = new PropertyInfoObj(rProperties[istr].valueStart, rProperties[istr].valueComplete, rProperties[istr].modifierFunction, rProperties[istr].modifierParameters);
 174              }
 175  
 176              nTween = new TweenListObj(
 177                  /* scope            */    rScopes[i],
 178                  /* timeStart        */    _currentTime + ((rDelay * 1000) / _timeScale),
 179                  /* timeComplete        */    _currentTime + (((rDelay * 1000) + (rTime * 1000)) / _timeScale),
 180                  /* useFrames        */    (p_obj.useFrames == true),
 181                  /* transition        */    rTransition
 182              );
 183  
 184              nTween.properties            =    nProperties;
 185              nTween.onStart                =    p_obj.onStart;
 186              nTween.onUpdate                =    p_obj.onUpdate;
 187              nTween.onComplete            =    p_obj.onComplete;
 188              nTween.onOverwrite            =    p_obj.onOverwrite;
 189              nTween.onError              =   p_obj.onError;
 190              nTween.onStartParams        =    p_obj.onStartParams;
 191              nTween.onUpdateParams        =    p_obj.onUpdateParams;
 192              nTween.onCompleteParams        =    p_obj.onCompleteParams;
 193              nTween.onOverwriteParams    =    p_obj.onOverwriteParams;
 194              nTween.rounded                =    p_obj.rounded;
 195              nTween.skipUpdates            =    p_obj.skipUpdates;
 196  
 197              // Remove other tweenings that occur at the same time

 198              if (!p_obj.quickAdd) removeTweensByTime(nTween.scope, nTween.properties, nTween.timeStart, nTween.timeComplete);
 199  
 200              // And finally adds it to the list

 201              _tweenList.push(nTween);
 202  
 203              // Immediate update and removal if it's an immediate tween -- if not deleted, it executes at the end of this frame execution

 204              if (rTime == 0 && rDelay == 0) {
 205                  myT = _tweenList.length-1;
 206                  updateTweenByIndex(myT);
 207                  removeTweenByIndex(myT);
 208              }
 209          }
 210  
 211          return true;
 212      }

 307      /**

 308       * Remove an specified tweening of a specified object the tweening list, if it conflicts with the given time

 309       *

 310       * @param        p_scope                Object                        List of objects affected

 311       * @param        p_properties        Object                         List of properties affected (PropertyInfoObj instances)

 312       * @param        p_timeStart            Number                        Time when the new tween starts

 313       * @param        p_timeComplete        Number                        Time when the new tween ends

 314       * @return                            Boolean                        Whether or not it actually deleted something

 315       */
 316  	public static function removeTweensByTime (p_scope:Object, p_properties:Object, p_timeStart:Number, p_timeComplete:Number):Boolean {
 317          var removed:Boolean = false;
 318          var removedLocally:Boolean;
 319  
 320          var i:Number;
 321          var tl:Number = _tweenList.length;
 322          var pName:String;
 323  
 324          for (i = 0; i < tl; i++) {
 325              if (p_scope == _tweenList[i].scope) {
 326                  // Same object...

 327                  if (p_timeComplete > _tweenList[i].timeStart && p_timeStart < _tweenList[i].timeComplete) {
 328                      // New time should override the old one...

 329                      removedLocally = false;
 330                      for (pName in _tweenList[i].properties) {
 331                          if (p_properties[pName] != undefined) {
 332                              // Same object, same property

 333                              // Finally, remove this old tweening and use the new one

 334                              if (_tweenList[i].onOverwrite != undefined) {
 335                                  try {
 336                                      _tweenList[i].onOverwrite.apply(_tweenList[i].scope, _tweenList[i].onOverwriteParams);
 337                                  } catch(e:Error) {
 338                                      handleError(_tweenList[i], e, "onOverwrite");
 339                                  }
 340                              }
 341                              _tweenList[i].properties[pName] = undefined;
 342                              delete _tweenList[i].properties[pName];
 343                              removedLocally = true;
 344                              removed = true;
 345                          }
 346                      }
 347                      if (removedLocally) {
 348                          // Verify if this can be deleted

 349                          if (AuxFunctions.getObjectLength(_tweenList[i].properties) == 0) removeTweenByIndex(i);
 350                      }
 351                  }
 352              }
 353          }
 354  
 355          return removed;
 356      }

 457      /**

 458       * Do some generic action on specific tweenings (pause, resume, remove, more?)

 459       *

 460       * @param        p_function            Function    Function to run on the tweenings that match

 461       * @param        p_scope                Object        Object that must have its tweens affected by the function

 462       * @param        p_properties        Array        Array of strings that must be affected

 463       * @return                            Boolean        Whether or not it successfully affected something

 464       */
 465  	private static function affectTweens (p_affectFunction:Function, p_scope:Object, p_properties:Array):Boolean {
 466          var affected:Boolean = false;
 467          var i:Number;
 468  
 469          if (!_tweenList) return false;
 470  
 471          for (i = 0; i < _tweenList.length; i++) {
 472              if (_tweenList[i].scope == p_scope) {
 473                  if (p_properties.length == 0) {
 474                      // Can affect everything

 475                      p_affectFunction(i);
 476                      affected = true;
 477                  } else {
 478                      // Must check whether this tween must have specific properties affected

 479                      var affectedProperties:Array = new Array();
 480                      var j:Number;
 481                      for (j = 0; j < p_properties.length; j++) {
 482                          if (_tweenList[i].properties[p_properties[j]] != undefined) {
 483                              affectedProperties.push(p_properties[j]);
 484                          }
 485                      }
 486                      if (affectedProperties.length > 0) {
 487                          // This tween has some properties that need to be affected

 488                          var objectProperties:Number = AuxFunctions.getObjectLength(_tweenList[i].properties);
 489                          if (objectProperties == affectedProperties.length) {
 490                              // The list of properties is the same as all properties, so affect it all

 491                              p_affectFunction(i);
 492                              affected = true;
 493                          } else {
 494                              // The properties are mixed, so split the tween and affect only certain specific properties

 495                              var slicedTweenIndex:Number = splitTweens(i, affectedProperties);
 496                              p_affectFunction(slicedTweenIndex);
 497                              affected = true;
 498                          }
 499                      }
 500                  }
 501              }
 502          }
 503          return affected;
 504      }

 506      /**

 507       * Splits a tweening in two

 508       *

 509       * @param        p_tween                Number        Object that must have its tweens split

 510       * @param        p_properties        Array        Array of strings containing the list of properties that must be separated

 511       * @return                            Number        The index number of the new tween

 512       */
 513  	public static function splitTweens (p_tween:Number, p_properties:Array):Number {
 514          // First, duplicates

 515          var originalTween:TweenListObj = _tweenList[p_tween];
 516          var newTween:TweenListObj = originalTween.clone(false);
 517  
 518          // Now, removes tweenings where needed

 519          var i:Number;
 520          var pName:String;
 521  
 522          // Removes the specified properties from the old one

 523          for (i = 0; i < p_properties.length; i++) {
 524              pName = p_properties[i];
 525              if (originalTween.properties[pName] != undefined) {
 526                  originalTween.properties[pName] = undefined;
 527                  delete originalTween.properties[pName];
 528              }
 529          }
 530  
 531          // Removes the unspecified properties from the new one

 532          var found:Boolean;
 533          for (pName in newTween.properties) {
 534              found = false;
 535              for (i = 0; i < p_properties.length; i++) {
 536                  if (p_properties[i] == pName) {
 537                      found = true;
 538                      break;
 539                  }
 540              }
 541              if (!found) {
 542                  newTween.properties[pName] = undefined;
 543                  delete newTween.properties[pName];
 544              }
 545          }
 546  
 547          // If there are empty property lists, a cleanup is done on the next updateTweens() cycle

 548          _tweenList.push(newTween);
 549          return (_tweenList.length - 1);
 550          
 551      }

 558       * Updates all existing tweenings

 559       *

 560       * @return                            Boolean        FALSE if no update was made because there's no tweening (even delayed ones)

 561       */
 562  	private static function updateTweens ():Boolean {
 563          if (_tweenList.length == 0) return false;
 564          var i:Number;
 565          for (i = 0; i < _tweenList.length; i++) {
 566              // Looping throught each Tweening and updating the values accordingly

 567              if (!_tweenList[i].isPaused) {
 568                  if (!updateTweenByIndex(i)) removeTweenByIndex(i);
 569                  if (_tweenList[i] == null) {
 570                      removeTweenByIndex(i, true);
 571                      i--;
 572                  }
 573              }
 574          }
 575  
 576          return true;
 577      }

 623      /**

 624       * Updates an specific tween

 625       *

 626       * @param        i                    Number        Index (from the tween list) of the tween that should be updated

 627       * @return                            Boolean        FALSE if it's already finished and should be deleted, TRUE if otherwise

 628       */
 629  	private static function updateTweenByIndex (i:Number):Boolean {
 630  
 631          var tTweening:Object = _tweenList[i];    // Shortcut to this tweening

 632  
 633          if (tTweening == null || !tTweening.scope) return false;
 634  
 635          var isOver:Boolean = false;        // Whether or not it's over the update time

 636          var mustUpdate:Boolean;            // Whether or not it should be updated (skipped if false)

 637  
 638          var nv:Number;                    // New value for each property

 639  
 640          var t:Number;                    // current time (frames, seconds)

 641          var b:Number;                    // beginning value

 642          var c:Number;                    // change in value

 643          var d:Number;                     // duration (frames, seconds)

 644  
 645          var pName:String;                // Property name, used in loops

 646  
 647          // Shortcut stuff for speed

 648          var tScope:Object;                // Current scope

 649          var tProperty:Object;            // Property being checked

 650  
 651          if (_currentTime >= tTweening.timeStart) {
 652              // Can already start

 653  
 654              tScope = tTweening.scope;
 655  
 656              if (tTweening.isCaller) {
 657                  // It's a 'caller' tween

 658                  do {
 659                      t = ((tTweening.timeComplete - tTweening.timeStart)/tTweening.count) * (tTweening.timesCalled+1);
 660                      b = tTweening.timeStart;
 661                      c = tTweening.timeComplete - tTweening.timeStart;
 662                      d = tTweening.timeComplete - tTweening.timeStart;
 663                      nv = tTweening.transition(t, b, c, d);
 664  
 665                      if (_currentTime >= nv) {
 666                          if (tTweening.onUpdate != undefined) {
 667                              try {
 668                                  tTweening.onUpdate.apply(tScope, tTweening.onUpdateParams);
 669                              } catch(e:Error) {
 670                                  handleError(tTweening, e, "onUpdate");
 671                              }
 672                          }
 673  
 674                          tTweening.timesCalled++;
 675                          if (tTweening.timesCalled >= tTweening.count) {
 676                              isOver = true;
 677                              break;
 678                          }
 679                          if (tTweening.waitFrames) break;
 680                      }
 681  
 682                  } while (_currentTime >= nv);
 683              } else {
 684                  // It's a normal transition tween

 685  
 686                  mustUpdate = tTweening.skipUpdates < 1 || tTweening.skipUpdates == undefined || tTweening.updatesSkipped >= tTweening.skipUpdates;
 687  
 688                  if (_currentTime >= tTweening.timeComplete) {
 689                      isOver = true;
 690                      mustUpdate = true;
 691                  }
 692  
 693                  if (!tTweening.hasStarted) {
 694                      // First update, read all default values (for proper filter tweening)

 695                      if (tTweening.onStart != undefined) {
 696                          try {
 697                              tTweening.onStart.apply(tScope, tTweening.onStartParams);
 698                          } catch(e:Error) {
 699                              handleError(tTweening, e, "onStart");
 700                          }
 701                      }
 702                      for (pName in tTweening.properties) {
 703                          var pv:Number = getPropertyValue (tScope, pName);
 704                          tTweening.properties[pName].valueStart = isNaN(pv) ? tTweening.properties[pName].valueComplete : pv;
 705                      }
 706                      mustUpdate = true;
 707                      tTweening.hasStarted = true;
 708                  }
 709  
 710                  if (mustUpdate) {
 711                      for (pName in tTweening.properties) {
 712                          tProperty = tTweening.properties[pName];
 713  
 714                          if (isOver) {
 715                              // Tweening time has finished, just set it to the final value

 716                              nv = tProperty.valueComplete;
 717                          } else {
 718                              if (tProperty.hasModifier) {
 719                                  // Modified

 720                                  t = _currentTime - tTweening.timeStart;
 721                                  d = tTweening.timeComplete - tTweening.timeStart;
 722                                  nv = tTweening.transition(t, 0, 1, d);
 723                                  nv = tProperty.modifierFunction(tProperty.valueStart, tProperty.valueComplete, nv, tProperty.modifierParameters);
 724                              } else {
 725                                  // Normal update

 726                                  t = _currentTime - tTweening.timeStart;
 727                                  b = tProperty.valueStart;
 728                                  c = tProperty.valueComplete - tProperty.valueStart;
 729                                  d = tTweening.timeComplete - tTweening.timeStart;
 730                                  nv = tTweening.transition(t, b, c, d);
 731                              }
 732                          }
 733  
 734                          if (tTweening.rounded) nv = Math.round(nv);
 735                          setPropertyValue(tScope, pName, nv);
 736                      }
 737  
 738                      tTweening.updatesSkipped = 0;
 739  
 740                      if (tTweening.onUpdate != undefined) {
 741                          try {
 742                              tTweening.onUpdate.apply(tScope, tTweening.onUpdateParams);
 743                          } catch(e:Error) {
 744                              handleError(tTweening, e, "onUpdate");
 745                          }
 746                      }
 747                  } else {
 748                      tTweening.updatesSkipped++;
 749                  }
 750              }
 751  
 752              if (isOver && tTweening.onComplete != undefined) {
 753                  try {
 754                      tTweening.onComplete.apply(tScope, tTweening.onCompleteParams);
 755                  } catch(e:Error) {
 756                      handleError(tTweening, e, "onComplete");
 757                  }
 758              }
 759  
 760              return (!isOver);
 761          }
 762  
 763          // On delay, hasn't started, so return true

 764          return (true);
 765  
 766      }
