 204  	public function closest( x:Number, y:Number )
 205      {
 206          var shortest:Number = Number.MAX_VALUE;
 207          var ex:PointBar = null;
 208          
 209          for( var i:Number=0; i < this.ExPoints.length; i++)
 210          {
 211              this.ExPoints[i].is_tip = false;
 212              
 213              if( (x > this.ExPoints[i].x) && (x < this.ExPoints[i].x+this.ExPoints[i].width) )
 214              {
 215                  // mouse is in position 1

 216                  shortest = Math.min( Math.abs( x - this.ExPoints[i].x ), Math.abs( x - (this.ExPoints[i].x+this.ExPoints[i].width) ) );
 217                  ex = this.ExPoints[i];
 218                  break;
 219              }
 220              else
 221              {
 222                  // mouse is in position 2

 223                  // get distance to left side and right side

 224                  var d1:Number = Math.abs( x - this.ExPoints[i].x );
 225                  var d2:Number = Math.abs( x - (this.ExPoints[i].x+this.ExPoints[i].width) );
 226                  var min:Number = Math.min( d1, d2 );
 227                  if( min < shortest )
 228                  {
 229                      shortest = min;
 230                      ex = this.ExPoints[i];
 231                  }
 232              }
 233          }
 234          var dy:Number = Math.abs( y - ex.y );
 235          
 236          return { point:ex, distance_x:shortest, distance_y:dy };
 237      }
