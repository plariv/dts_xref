 161  	public function closest( x:Number, y:Number )
 162      {
 163          //

 164          // because this is a scatter chart, we may have

 165          // many items for the same X axis value, so we 

 166          // keep them all, then find the closest to the

 167          // Y position (see data-32.txt for a test)

 168          //

 169          var shortest:Number = Number.MAX_VALUE;
 170          
 171          // find the closest point(s) in X

 172          for( var i:Number=0; i < this.ExPoints.length; i++)
 173          {
 174              this.ExPoints[i].is_tip = false;
 175              
 176              var dx:Number = Math.abs( x - this.ExPoints[i].x );
 177          
 178              if( dx < shortest )
 179                  shortest = dx;
 180          }
 181          
 182          var points:Array = Array();
 183          // get all the points at this X distance

 184          for( var i:Number=0; i < this.ExPoints.length; i++)
 185          {
 186              var dx:Number = Math.abs( x - this.ExPoints[i].x );
 187          
 188              if( dx == shortest )
 189                  points.push( this.ExPoints[i] );
 190          }
 191          var dist_x = shortest;
 192          
 193          //

 194          // find the closest in the Y

 195          //

 196          shortest = Number.MAX_VALUE;
 197          var point:Point = null;
 198          
 199          for( var i:Number=0; i < points.length; i++)
 200          {
 201              var dy:Number = Math.abs( y - points[i].y );
 202              
 203              if( dy < shortest ) 
 204              {
 205                  shortest = dy;
 206                  point = points[i];
 207              }
 208          }
 209          
 210          var dy:Number = Math.abs( y - point.y );
 211          return { point:point, distance_x:dist_x, distance_y:dy };
 212      }
