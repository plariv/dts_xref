  52      $('.html_editor_check').click(function(){
  53          var html = $(this).parent().parent().find('.html_editor').val();
  54  
  55          // Define common regexes
  56          var starts_with_chaff = /^[^<>]+/;
  57          var ends_with_chaff = /[^<>]+$/;
  58          // parsing the opening_tag regex: 
  59              // at the start of the string
  60              // opening tag 
  61              // 1+ letters (captured)
  62              // 0 or more of:
  63                  // a space followed by
  64                  // 1+ letters followed by
  65                  // an = sign followed by
  66                  // either a single-quoted or double-quoted alphanumeric string,
  67                  // which can also contain . and /
  68              // ending with a closing bracket
  69          var opening_tag = /^<(\w+\d?)(\s+\w+=('[\w\d\s\.\/%]+'|"[\w\d\s\.\/%]+"))*>/;
  70          var self_contained_tag = /^<(\w+\d?)(\s+\w+=('[\w\d\s\.\/%]+'|"[\w\d\s\.\/%]+"))*\s?\/>/;
  71  
  72          var error_short = "";
  73          var error_full = "";
  74  
  75          // Strategy: look at each piece of HTML and, if it's understandable and valid, remove it. Keep going until nothing is left, or you hit something you don't understand.
  76          while (error_full == "") {
  77              // alert(html);
  78  
  79              // nothing to parse? 
  80              if (html == "") { error_full = "none"; }
  81              // if there's non-tags at the beginning or end of the string, cut them out
  82              else if (html.search(starts_with_chaff) >= 0 || html.search(ends_with_chaff) >= 0) {
  83                  html = html.replace(starts_with_chaff, '').replace(ends_with_chaff, '');
  84              }
  85              // if an opening tag starts the string
  86              else if (html.search(opening_tag) >= 0) {
  87                  // get the opening tag
  88                  var tag_full = html.match(opening_tag)[0];
  89                  var tag_text = html.match(opening_tag)[1];
  90                  var this_closing_tag = new RegExp("</"+tag_text+">");
  91                  // the matching closing tag should be at the end of the string.
  92                  if (html.search(this_closing_tag) == -1) {
  93                      error_short = "Missing a closing tag: </"+tag_text+">";
  94                      error_full = "I expected to find a closing tag </"+tag_text+">.";
  95                  }
  96                  // remove both tags. Should only do so once.
  97                  html = html.replace(opening_tag, '').replace(this_closing_tag, '');
  98              } 
  99              // if the string starts with a self-contained tag
 100              else if (html.search(self_contained_tag) >= 0) {
 101                  html = html.replace(self_contained_tag, '');
 102              } 
 103              // Otherwise, the parser doesn't understand the next bit of HTML.
 104              else {
 105                  error_short = "Invalid html near '"+html.substring(0, 40)+"'";
 106                  error_full = "It looks like you have invalid HTML near the text '"+html.substring(0, 40)+"'.";
 107              }
 108  
 109          }
 110          
 111          // display error message, if any
 112          if (error_full == "none") {
 113              $(this).parent().find('.html_editor_validity').text('Valid!').css('color', 'green');
 114          } else {
 115              alert(error_full);
 116              $(this).parent().find('.html_editor_validity').text(error_short).css('color', 'red');
 117          } 
 118  
 119      });
 120  
 121  });
