  16  function SQLstr($theValue, $theType, $theDefinedValue = "", $theNotDefinedValue = "") 
  17  {
  18      // The only thing SQLstr is useful for, in comparison to GetSQLValueString(), is that the name is shorter and the function is kept on a centralized include.

  19      // Sanitize input to prevent SQL injection attacks. 

  20      // All SQL-input locations should run through this function.

  21      // It took me a while to realize that Dreamweaver's GetSQLValueString already did indeed sanitize inputs. But it does so. 

  22    $theValue = function_exists("mysql_real_escape_string") ? 
  23              mysql_real_escape_string($theValue) : mysql_escape_string($theValue);
  24  
  25    switch ($theType) {
  26      case "text":
  27        $theValue = ($theValue != "") ? "'" . $theValue . "'" : "NULL";
  28        break;    
  29      case "long":
  30      case "int":
  31        $theValue = ($theValue != "") ? intval($theValue) : "NULL";
  32        break;
  33      case "double":
  34        $theValue = ($theValue != "") ? doubleval($theValue) : "NULL";
  35        break;
  36      case "date":
  37        $theValue = ($theValue != "") ? "'" . $theValue . "'" : "NULL";
  38        break;
  39      case "defined":
  40        $theValue = ($theValue != "") ? $theDefinedValue : $theNotDefinedValue;
  41        break;
  42    }
  43    return $theValue;
  44  }

 159  function randomPassword($pw_length = 8) { 
 160      // * depends on sendEmail to alert us if errors are found.

 161      // Also depends on query functionality, which most of my other functions do not do.

 162      global $database_Assessment, $Assessment;
 163      
 164      
 165      //                                     Generate a random password

 166          
 167      // by default, all passwords are 8 characters in length.

 168      // the pw will have between 1 and (length-1) alphabetic characters,

 169      $char_length = mt_rand(1, $pw_length - 1);
 170      // and the rest will be numbers.

 171      $num_length = $pw_length - $char_length;
 172      // except there might be an initial capital letter,

 173      $capital = "";
 174      if (mt_rand(0,1) == 1) {
 175          $capital_pool = "ABCDEFGHNJKLMNOPQRSTUVWXYZ";
 176          $capital = $capital_pool[mt_rand(0, 25)];
 177          $char_length --;
 178      }
 179      // and there also might be a final punctuation mark.

 180      $punc = "";
 181      if (mt_rand(0,1) == 1) {
 182          $punc_pool = "!@#$%^&*?-=+";
 183          $punc = $punc_pool[mt_rand(0, 11)];
 184          $num_length --;
 185      }
 186      // choose the random letters (weighted towards the more familiar ones)

 187      $chars = "";
 188      while ($char_length > 0) {
 189          $char_pool = "abcdefghijkmnopqrstuvwxyzabcdefghijkmnop";
 190          $chars .= $char_pool[mt_rand(0, 39)];
 191          $char_length --;
 192      }
 193      // and choose the random numbers

 194      $nums = "";
 195      while ($num_length > 0) {
 196          $nums .= "".mt_rand(0, 9);
 197          $num_length --;
 198      }
 199      // and put them all together

 200      $pw = "";
 201      $pw .= $capital;
 202      // with either the letters first or the numbers first

 203      if (mt_rand(0,1) == 1) { $pw .= $chars.$nums; } else { $pw .= $nums.$chars; }
 204      $pw .= $punc;
 205      
 206      // Example passwords produced in this way include: 

 207      // 496757zi, X2451whu, 023fknfn, M8ixiue

 208      
 209      /* This method, by my estimate, should produce passwords of roughly 35 bit-strength.

 210         (possible capital letter) + (unknown order of (unknown length of chars) + 

 211         (unknown length of nums) ) + (possible punctuation mark) = ?

 212         36 * (~18^6) * 23 */
 213      
 214      
 215      //                                     Check if this password already exists in Registrants

 216          
 217      $query_rsPw = sprintf("SELECT userkey, email, user_name, password 
 218      FROM registrants WHERE password = %s", 
 219              SQLstr($pw, "text"));
 220      $rsPw = mysql_query($query_rsPw, $Assessment) or die(mysql_error());
 221      $row_rsPw = mysql_fetch_assoc($rsPw);
 222      $totalRows_rsPw = mysql_num_rows($rsPw);
 223      
 224      // If this password already exists in our system, then freak out.

 225      if ($totalRows_rsPw > 0) {
 226          $email_to = "service@lectica.org";
 227          $email_from = "service@lectica.org";
 228          $email_cc = "";
 229          $subject = "ALERT - duplicate random password found in DTS.registrants table";
 230          $body = "Warning!! The PHP random password generator on the password reset page has generated a password which already exists in the DTS registrants database. The offending string is: '".$pw."' Please check to ensure that your PW generator algorithm is up to snuff. This is an automated email sent by dtsservice.";
 231          sendEmail($email_to, $email_from, $email_cc, $subject, $body);
 232          return false;
 233      }
 234      
 235      return $pw;

 413       - You may optionally specify how you would like the table to sort by default, either the column name or the column name and sort order. Examples: "group" or "finalized DESC" etc.

 506                  $thisCondition = " AND ".$thisSearchCol;
 507                  // finish the clause differently depending on the appropriate comparison.

 508                  if ($thisSearchVal == "NULL") { $thisCondition .= " IS NULL "; } 
 509                  else if ($comparison == "LIKE") { $thisCondition .= " LIKE '%".$thisSearchVal."%' "; }
 510                  else { $thisCondition .= " ".$comparison." '".$thisSearchVal."' "; }
 511                  // echo "Adding search term: ".$thisCondition." <br>"; // <-- Uncomment to troubleshoot

 512                  // add this condition to the WHERE or HAVING clauses, depending on $location

 513                  if ($location == "WHERE") { $searchClause .= $thisCondition; } 
 514                  else if ($location == "HAVING") { $havingClause .= $thisCondition; }
 515              } 
 516          } while (next($sessionSearch));
 517          // produces something like: " AND userkey LIKE '%1050%' AND date_completed >= '2011-01-01' "

 518          // echo "searchClause = ".$searchClause."<br />havingClause = ".$havingClause."<br />"; // **** troubleshoot

 519      }
 520      
 521      
 522      // Add sort clause to query, if that sort column is valid for this query.

 523      if (isset($_SESSION['sort']) AND $_SESSION['sort'] <> "" AND $sortActive) {
 524          // Get the sort column name & sort direction

 525          $sortPairing = explode(":", $_SESSION['sort']);
 526          $sortCol = $sortPairing[0];
 527          $sortDir = $sortPairing[1];
 528          
 529          // Skim through validColumns (list of columns acceptable for this query) and see if there is any match with this column name. If that match has some table-specific label on it (such as d. ), then make $thisSearchCol match that exact column name.

 530          // echo "search for the sorting column ".$sortCol.": "; // <-- Uncomment for troubleshooting

 531          
 532          do {
 533              // IGNORE any words that have a parentheses in them, as these indicate SQL functions 

 534              // and could cause an error with the sort clause.

 535              
 536              // echo "| ".current($validColumns)." "; // <-- Uncomment for troubleshooting

 537              
 538              if (strpos(" ".current($validColumns), "(") OR strpos(" ".current($validColumns), ")")) 
 539              { next($validColumns); }
 540              // IF: 1) this sort column is matched exactly, 

 541              if (current($validColumns) == $sortCol) {  
 542                  // The direction of sort was determined in initializeSortSearch().

 543                  $sortClause = " ORDER BY ".current($validColumns)." ".$sortDir." "; 
 544              }
 545              // OR 2) this sort column is matched with a table prefix and '.', 

 546              else if (substr( current($validColumns), strpos(" ".current($validColumns), ".") ) == $sortCol) {
 547                  // write the WHOLE column phrase into sortCol. (NOT just the matching part, include the table prefix!)

 548                  $sortCol = current($validColumns);
 549                  $sortClause = " ORDER BY ".$sortCol." ".$sortDir." ";
 550              }
 551          } while (next($validColumns)); 
 552          reset($validColumns);
 553          
 554           // topher();    // <-- Uncomment for troubleshooting

 555      }
 556      // By default, sort by the default sort column / ordering, if a default was set for this query.

 557      if ($sortClause == "" AND $defaultSort <> "") {
 558          $sortClause = "ORDER BY ".$defaultSort." ";
 559      }
 560      
 561      // The searchClause will usually be part of the WHERE clause, rather than the HAVING clause. 

 562      // There are multiple considerations in this choice:

 563      //  - I think WHERE clause limitations can improve efficiency by narrowing the conditions during, rather than after, the search.

 564      //  - WHERE clause conditions can reference columns that aren't called in the SELECT field, whereas the HAVING clause requires referenced columns to be specified in the SELECT field and will generate an error whenever they aren't present. 

 565      //  - WHERE clause conditions cannot reference aliases of columns, nor can they reference the result of calculations in the SELECT field. HAVING clause conditions operate upon the results of the query up to that point, so they can reference calculations and column aliases. 

 566      //  - Adding more text onto the WHERE clause means that no space is left for a "GROUP BY" clause, unless I figure out a way to pass it through this sort/search function. 

 567       
 568      // If sortActive = 2, then omit the search clause. 

 569      // If a GROUP clause is present, the search clause should go just before it.

 570      // If there were any HAVING conditions, they go just before the sort. 

 571      return " ".($sortActive == 2 ? "" : $searchClause)." ".
 572          ($groupBy <> "" ? " GROUP BY ".$groupBy : "")." ".
 573          ($havingClause <> "" ? " HAVING 1=1 ".$havingClause : "")." ".
 574          $sortClause." "; 
 575  }
