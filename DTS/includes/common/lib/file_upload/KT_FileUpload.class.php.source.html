 197      /**

 198       *     Handle the uploaded file by moving it to a destination file.

 199       * The destination file = folder + fileName

 200       * @param string fileName     the name for saving the uploaded file;

 201       * @param string     $oldFileName    the previous file name, or null on insert

 202       * @return string file name if succeded or null if not;

 203       * @access public

 204       */
 205  	function uploadFile($fileName, $oldFileName="")
 206      {
 207          if ($this->hasError()) {
 208              return;
 209          }
 210          $this->checkUpload();
 211          $this->checkFolder();
 212          $this->checkSize();
 213          $this->checkExtensions();
 214          $this->checkFileName($fileName);
 215          if ($this->hasError()) {
 216              return;
 217          }
 218  
 219          if ($this->fileExists) {
 220              $folder = KT_realpath($this->folder);
 221              $fileName = KT_replaceSpecialChars($fileName, 'filter');
 222              $destinationName = KT_realpath($folder . $fileName, false);
 223              if (file_exists($destinationName)) {
 224                  if (strtolower($fileName) != strtolower($oldFileName)) {
 225                      // if the destination file really exists

 226                      if (!$this->autoRename) {
 227                          $this->setError('UPLOAD_FILE_EXISTS', array(), array($fileName));
 228                          return;
 229                      } else {
 230                          $destinationName = $this->getTempName($destinationName);
 231                      }
 232                  }
 233              }
 234              
 235              if ($oldFileName!='') {
 236                  @unlink($folder . DIRECTORY_SEPARATOR . $oldFileName);
 237              }
 238  
 239              if (!@move_uploaded_file($this->fileInfo['tmp_name'], $destinationName)) {
 240                  unlink($this->fileInfo['tmp_name']);
 241                  $this->setError('PHP_UPLOAD_MOVE_TMP_ERROR', array(), array());
 242                  return;
 243              } else {
 244                  $arr = split("[\\/]", $destinationName);
 245                  $this->destinationName = $destinationName;
 246                  KT_setFilePermissions($this->destinationName);
 247                  return array_pop($arr);
 248              }
 249          }

 253      /**

 254       * Gets a temporary file name starting from the given file name.

 255       * Ex: filename = 'zone.jpg', and zone_1.jpg alreagy exists, the new file name is zone_2.jpg.

 256       * @param string $filename the filename on which to create the new temporary file name.

 257       * @return string the temporary file name;

 258       * @access public

 259       */
 260  	function getTempName($filename) {
 261          $pos = strrpos($filename,'.');
 262          if ($pos !== false) {
 263              $tmpName = substr($filename,0,$pos);
 264              $tmpExt = substr($filename, $pos);
 265          } else {
 266              $tmpName = $filename;
 267              $tmpExt = '';
 268          }
 269          $i=1;
 270          while (file_exists($tmpName.'_'.$i.$tmpExt) && $i < 1000) {
 271              $i++;
 272          }
 273          $retVar = $tmpName.'_'.$i.$tmpExt;
 274          return $retVar;
 275      }

 359      /**

 360       * Checks if the upload has performed OK and if the file is required.

 361       * @return nothing;

 362       * @access public

 363       */
 364  	function checkUpload() {
 365          if ($this->fileInfo['tmp_name'] == '') {
 366              $this->fileExists = false;
 367              if (isset($this->fileInfo['error']) && !is_array($this->fileInfo['error']) && $this->fileInfo['error'] >0) {
 368                  switch ($this->fileInfo['error']){
 369                      case 1:
 370                          $this->setError('PHP_UPLOAD_ERR_INI_SIZE', array(), array());
 371                          return;
 372                      case 2:
 373                          $this->setError('PHP_UPLOAD_ERR_FORM_SIZE', array(), array());
 374                          return;
 375                      case 3:
 376                          $this->setError('PHP_UPLOAD_ERR_PARTIAL', array(), array());
 377                          return;
 378                      case 4:
 379                          break;
 380                      case 6:
 381                          $this->setError('PHP_UPLOAD_ERR_NO_TMP_DIR', array(), array());
 382                          return;
 383                      case 7:
 384                          $this->setError('PHP_UPLOAD_ERR_CANT_WRITE', array(), array());
 385                          return;
 386                      default:
 387                          $this->setError('PHP_UPLOAD_ERR_UNKNOWN', array(), array($this->fileInfo['error']));
 388                          return;
 389                  }
 390              } else {
 391                  $this->setError('PHP_UPLOAD_ERR_TMP_DIR', array(), array());
 392              }
 393          } else {
 394              $this->fileExists = true;
 395              if (isset($this->fileInfo['size']) && $this->fileInfo['size'] == 0) {
 396                  $this->fileExists = false;
 397                  $this->setError('UPLOAD_CHECK_SIZE_S', array($this->minSize-1), array($this->minSize-1));
 398              }
 399          }
 400          
 401          if (!is_uploaded_file($this->fileInfo['tmp_name'])) {
 402              if($this->isRequired) {
 403                  if ($this->fileInfo['tmp_name'] != ''){
 404                      $this->setError('PHP_UPLOAD_ERR_TMP_FILE', array(), array($this->fileInfo['tmp_name']));
 405                      return;
 406                  } elseif ($this->fileInfo['error'] == 4) {
 407                      $this->setError('PHP_UPLOAD_FILE_REQUIRED', array(), array());
 408                      return;
 409                  }
 410              }
 411          }
 412      }
