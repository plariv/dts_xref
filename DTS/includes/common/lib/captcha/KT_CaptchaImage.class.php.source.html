 192      * garbage collector; delete all images older than 10 minutes;

 193      * @return nothing

 195  	function garbageCollector() {
 196          if (!file_exists(KT_CAPTCHA_TEMP_FOLDER)) {
 197              mkdir(KT_CAPTCHA_TEMP_FOLDER);
 198              return;
 199          }
 200          $d = dir(KT_CAPTCHA_TEMP_FOLDER); 
 201          while (false !== ($entry = $d->read())) { 
 202              if ( is_file(KT_CAPTCHA_TEMP_FOLDER . $entry) && filemtime(KT_CAPTCHA_TEMP_FOLDER . $entry) < (time() - 60*10) ) {
 203                  @unlink(KT_CAPTCHA_TEMP_FOLDER . $entry);
 204              }
 205          } 
 206          $d->close(); 
 207      }
