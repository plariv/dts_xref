 109      /**

 110       * read the content of the folder.

 111       * @param string  $folder         - the absolute path to the folder

 112       * @param boolean $details         - Will return in the array the size for each entry (0 size for any folder).

 113       * @return array  [array] - Returns: array with the listing of the folder/files, empty if error occured;

 114       * @access public

 115       */
 116  	function readFolder($folder, $details=false)
 117      {
 118          if (!$this->checkRights($folder, 'read')) {
 119              $this->setError('PHP_FOLDER_READ_ERR', array(), array($folder));
 120              return array();
 121          }
 122          
 123          $arrDir = array();
 124          $arrFil = array();
 125          $folder = $this->preparePath($folder);
 126          $folder = KT_realpath($folder);
 127          $dir = dir($folder);
 128          while (false !== ($entry = $dir->read())) {
 129              if ($entry != '.' && $entry != '..') {
 130                  if (is_dir($folder . $entry)) {
 131                      $arrDir[] = array('name'=>$entry, 'size'=>0);    
 132                  } else {
 133                      if (@filesize($dir->path . $entry)===false) {
 134                          $arrFil[] = array('name'=>$entry, 'size'=>($details==true ? 0 :0 ));
 135                      } else {
 136                          $arrFil[] = array('name'=>$entry, 'size'=>($details==true ? filesize($dir->path . $entry) :0 ));
 137                      }
 138                  }    
 139              }      
 140          } 
 141          $dir->close();
 142          return array('files'=>$arrFil, 'folders'=>$arrDir);
 143      }

 146       * create recursively the folder;

 147       * @param string $path  the absolute path to the folder

 148       * @return nothing

 149       * @access public

 150       */
 151  	function createFolder($path)
 152      {
 153          $path = $this->preparePath($path);
 154          $path = KT_realpath($path);
 155          
 156          $arrCreate = array();
 157          $arrPath = split("[\\/]", $path);
 158          if ($arrPath[count($arrPath)-1]=='') {
 159              array_pop($arrPath);
 160          }
 161          
 162          while (!file_exists(implode(DIRECTORY_SEPARATOR, $arrPath)) && count($arrPath)>0) {
 163              $arrCreate[] = array_pop($arrPath);
 164          }
 165          
 166          if (count($arrCreate)>0) {
 167              $arrCreate = array_reverse($arrCreate);
 168              $folder = implode(DIRECTORY_SEPARATOR, $arrPath);
 169              
 170              foreach ($arrCreate as $key => $dir) {
 171                  $folder .= DIRECTORY_SEPARATOR .$dir;
 172                  @mkdir($folder);
 173                  KT_setFilePermissions($folder,true);
 174                  if (!$this->is_writable($folder)) {
 175                      $this->setError('PHP_FOLDER_CREATE_ERR', array(), array($folder));
 176                      return;
 177                  }
 178              }
 179          }
 180      }

 190          if (!$this->checkRights($folder, 'write')) {

 192              return;

 198          while (false!==($entry = $d->read())) {

 200                  continue;
 201              }
 202              if (is_dir($d->path . $entry)) {
 203                  $this->deleteFolder($d->path . $entry);
 204              } else {
 205                  $ret = @unlink($folder . $entry);

 207                      $this->setError('PHP_FOLDER_DELETE_ERR', array(), array($folder . $entry));

 209              }

 211          $d->close();
 212          $ret = @rmdir($folder);
 213          if ($ret === false) {
 214              $this->setError('PHP_FOLDER_DELETE_ERR', array(), array($folder));

 300       * @param string $oldName old folder name

 302       * @return nothing

 303       * @access public

 305  	function renameFolder($folder, $oldName, $newName) {
 306          $folder = KT_realpath($folder);
 307          if (!$this->checkRights($folder, 'write')) {
 308              $this->setError('PHP_FOLDER_RENAME_RIGHTS', array(), array($folder));
 309              return;
 310          }
 311          
 312          if (!file_exists($folder.$oldName)) {
 313              $this->setError('PHP_FOLDER_RENAME_NO_FILE', array(), array($folder.$oldName));
 314              return;
 315          }
 316          if (file_exists($folder.$newName)) {
 317              $this->setError('PHP_FOLDER_RENAME_EXISTS', array(), array($folder.$newName));
 318              return;
 319          }
 320          if (!@rename($folder.$oldName, $folder.$newName)) {
 321              $this->setError('PHP_FOLDER_RENAME', array(), array($folder.$oldName, $folder.$newName));
 322          }
 323      }

 325      /**

 326       * copy a folder

 327       * @param string $folder the path to the folder

 328       * @param string $parentFolder the parent of the copied folder

 329       * @return nothing

 330       * @access public

 331       */
 332  	function copyFolder($folder, $parentFolder) {
 333          $folder = $this->preparePath($folder);
 334          $folder = KT_realpath($folder);
 335          $parentFolder = $this->preparePath($parentFolder);
 336          $parentFolder = KT_realpath($parentFolder);
 337          if (!$this->checkRights($folder, 'read')) {
 338              $this->setError('PHP_FOLDER_COPY_RIGHTS', array(), array($folder));
 339              return;
 340          }
 341          if (!$this->checkRights($parentFolder, 'write')) {
 342              $this->setError('PHP_FOLDER_COPY_RIGHTS', array(), array($parentFolder));
 343              return;
 344          }
 345          
 346          $destFolder = $parentFolder . basename($folder);
 347          $this->createFolder($destFolder);
 348          if ($this->hasError()) {
 349              $err = $this->getError();
 350              $this->setError('PHP_FOLDER_COPY', array(), array($destFolder, $err[1]));
 351              return;
 352          }
 353          $destFolder = KT_realpath($destFolder);
 354          
 355          $d = dir($folder);
 356          while (false!==($entry = $d->read())) {
 357              if ($entry == '.' || $entry == '..') {
 358                  continue;
 359              }
 360              if (is_dir($folder . $entry)) {
 361                  $this->copyFolder($folder . $entry, $destFolder);
 362              } else {
 363                  @copy($folder . $entry, $destFolder . $entry);
 364                  KT_setFilePermissions($destFolder . $entry);
 365              }
 366          }
 367          $d->close();
 368      }
