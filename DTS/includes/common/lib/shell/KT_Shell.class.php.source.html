  56      /**

  57       * execute the command

  58       * @var array $command containing the shell command to be executed (just the name of the program, without the full path, just filename)

  59       * @param array $arguments containing the arguments for the shell command

  60       * @return string the output of the execute command

  61       * @access public

  63  	function execute($commands, $arguments) {
  64          session_write_close();
  65          $this->setArguments($arguments);
  66          
  67          $this->checkSystem();
  68          if  ($this->hasError()) {
  69              return;
  70          }
  71          
  72          $tmp_e = '';
  73          
  74          foreach ($commands as $key => $command) {
  75              $command = $this->prepareCommand($command);
  76              $cmd = $command;
  77              if (strpos($command, ' ') !== false) {
  78                  $cmd = '"' . $command . '"';
  79                  if (strtolower(substr(PHP_OS, 0, 1)) == 'w') {
  80                      $cmd = 'CALL ' . $cmd;
  81                  }
  82              }
  83              $cmd = $cmd .' '. $this->getArguments();
  84              ob_start();
  85              $exit_code = 1;
  86              @passthru($cmd, $exit_code);
  87              //@system($cmd, $exit_code); 

  88              $output = ob_get_contents();
  89              ob_end_clean();
  90              
  91              if ($exit_code==0) {
  92                  $this->setError('', array(), array());
  93                  $tmp_e = '';
  94                  $this->executedCommand = $command;
  95                  break;
  96              } else {
  97                  $tmp_e = 'PHP_SHELL_EXEC_ERROR'; 
  98              }
  99          }
 100          @session_start();
 101          
 102          if ($exit_code!=0) {
 103              $this->setError($tmp_e, array(), array($command . ' '. $this->getArguments(), $exit_code, $output));
 104              
 105              /* safe mode */

 106              if (ini_get("safe_mode")) {
 107                  $this->setError('PHP_SHELL_ERR_SAFE_MODE', array(), array());
 108              }
 109              return;
 110          } else {
 111              return $output;
 112          }
 113      }

 187       * @var string $arg

 191  	function escapeshellarg($arg) {
 192          if ($arg != '<' && $arg != '>') {
 193              $arg = preg_replace("/^\"/", "", $arg);
 194              $arg = preg_replace("/\"$/", "", $arg);
 195              $arg = preg_replace("/([^\\\])\"/ims", "$1\\\"", $arg);
 196              return '"'. $arg .'"';
 197          } else {
 198              return $arg;
 199          }
 200      }
