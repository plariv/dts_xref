 152      /**

 153       * send the email

 154       * @param string $server server name

 155       * @param string $port port number

 156       * @param string $user user_name

 157       * @param string $password the password

 158       * @param string $from from address

 159       * @param string $to to address

 160       * @param string $cc copy carbon address

 161       * @param string $bcc blind copy carbon address

 162       * @param string $encoding encoding of the email

 163       * @param string $textBody text body of the email

 164       * @param string $htmlBody the HTML body of the email

 165       * @return string empty on succes, error message on error;

 166       * @access public

 167       */
 168  	function sendemail($server, $port, $user, $password, $from, $to, $cc, $bcc, $subject, $encoding, $textBody, $htmlBody)
 169      {
 170          $this->server = $server;
 171          $this->port = $port;
 172          $this->user    = $user;
 173          $this->password = $password;
 174          $this->from = $from;
 175          $this->to = $to;
 176          $this->cc = $cc;
 177          $this->bcc = $bcc;
 178          $this->subject = $subject;
 179          $this->encoding = $encoding;
 180          if ($textBody=='' && $htmlBody!='') {
 181              $this->textBody = strip_tags($htmlBody);
 182          } else {
 183              $this->textBody = $textBody;
 184          }
 185          $this->htmlBody = $htmlBody;
 186          if ($this->KT_CRLF=='') {
 187              // fix for different os's line endings

 188              if (strtoupper(substr(PHP_OS, 0, 3) == 'WIN')) {
 189                  $this->KT_CRLF = "\r\n";
 190              } elseif (strtoupper(substr(PHP_OS, 0, 3) == 'MAC')) {
 191               $this->KT_CRLF = "\r";
 192              } else {
 193               $this->KT_CRLF = "\n";
 194              }
 195          }
 196          $auth = true;
 197          if (strlen(trim($user)) == 0 && strlen(trim($password)) == 0) {
 198            $auth = false;
 199          }
 200          
 201          // load the required PEAR classes

 202          $this->loadPEARclasses();
 203          
 204          $mime = new Mail_mime($this->KT_CRLF);
 205          $mime->setTXTBody($this->getTextBody());
 206          $mime->setHTMLBody($this->getHtmlBody());
 207          foreach ($this->attachements as $filename) {
 208              $mime->addAttachment($filename, function_exists('mime_content_type') ? mime_content_type($filename) : 'application/octet-stream');
 209          }    
 210          $body = $mime->get($this->getEncodingSettings());
 211          $headers = $mime->headers($this->getHeaders());
 212          
 213          if ($this->server != '') {
 214              if ($this->from == '') {
 215                  $this->setError('PHP_EMAIL_FROM_NOT_SET' , array(), array());
 216                  return ;
 217              }
 218              $arrSMTPParameters = array(
 219                                  'host'=>$server,
 220                                  'port'=>$port,
 221                                  'auth'=>$auth,
 222                                  'user_name'=>$user,
 223                                  'password'=>$password
 224                                  ); 
 225              $mail_object = &Mail::factory('smtp', $arrSMTPParameters);
 226              $headers['To'] = $this->getTo();
 227              $mailresult    = $mail_object->send($this->getSendToSMTP(), $headers, $body);
 228          } else {
 229              $mail_object = &Mail::factory('mail');
 230              $mailresult    = $mail_object->send($this->getSendTo(), $headers, $body);
 231          }
 232          
 233          if ($mailresult!==true) {
 234              $this->setError('EMAIL_ERROR_SENDING' , array(), array($mailresult->message));
 235          }
 236      }

 380      /**

 381       * getter

 382       * @return array to, cc, bcc email addresses;

 383       * @access private

 384       */
 385  	function getSendToSMTP()
 386      {
 387          $ret = array();
 388          $to = $this->prepareText($this->to);
 389          $cc = '';
 390          $bcc = '';
 391          if ($this->getCc()!='') {
 392              $cc = $this->getCc();
 393          }
 394          if ($this->getBcc()!='') {
 395              $bcc = $this->getBcc();
 396          }
 397          
 398          if (preg_match('/.*<([^@]+@[^>]+)>.*/', $to, $matches)) {
 399              $to = $matches[1];
 400          }
 401          $ret[] = $to;
 402          if ($cc != '') {
 403              if (preg_match('/.*<([^@]+@[^>]+)>.*/', $cc, $matches)) {
 404                  $cc = $matches[1];
 405              }
 406              $ret[] = $cc;
 407          }
 408          if ($bcc != '') {
 409              if (preg_match('/.*<([^@]+@[^>]+)>.*/', $bcc, $matches)) {
 410                  $bcc = $matches[1];
 411              }
 412              $ret[] = $bcc;
 413          }
 414          return $ret;
 415      }
