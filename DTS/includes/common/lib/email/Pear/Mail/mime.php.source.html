 136      /**

 137       * Accessor function to set the body text. Body text is used if

 138       * it's not an html mail being sent or else is used to fill the

 139       * text/plain part that emails clients who don't support

 140       * html should show.

 141       *

 142       * @param  string  $data   Either a string or

 143       *                         the file name with the contents

 144       * @param  bool    $isfile If true the first param should be treated

 145       *                         as a file name, else as a string (default)

 146       * @param  bool    $append If true the text or file is appended to

 147       *                         the existing body, else the old body is

 148       *                         overwritten

 149       * @return mixed   true on success or PEAR_Error object

 150       * @access public

 151       */
 152      function setTXTBody($data, $isfile = false, $append = false)
 153      {
 154          if (!$isfile) {
 155              if (!$append) {
 156                  $this->_txtbody = $data;
 157              } else {
 158                  $this->_txtbody .= $data;
 159              }
 160          } else {
 161              $cont = $this->_file2str($data);
 162              if (PEAR::isError($cont)) {
 163                  return $cont;
 164              }
 165              if (!$append) {
 166                  $this->_txtbody = $cont;
 167              } else {
 168                  $this->_txtbody .= $cont;
 169              }
 170          }
 171          return true;
 172      }

 234       * Adds a file to the list of attachments.

 236       * @param  string  $file       The file name of the file to attach

 237       *                             OR the file data itself

 238       * @param  string  $c_type     The content type

 240       *                             Only use if $file is the file data

 244       * @access public

 246      function addAttachment($file, $c_type = 'application/octet-stream',
 247                             $name = '', $isfilename = true,
 248                             $encoding = 'base64')
 249      {
 250          $filedata = ($isfilename === true) ? $this->_file2str($file)
 251                                             : $file;
 252          if ($isfilename === true) {
 253              // Force the name the user supplied, otherwise use $file

 254              $filename = (!empty($name)) ? $name : $file;
 255          } else {
 256              $filename = $name;
 257          }
 258          if (empty($filename)) {
 259              return PEAR::raiseError(
 260                'The supplied filename for the attachment can\'t be empty'
 261              );
 262          }
 263          $filename = basename($filename);
 264          if (PEAR::isError($filedata)) {
 265              return $filedata;
 266          }
 267  
 268          $this->_parts[] = array(
 269                                  'body'     => $filedata,
 270                                  'name'     => $filename,
 271                                  'c_type'   => $c_type,
 272                                  'encoding' => $encoding
 273                                 );
 274          return true;
 275      }

 277      /**

 278       * Get the contents of the given file name as string

 279       *

 280       * @param  string  $file_name  path of file to process

 281       * @return string  contents of $file_name

 282       * @access private

 283       */
 284      function &_file2str($file_name)
 285      {
 286          if (!is_readable($file_name)) {
 287              return PEAR::raiseError('File is not readable ' . $file_name);
 288          }
 289          if (!$fd = fopen($file_name, 'rb')) {
 290              return PEAR::raiseError('Could not open ' . $file_name);
 291          }
 292          $filesize = filesize($file_name);
 293          if ($filesize == 0){
 294              $cont =  "";
 295          }else{
 296              $cont = fread($fd, $filesize);
 297          }
 298          fclose($fd);
 299          return $cont;
 300      }

 441      /**

 442       * Builds the multipart message from the list ($this->_parts) and

 443       * returns the mime content.

 444       *

 445       * @param  array  Build parameters that change the way the email

 446       *                is built. Should be associative. Can contain:

 447       *                text_encoding  -  What encoding to use for plain text

 448       *                                  Default is 7bit

 449       *                html_encoding  -  What encoding to use for html

 450       *                                  Default is quoted-printable

 451       *                7bit_wrap      -  Number of characters before text is

 452       *                                  wrapped in 7bit encoding

 453       *                                  Default is 998

 454       *                html_charset   -  The character set to use for html.

 455       *                                  Default is iso-8859-1

 456       *                text_charset   -  The character set to use for text.

 457       *                                  Default is iso-8859-1

 458       *                head_charset   -  The character set to use for headers.

 459       *                                  Default is iso-8859-1

 460       * @return string The mime content

 461       * @access public

 462       */
 463      function &get($build_params = null)
 464      {
 465          if (isset($build_params)) {
 466              while (list($key, $value) = each($build_params)) {
 467                  $this->_build_params[$key] = $value;
 468              }
 469          }
 470  
 471          if (!empty($this->_html_images) AND isset($this->_htmlbody)) {
 472              foreach ($this->_html_images as $key => $value) {
 473                  $regex = '#(\s)((?i)src|background|href(?-i))\s*=\s*(["\']?)' . preg_quote($value['name'], '#') .
 474                           '\3#';
 475                  $rep = '\1\2=\3cid:' . $value['cid'] .'\3';
 476                  $this->_htmlbody = preg_replace($regex, $rep,
 477                                         $this->_htmlbody
 478                                     );
 479              }
 480          }
 481  
 482          $null        = null;
 483          $attachments = !empty($this->_parts)                ? true : false;
 484          $html_images = !empty($this->_html_images)          ? true : false;
 485          $html        = !empty($this->_htmlbody)             ? true : false;
 486          $text        = (!$html AND !empty($this->_txtbody)) ? true : false;
 487  
 488          switch (true) {
 489          case $text AND !$attachments:
 490              $message =& $this->_addTextPart($null, $this->_txtbody);
 491              break;
 492  
 493          case !$text AND !$html AND $attachments:
 494              $message =& $this->_addMixedPart();
 495              for ($i = 0; $i < count($this->_parts); $i++) {
 496                  $this->_addAttachmentPart($message, $this->_parts[$i]);
 497              }
 498              break;
 499  
 500          case $text AND $attachments:
 501              $message =& $this->_addMixedPart();
 502              $this->_addTextPart($message, $this->_txtbody);
 503              for ($i = 0; $i < count($this->_parts); $i++) {
 504                  $this->_addAttachmentPart($message, $this->_parts[$i]);
 505              }
 506              break;
 507  
 508          case $html AND !$attachments AND !$html_images:
 509              if (isset($this->_txtbody)) {
 510                  $message =& $this->_addAlternativePart($null);
 511                  $this->_addTextPart($message, $this->_txtbody);
 512                  $this->_addHtmlPart($message);
 513              } else {
 514                  $message =& $this->_addHtmlPart($null);
 515              }
 516              break;
 517  
 518          case $html AND !$attachments AND $html_images:
 519              if (isset($this->_txtbody)) {
 520                  $message =& $this->_addAlternativePart($null);
 521                  $this->_addTextPart($message, $this->_txtbody);
 522                  $related =& $this->_addRelatedPart($message);
 523              } else {
 524                  $message =& $this->_addRelatedPart($null);
 525                  $related =& $message;
 526              }
 527              $this->_addHtmlPart($related);
 528              for ($i = 0; $i < count($this->_html_images); $i++) {
 529                  $this->_addHtmlImagePart($related, $this->_html_images[$i]);
 530              }
 531              break;
 532  
 533          case $html AND $attachments AND !$html_images:
 534              $message =& $this->_addMixedPart();
 535              if (isset($this->_txtbody)) {
 536                  $alt =& $this->_addAlternativePart($message);
 537                  $this->_addTextPart($alt, $this->_txtbody);
 538                  $this->_addHtmlPart($alt);
 539              } else {
 540                  $this->_addHtmlPart($message);
 541              }
 542              for ($i = 0; $i < count($this->_parts); $i++) {
 543                  $this->_addAttachmentPart($message, $this->_parts[$i]);
 544              }
 545              break;
 546  
 547          case $html AND $attachments AND $html_images:
 548              $message =& $this->_addMixedPart();
 549              if (isset($this->_txtbody)) {
 550                  $alt =& $this->_addAlternativePart($message);
 551                  $this->_addTextPart($alt, $this->_txtbody);
 552                  $rel =& $this->_addRelatedPart($alt);
 553              } else {
 554                  $rel =& $this->_addRelatedPart($message);
 555              }
 556              $this->_addHtmlPart($rel);
 557              for ($i = 0; $i < count($this->_html_images); $i++) {
 558                  $this->_addHtmlImagePart($rel, $this->_html_images[$i]);
 559              }
 560              for ($i = 0; $i < count($this->_parts); $i++) {
 561                  $this->_addAttachmentPart($message, $this->_parts[$i]);
 562              }
 563              break;
 564  
 565          }
 566  
 567          if (isset($message)) {
 568              $output = $message->encode();
 569              $this->_headers = array_merge($this->_headers,
 570                                            $output['headers']);
 571              return $output['body'];
 572  
 573          } else {
 574              $ret = false;
 575              return $ret;
 576          }
 577      }

 589      function &headers($xtra_headers = null)
 590      {
 591          // Content-Type header should already be present,

 592          // So just add mime version header

 593          $headers['MIME-Version'] = '1.0';
 594          if (isset($xtra_headers)) {
 595              $headers = array_merge($headers, $xtra_headers);
 596          }
 597          $this->_headers = array_merge($headers, $this->_headers);
 598  
 599          return $this->_encodeHeaders($this->_headers);
