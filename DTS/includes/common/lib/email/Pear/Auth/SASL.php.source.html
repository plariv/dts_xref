  48  class Auth_SASL
  49  {
  50      /**

  51      * Factory class. Returns an object of the request

  52      * type.

  53      *

  54      * @param string $type One of: Anonymous

  55      *                             Plain

  56      *                             CramMD5

  57      *                             DigestMD5

  58      *                     Types are not case sensitive

  59      */
  60      function &factory($type)
  61      {
  62          switch (strtolower($type)) {
  63              case 'anonymous':
  64                  $filename  = 'Auth/SASL/Anonymous.php';
  65                  $classname = 'Auth_SASL_Anonymous';
  66                  break;
  67  
  68              case 'login':
  69                  $filename  = 'Auth/SASL/Login.php';
  70                  $classname = 'Auth_SASL_Login';
  71                  break;
  72  
  73              case 'plain':
  74                  $filename  = 'Auth/SASL/Plain.php';
  75                  $classname = 'Auth_SASL_Plain';
  76                  break;
  77  
  78              case 'crammd5':
  79                  $filename  = 'Auth/SASL/CramMD5.php';
  80                  $classname = 'Auth_SASL_CramMD5';
  81                  break;
  82  
  83              case 'digestmd5':
  84                  $filename  = 'Auth/SASL/DigestMD5.php';
  85                  $classname = 'Auth_SASL_DigestMD5';
  86                  break;
  87  
  88              default:
  89                  return PEAR::raiseError('Invalid SASL mechanism type');
  90                  break;
  91          }
  92  
  93          // require_once(PEARDIR . $filename);

  94          $ret = new $classname();
  95          return $ret;
  96      }
  97  }
