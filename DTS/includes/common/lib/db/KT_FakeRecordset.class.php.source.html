  97  	function getMySQLfakeRS() {
  98          $this->tmpTableName = $this->tmpTableName . '_' . date('Ymd');
  99          $test_create_sql = 'CREATE TEMPORARY TABLE ' . $this->tmpTableName . ' (kt_test TEXT)';
 100          $delete_sql = 'DROP TABLE ' . $this->tmpTableName;
 101          $create_sql = 'CREATE TEMPORARY TABLE ' . $this->tmpTableName . ' (';
 102          $insert_sql = 'INSERT INTO ' . $this->tmpTableName . ' (';
 103          $select_sql = 'SELECT * FROM ' . $this->tmpTableName;
 104          $insert_values = '';
 105          $result = '';
 106          $doInsert = true;
 107          $multiple = false;
 108          
 109          if (count($this->fields) == 0) {
 110              // empty fake rs

 111              $doInsert = false;
 112          }
 113          
 114          $this->connection->Execute($delete_sql);
 115          $response = $this->connection->Execute($test_create_sql);
 116          if ($response === false) {
 117              return $this->getMySQL4fakeRS();
 118          }
 119          if (count($this->fields) == 0) {
 120              $this->fields['KT_fakeField'] = '';
 121          }
 122          foreach ($this->fields as $key => $value) {
 123              $create_sql .= $key . ' TEXT, ';
 124              $insert_sql .= $key . ', ';
 125              if (!is_array($value)) {
 126                  $insert_values .= "'" . mysql_escape_string($value) . "', ";
 127              } else {
 128                  $multiple = true;
 129              }
 130          }
 131          
 132          $create_sql = substr(trim($create_sql), 0, -1) . ')';
 133          $insert_sql = substr(trim($insert_sql), 0, -1) . ') VALUES (';
 134          if ($multiple) {
 135              $multiple_values = array();
 136              foreach ($this->fields as $key => $values) {
 137                  $i = 0;
 138                  foreach ($values as $key => $value) {
 139                      if (!isset($multiple_values[$i])) {
 140                          $multiple_values[$i] = '';
 141                      }
 142                      $multiple_values[$i] .= "'" . mysql_escape_string($value) . "', ";
 143                      $i++;
 144                  }
 145              }
 146              for ($i = 0; $i < count($multiple_values); $i++) {
 147                  $multiple_values[$i] = substr(trim($multiple_values[$i]), 0, -1);
