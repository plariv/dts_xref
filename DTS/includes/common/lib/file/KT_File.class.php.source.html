 203  	function checkFolder($file, $mode, $from) 
 204      {
 205          $folderName = $this->getFolder($file);
 206          $folder = new KT_folder();
 207          $folder->createFolder($folderName);
 208          switch ($mode) {
 209              case 'read':
 210                  $right = $folder->checkRights($folderName, 'read');
 211                  break;
 212              case 'write':
 213              default:
 214                  $right = $folder->checkRights($folderName, 'write');
 215                  break;
 216          }
 217          if ($folder->hasError()) {
 218              $arr = $folder->getError();
 219              $this->setError('PHP_FILE_FOLDER_ERROR', array($from, $arr[0]), array($from, $arr[1]));
 220          }
 221          if ($right !== true) {
 222              $this->setError('PHP_FILE_CHECK_FOLDER_ERROR', array($from), array($from, $mode, $folderName));
 223          }

 227      /**

 228       * getter. return the folder from a file path with the correct directory separator

 229       * @param string $file the path to the file 

 230       * return string

 231       * @access public

 232       */
 233  	function getFolder($file)
 234      {
 235          if (strtolower(substr(PHP_OS, 0, 1))=='w') {
 236              $file = str_replace('/', '\\', $file);
 237          } else {
 238              $file = str_replace('\\', '/', $file);
 239          }
 240          $arr = split('[\\/]', $file);
 241          array_pop($arr);
 242          return implode(DIRECTORY_SEPARATOR, $arr);
 243      }
