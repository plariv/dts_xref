 100   * @param string $pageUrl the url of the calling script;

 101   * @param string $templateUrl the url of the template url;

 102   * @param string $relUrl the url to transform in absolute if case;

 103   * @param boolean $makeAbsolut flag to transform or not in absolut url;

 104   * @return string return the absolute url;

 105   */
 106  function KT_Rel2AbsUrl($pageUrl, $templateUrl, $relUrl, $makeAbsolut) {
 107      $server = KT_getServerName();
 108      if (!$makeAbsolut) {
 109          $pageUrl = str_replace($server, '', $pageUrl);
 110          $templateUrl = str_replace($server, '', $templateUrl);
 111          $templateUrl = str_replace($pageUrl, '', $templateUrl);
 112          $server = '';
 113      } 
 114      if (substr($relUrl,0,1) == "/") {
 115          return $server.$relUrl;
 116      }
 117      if (strpos($relUrl,"://") !== false) {
 118          return $relUrl;
 119      }
 120  
 121      $arrTemplateUrl = explode('/', $templateUrl);
 122      array_pop($arrTemplateUrl);
 123      if (strpos($templateUrl,"://") !== false) {
 124          $ret = implode('/', $arrTemplateUrl) . (count($arrTemplateUrl)>0?'/':'') . $relUrl;
 125      } else {
 126          $arrPageUrl = explode('/', $pageUrl);
 127          array_pop($arrPageUrl);
 128          $ret = implode('/', $arrPageUrl) . '/' . implode('/', $arrTemplateUrl) . (count($arrTemplateUrl)>0?'/':'') . $relUrl;
 129      }
 130      
 131      $ret = KT_CanonizeRelPath($ret);
 132      return $ret;
 133  }

 136   * Try to canonize a path (replace ../ and ./ ); 

 137   * @param string $relPath the path to be canonize;

 138   * @return string return the canonized path;

 139   */
 140  function KT_CanonizeRelPath($relPath) {
 141      if (strpos($relPath, "..") !== false || strpos($relPath, "/.") !== false) {
 142          $parts = explode('/',$relPath);
 143          $newParts = array();
 144          for($i=0;$i<count($parts);$i++) {
 145              if ($parts[$i] == '..') {
 146                  if (count($newParts) > 0 && $newParts[count($newParts) - 1] != '..') {
 147                      array_pop($newParts);
 148                  } else {
 149                      $newParts[] = $parts[$i];
 150                  }
 151              } elseif ($parts[$i] != '.') {
 152                  $newParts[] = $parts[$i];
 153              }
 154          }
 155          $ret = implode('/',$newParts);
 156      } else {
 157          $ret = $relPath;
 158      }
 159      return $ret;
 160  }

 373  /**

 374   * Adding or replacing a param/value in the string. 

 375   * @param string $qstring the query string wich will be the subject (ex. ?myvar=1&myvar2=a or myvar=3&myvar2=v);

 376   * @param string $paramName the name of the variable wich will be strip from or will have a new value (ex. myvar or othervar);

 377   * @param string $paramValue if is not null this will be the value of the $paramName, if not the $paramName will be strip off (ex. myvalue or null);

 378   * @return string return the URL (ex. ?myvar=myvalue&myvar2=a or ?myvar2=v );

 379   */
 380  function KT_addReplaceParam($qstring, $paramName, $paramValue=null) {
 381  
 382      // extract the URI if any

 383      if (strpos($qstring, "?") !== false) {
 384          $uri = preg_replace("/\?.*$/", "?", $qstring);
 385          $qstring = preg_replace("/^.*\?/", "", $qstring);
 386      } else {
 387          if (strpos($qstring, "=") !== false) {
 388              $uri = "";
 389          } else {
 390              $uri = $qstring;
 391              if ($paramValue !== null) {
 392                  $uri .= "?";
 393              }
 394              $qstring = "";
 395          }
 396      }
 397  
 398      // the list of parameters

 399      $arr = explode('&',$qstring);
 400  
 401      // remove $paramName from the list

 402      foreach($arr as $key=>$value) {
 403          $tmpArr = explode('=',$value);
 404          if (urldecode($tmpArr[0]) == $paramName) {
 405              unset($arr[$key]);
 406              break;
 407          } else {
 408              if (strpos($paramName, "/") === 0) {
 409                  if (preg_match($paramName, urldecode($tmpArr[0]))) {
 410                      unset($arr[$key]);
 411                      break;
 412                  }
 413              }
 414          }
 415      }
 416  
 417      // add $paramName to the list

 418      if ($paramValue !== null) {
 419          $arr[] = rawurlencode($paramName).'='.rawurlencode($paramValue);
 420      }
 421  
 422      $ret = implode('&',$arr);
 423      $ret = preg_replace("/^&/", "", $ret);
 424  
 425      // if no parameters, remove the trailing ? from the URI

 426      if ($ret == '') {
 427          $uri = preg_replace("/\?$/", "", $uri);
 428      }
 429  
 430      // merge the URI with the new list

 431      $ret = $uri . $ret;
 432  
 433      return $ret;
 434  }

1075  function KT_applyDate2rule($date, $rule) {

1087          foreach ($rule as $ruleKey => $ruleValue) {
1088              $index = $ruleValue['piece'];
1089              if (isset($datePieces[$index])) {
1090                  $dateArr[$ruleKey] = $datePieces[$index];
1091              }
1092          }
1093      }
1094

1096      if (in_array('h', $ruleKeys) && isset($dateArr['h'])) {
1097          $dateArr['H'] = $dateArr['h'];
1098          unset($dateArr['h']);
1099      }
1100      if (in_array('t', $ruleKeys)) {
1101          $value = isset($dateArr['t'])?$dateArr['t']:'A';
1102          if (strtoupper(substr($value, 0, 1)) == 'P') {
1103              if ($dateArr['H'] < 12) {
1104                  $dateArr['H'] = $dateArr['H'] + 12;
1105              }
1106          } else {
1107              if ($dateArr['H'] == 12) {
1108                  $dateArr['H'] = 0;
1109              }
1110          }
1111          unset($dateArr['t']);
1112      }
1113  
1114      foreach (array('y', 'm', 'd', 'H', 'i', 's') as $key => $piece) {
1115          if (strlen($dateArr[$piece]) == 1) {
1116              $dateArr[$piece] = '0' . $dateArr[$piece];
1117          }
1118      }
1119  
1120      if (strlen($dateArr['y']) == 2) {
1121          if ($dateArr['y'] < 70) {
1122              $dateArr['y'] = '20' . $dateArr['y'];
1123          } else {
