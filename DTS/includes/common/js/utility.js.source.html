 289   * utility.dom.getBorderBox

 290   *     returns a border box object (x,y,width,height) which perfectly covers the el element and its borders

 291   *     the x, y are absolute coordinates measured from from the window viewport

 292   *     use the box as the second parameter in utility.dom.setBorderBox

 293   * @param {DOMElement or String} el - 

 294   * @param {DOMDocument,optional} doc - 

 295   */
 296  utility.dom.getBorderBox = function (el, doc) {
 297      doc = doc || document;
 298      if (typeof(el) == 'string') {
 299          el = doc.getElementById(el);
 300      }
 301  
 302      if (!el) {
 303          return false;
 304      }
 305  
 306      if (el.parentNode === null || utility.dom.getStyleProperty(el, 'display') == 'none') {
 307          //element must be visible to have a box

 308          return false;
 309      }
 310  
 311      var ret = {x:0, y:0, width:0, height:0};
 312      var parent = null;
 313      var box;
 314  
 315      if (el.getBoundingClientRect) { // IE
 316          box = el.getBoundingClientRect();
 317          var scrollTop = doc.documentElement.scrollTop || doc.body.scrollTop;
 318          var scrollLeft = doc.documentElement.scrollLeft || doc.body.scrollLeft;
 319          ret.x = box.left + scrollLeft;
 320          ret.y = box.top + scrollTop;
 321          ret.width = box.right - box.left;
 322          ret.height = box.bottom - box.top;
 323      } else if (doc.getBoxObjectFor) { // gecko
 324          box = doc.getBoxObjectFor(el);
 325          ret.x = box.x;
 326          ret.y = box.y;
 327          ret.width = box.width;
 328          ret.height = box.height;
 329          var btw = utility.dom.getPixels(el, "border-top-width");
 330          var blw = utility.dom.getPixels(el, "border-left-width");
 331          ret.x -= blw;
 332          ret.y -= btw;
 333      } else { // safari/opera
 334          ret.x = el.offsetLeft;
 335          ret.y = el.offsetTop;
 336          ret.width = el.offsetWidth;
 337          ret.height = el.offsetHeight;
 338          parent = el.offsetParent;
 339          if (parent != el) {
 340              while (parent) {
 341                  ret.x += parent.offsetLeft;
 342                  ret.y += parent.offsetTop;
 343                  parent = parent.offsetParent;
 344              }
 345          }
 346          var blw = utility.dom.getPixels(el, "border-left-width");
 347          var btw = utility.dom.getPixels(el, "border-top-width");
 348          ret.x -= blw;
 349          ret.y -= btw;
 350          // opera & (safari absolute) incorrectly account for body offsetTop

 351          var ua = navigator.userAgent.toLowerCase();
 352          if (is.opera || is.safari && utility.dom.getStyleProperty(el, 'position') == 'absolute') {
 353              ret.y -= doc.body.offsetTop;
 354          }
 355      }
 356      if (el.parentNode) {
 357              parent = el.parentNode;
 358      } else {
 359          parent = null;
 360      }
 361      while (parent && parent.tagName != 'BODY' && parent.tagName != 'HTML') {
 362          ret.x -= parent.scrollLeft;
 363          ret.y -= parent.scrollTop;
 364          if (parent.parentNode) {
 365              parent = parent.parentNode;
 366          }    else {
 367              parent = null;
 368          }
 369      }
 370      return ret;
 371  };

 381      var pos = utility.dom.getBorderBox(el, el.ownerDocument);
 382      if (pos === false) {

 384      }
 385  
 386      var delta = {
 387          x:utility.dom.getPixels(el, 'left'),

 390  
 391      var new_pos = {x:0, y:0};
 392      if (box.x !== null) {
 393          new_pos.x = box.x - pos.x + delta.x;
 394      }
 395      if (box.y !== null) {
 396          new_pos.y = box.y - pos.y + delta.y;
 397      }
 398  
 399      if (box.x !== null) {
 400          el.style.left = new_pos.x + 'px';
 401      }
 402      if (box.y !== null) {
 403          el.style.top = new_pos.y + 'px';
 404      }
 405      return true;
 406  };

 413  utility.dom.bringIntoView = function (source) {
 414      var box = utility.dom.getBorderBox(source, source.ownerDocument);
 415      if (box === false) {
 416          return false;
 417      }
 418  
 419      var current = {
 420          x:utility.dom.getPixels(source, 'left'),
 421           y:utility.dom.getPixels(source, 'top')
 422      };
 423  
 424      var delta = {x:0, y:0};
 425      var offset_fix = {x:0, y:0};
 426      var strictm = source.ownerDocument.compatMode == "CSS1Compat";
 427      var doc = (is.ie && strictm || is.mozilla)?source.ownerDocument.documentElement:source.ownerDocument.body;
 428  
 429       offset_fix.x = utility.dom.getPixels(doc, 'border-left-width');
 430       offset_fix.y = utility.dom.getPixels(doc, 'border-top-width');
 431  
 432      var st = doc.scrollTop;
 433      var ch = doc.clientHeight;
 434      var t = box.y + (is.ie?-offset_fix.y:offset_fix.y);
 435      var b = box.y + box.height + (is.ie?-offset_fix.y:offset_fix.y);
 436  
 437      if ( b - st > ch) {
 438          delta.y = ch - (b - st);
 439          if (t + delta.y < st) {
 440              delta.y = st-t;
 441          }
 442      } else if (t < st) {
 443          delta.y = st - t;
 444      }
 445  
 446      if (delta.y != 0) {
 447          source.style.top = (current.y + delta.y) + 'px';
 448      }
 449      
 450      var sl = doc.scrollLeft;
 451      var cw = doc.clientWidth;
 452      var l = box.x + (is.ie?-offset_fix.x:offset_fix.x);
 453      var r = box.x + box.width + (is.ie?-offset_fix.x:offset_fix.x);
 454  
 455      if ( r - sl > cw) {
 456          delta.x = cw - (r - sl);
 457          if (l + delta.x < sl) {
 458              delta.x = sl-l;
 459          }
 460      } else if (l < sl) {
 461          delta.x = sl - l;
 462      }
 463  
 464      if (delta.x != 0) {
 465          source.style.left = (current.x + delta.x) + 'px';
 466      }
 467  };

 921                    currentContext[currentContextIndex++] = found[k];
 922                  }
 923              }
 924              continue; // Skip to next token

 925          }
 926          // Code to deal with attribute selectors

 927          if (token.match(/^(\w*)\[(\w+)([=~\|\^\$\*]?)=?"?([^\]"]*)"?\]$/)) {
 928              var tagName = RegExp.$1;
 929              var attrName = RegExp.$2;
 930              var attrOperator = RegExp.$3;
 931              var attrValue = RegExp.$4;
 932              if (!tagName) {
 933                  tagName = '*';
 934              }
 935              // Grab all of the tagName elements within current context

 936              var found = new Array;
 937              var foundCount = 0;

 939                  var elements;
 940                  if (tagName == '*') {
 941                      elements = utility.dom.getAllChildren(currentContext[h]);
 942                  } else {
 943                      elements = currentContext[h].getElementsByTagName(tagName);
 944                  }
 945                  for (var j = 0; j < elements.length; j++) {
 946                      found[foundCount++] = elements[j];
 947                  }
 948              }
 949              currentContext = new Array;
 950              var currentContextIndex = 0;
 951              var checkFunction; // This function will be used to filter the elements

 952              switch (attrOperator) {
 953                  case '=': // Equality
 954                      checkFunction = function(e) { try {return (e.getAttribute(attrName).toString() == attrValue);} catch(ex) { } };
 955                  break;
 956                  case '~': // Match one of space seperated words

 958                  break;
 959                  case '|': // Match start with value followed by optional hyphen
 960                      checkFunction = function(e) { return (e.getAttribute(attrName).toString().match(new RegExp('^'+attrValue+'-?'))); };
 961                  break;
 962                  case '^': // Match starts with value
 963                      checkFunction = function(e) { return (e.getAttribute(attrName).toString().indexOf(attrValue) == 0); };

 965                  case '$': // Match ends with value - fails with "Warning" in Opera 7

 967                  break;
 968                  case '*': // Match ends with value
 969                      checkFunction = function(e) { return (e.getAttribute(attrName).toString().indexOf(attrValue) > -1); };
 970                  break;
 971                  default :
 972                      // Just test for existence of attribute

 973                      checkFunction = function(e) { return e.getAttribute(attrName); };
 974              }
 975              currentContext = new Array;
 976              var currentContextIndex = 0;
 977              for (var k = 0; k < found.length; k++) {
 978                  if (checkFunction(found[k])) {
 979                      currentContext[currentContextIndex++] = found[k];
 980                  }
 981              }
 982              //alert('Attribute Selector: '+tagName+' '+attrName+' '+attrOperator+' '+attrValue);

 984          }

 986          tagName = token;
 987          var found = new Array;
 988          var foundCount = 0;
 989          for (var h = 0; h < currentContext.length; h++) {
 990              if (currentContext[h] != null) {
 991                  var elements = currentContext[h].getElementsByTagName(tagName);
 992                  for (var j = 0; j < elements.length; j++) {
 993                      found[foundCount++] = elements[j];
 994                  }
 995              }
 996          }

 998      }
