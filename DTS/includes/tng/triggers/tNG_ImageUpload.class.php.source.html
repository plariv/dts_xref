 101          $ret = NULL;
 102          $image = new KT_image();
 103          $image->setPreferedLib($GLOBALS['tNG_prefered_image_lib']);
 104          $image->addCommand($GLOBALS['tNG_prefered_imagemagick_path']);
 105          $image->resize($this->dynamicFolder.$this->uploadedFileName, $this->dynamicFolder, $this->uploadedFileName, $this->resizeWidth, $this->resizeHeight, $this->resizeProportional);
 106          if ($image->hasError()) {
 107              $arrError = $image->getError();
 108              $errObj = new tNG_error('IMG_RESIZE', array(), array($arrError[1]));
 109              if ($this->dbFieldName != '') {
 110                  $errObj->addFieldError($this->dbFieldName, 'IMG_RESIZE', array());
 111              }
 112              $ret = $errObj;            
 113          }
 114          return $ret;
