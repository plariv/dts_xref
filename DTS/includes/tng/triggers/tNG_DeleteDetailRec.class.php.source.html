 149       * contruct the SQL and execute it. it is using as value for the field the primarey key value from the transaction;

 153  	function Execute() {
 154          $pk_value = $this->tNG->getPrimaryKeyValue();
 155          $pk_type = $this->tNG->getColumnType($this->tNG->getPrimaryKey());
 156          $pk_value = KT_escapeForSql($pk_value, $pk_type);
 157          
 158          if (count($this->fileRenameRule) > 0 || count($this->folderRenameRule) > 0 ) {
 159              $sql = 'SELECT * FROM '. $this->table .' WHERE '. KT_escapeFieldName($this->field) . " = " . $pk_value;
 160              $rs = $this->tNG->connection->Execute($sql);
 161              if ($rs === false) {
 162                  return new tNG_error('DEL_DR_SQL_ERROR', array(), array($this->tNG->connection->ErrorMsg(), $sql));
 163              }
 164              if ($rs->RecordCount() == 0) {
 165                  return null;
 166              }
 167          }        
 168          
 169          // prepare to delete files

 170          if (count($this->fileRenameRule) > 0 ) {
 171              $fullFileName = array();
 172              $fullFileNameFolder = array();
 173              for ($i=0; $i<count($this->fileRenameRule); $i++) {    
 174                  while (!$rs->EOF) {
 175                      $arr = array();
 176                      foreach ($rs->fields as $col => $value) {
 177                          $arr[$col] = $value;
 178                      }
 179                      $folder = $this->fileFolder[$i];
 180                      $fileName = KT_DynamicData($this->fileRenameRule[$i], $this->tNG, '', false, $arr);
 181                      // security

 182                                          if (substr(KT_realpath($folder . $fileName), 0, strlen($folder)) != $folder) {
 183                                                  $baseFileName = dirname(KT_realpath($folder . $fileName, false));
 184                          $ret = new tNG_error("FOLDER_DEL_SECURITY_ERROR", array(), array($baseFileName, $folder));
 185                          return $ret;
 186                      }
 187                      $fullFileName[] = $fileName;
 188                      $fullFileNameFolder[] = $folder;
 189                      $rs->MoveNext();
 190                  }
 191                  $rs->MoveFirst();
 192              }
 193          }
 194          
 195          // prepare to delete related folders

 196          if (count($this->folderRenameRule) > 0 ) {
 197              $relatedFolder = array();
 198              for ($i=0; $i<count($this->folderRenameRule); $i++) {
 199                  while (!$rs->EOF) {
 200                      $arr = array();
 201                      foreach ($rs->fields as $col => $value) {
 202                          $arr[$col] = $value;
 203                      }    
 204                      $folder = $this->folder[$i];
 205                      $f = KT_DynamicData($this->folderRenameRule[$i], $this->tNG, '', false, $arr);
 206                      // security

 207                                          if (substr(KT_realpath($folder . $f), 0, strlen($folder)) != $folder) {
 208                                                  $baseFileName = dirname(KT_realpath($folder . $f, false));
 209                          $ret = new tNG_error("FOLDER_DEL_SECURITY_ERROR", array(), array($baseFileName, $folder));
 210                          return $ret;
 211                      }
 212                      $relatedFolder[] = $folder . $f;
 213                      $rs->MoveNext();
 214                  }
 215                  $rs->MoveFirst();
 216              }
 217          }
 218      
 219          // delete reocords

 220          $sql = "DELETE FROM " . $this->table . " WHERE " . KT_escapeFieldName($this->field) . " = " . $pk_value;
 221          $ret = $this->tNG->connection->Execute($sql);
 222          if ($ret === false) {
 223              return new tNG_error('DEL_DR_SQL_ERROR', array(), array($this->tNG->connection->ErrorMsg(), $sql));
 224          }
 225          // delete files

 226          if (count($this->fileRenameRule) > 0 ) {
 227              for ($i=0; $i<count($fullFileName); $i++) {            
 228                  if (file_exists($fullFileNameFolder[$i] . $fullFileName[$i])) {
 229                      $delRet = @unlink($fullFileNameFolder[$i] . $fullFileName[$i]);
 230                      $path_info = KT_pathinfo($fullFileNameFolder[$i] . $fullFileName[$i]);
 231                      $this->deleteThumbnails($path_info['dirname'] . '/thumbnails/', $path_info['basename']);                    
 232                  }
 233              }
 234          }
 235          // delete related folder

 236          if (count($this->folderRenameRule) > 0 ) {

 238                  $folder = new KT_Folder();
 239                  // delete thumbnails

 240                  $folder->deleteFolderNR($relatedFolder[$i]);                
 241              }
 242          }
 243          return null;
 244      }
