 116  	function Execute() {
 117          $ret = NULL;
 118          $baseFolder = KT_realpath($this->baseFolder);
 119          if ($this->rename == false && $this->dbFieldName != '') {
 120              $fileName = $this->tNG->getSavedValue($this->dbFieldName);
 121          } else {
 122              $fileName = KT_DynamicData($this->renameRule, $this->tNG, '', true);
 123          }
 124          $folder = KT_DynamicData($this->folder, $this->tNG, '', true);
 125          // security

 126          if (substr(KT_realpath($baseFolder . $folder . $fileName), 0, strlen($baseFolder)) != $baseFolder) {
 127              $ret = new tNG_error("FOLDER_DEL_SECURITY_ERROR", array(), array(dirname(KT_realpath($baseFolder . $folder . $fileName, false)), $baseFolder));
 128              return $ret;
 129          }
 130          if ($fileName != "") {
 131              $fullFileName = $baseFolder . $folder . $fileName;
 132              if (file_exists($fullFileName)) {
 133                  $delRet = @unlink($fullFileName);
 134                  if ($delRet !== true) {
 135                      $ret = new tNG_error('FILE_DEL_ERROR', array(), array($fullFileName));
 136                      $ret->setFieldError($this->fieldName, 'FILE_DEL_ERROR_D', array($fullFileName));
 137                  } else {
 138                      $path_info = KT_pathinfo($fullFileName);
 139                      $this->deleteThumbnails($path_info['dirname'] . '/thumbnails/', $path_info['basename']);
 140                  }
 141              }
 142          }
 143          return $ret;
 144      }
