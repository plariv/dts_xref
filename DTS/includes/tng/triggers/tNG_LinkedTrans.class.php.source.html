  32              var $detailTNG;

  46  			function tNG_LinkedTrans(&$masterTNG, &$detailTNG) {
  47                  $this->masterTNG = &$masterTNG;
  48                  $this->detailTNG = &$detailTNG;
  49              }
  50              /**

  51               * setter. set the field name that's link these transactions

  55  			function setLink($linkField) {

  59               * Main method of the class. Execute the code

  60               * @return mix null or error object

  61               * @access public

  62               */
  63  			function Execute() {
  64                  if ($this->masterTNG->getError()) {
  65                      return $this->onError();
  66                  } else {
  67                      return $this->onSuccess();
  68                  }
  69              }
  70              /**

  71               * Execute the detail transaction ad retrieve the result (error)

  72               * @return mix null or error object

  73               * @access private

  74               */
  75  			function onSuccess() {
  76                  $this->detailTNG->setColumnValue($this->linkField, $this->masterTNG->getPrimaryKeyValue());
  77                  $this->detailTNG->executeSubSets = false;
  78                  $this->detailTNG->setStarted(true);    
  79                  $this->detailTNG->compileColumnsValues();
  80                  $this->detailTNG->doTransaction();    
  81  
  82                  return $this->detailTNG->getError();
  83              }
  84              /**

  85               * Executed if the master has error;

  86               * execute the rollbacktransaction on the detail transaction

  87               * @return nothing

  88               * @access private

  89               */
  90  			function onError() {
  91                  if ($this->detailTNG->isStarted()) {
  92                      // if the 2nd transaction has started

  93                      if (!$this->detailTNG->getError()) {
  94                          // if it did not throw any error

  95                          $this->detailTNG->rollBackTransaction($this->masterTNG->getError());
  96                      }
  97                  } else {
  98                      $this->detailTNG->setColumnValue($this->linkField, $this->masterTNG->getPrimaryKeyValue());
  99                      $this->detailTNG->executeSubSets = false;
 100                      $this->detailTNG->setError($this->masterTNG->getError());
 101                      $this->detailTNG->setStarted(true);
 102                      $this->detailTNG->compileColumnsValues();
 103                      $this->detailTNG->doTransaction();
 104                  }
 105                  return null;
 106              }
