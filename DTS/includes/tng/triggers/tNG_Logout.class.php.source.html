  26      var $logoutType = 'load';

  28       * redirect to page:

  29       * @var striung

  33      /**

  34       * setter.

  35       * @param string logoutType

  36       * @access public

  38  	function setLogoutType($logoutType = 'load') {
  39          $this->logoutType = $logoutType;

  46  	function setPageRedirect($pageRedirect) {

  49      /**

  50       * Unset the session variables that have to do with user login;

  51       * @return nothing

  52       * @access public

  53       */
  54  	function unsetAll() {
  55          unset($_SESSION['log_userkey']);
  56          KT_unsetSessionVar('log_userkey');
  57          unset($_SESSION['kt_login_user']);
  58          KT_unsetSessionVar('kt_login_user');
  59          unset($_SESSION['kt_login_level']);
  60          KT_unsetSessionVar('kt_login_level');
  61      
  62          unset($_SESSION['KT_denied_pageuri']);
  63          KT_unsetSessionVar('KT_denied_pageuri');
  64          unset($_SESSION['KT_denied_pagelevels']);
  65          KT_unsetSessionVar('KT_denied_pagelevels');
  66          
  67          // remove cookies

  68          $cookie_path = tNG_getRememberMePath();
  69          setcookie("log_userkey", "" , time() - 3600, $cookie_path);
  70          setcookie("kt_login_test", "" , time() - 3600, $cookie_path);
  71          unset($_COOKIE['log_userkey']);
  72          unset($_COOKIE['kt_login_test']);
  73          if (is_array($GLOBALS['tNG_login_config_session'])) {
  74              $ses_arr = $GLOBALS['tNG_login_config_session'];
  75              foreach ($ses_arr as $ses_name => $ses_value) {
  76                  unset($_SESSION[$ses_name]);
  77                  KT_unsetSessionVar($ses_name);

  79          }

  82       * Main method of the class. Execute the code; Make the redirect

  84       * @access public

  86  	function Execute() {
  87          // remove sessions

  88          if (strtolower($this->logoutType) == "load") {
  89              $this->unsetAll();
  90              if($this->pageRedirect != "") {
  91                  KT_redir ($this->pageRedirect);
  92              }    
  93          }
  94          else {
  95              if(isset($_GET['KT_logout_now']) && $_GET['KT_logout_now'] == "true"){
  96                  $this->unsetAll();
  97                  if($this->pageRedirect != "") {
  98                      KT_redir ($this->pageRedirect);
  99                  }else {
 100                      // redirect to self - after removing value for KT_logout_now

 101                      KT_redir (KT_addReplaceParam(KT_getFullUri(), 'KT_logout_now', ''));
 102                  }    
 103              }
 104          }    
 105      }
 106      /**

 113      }
