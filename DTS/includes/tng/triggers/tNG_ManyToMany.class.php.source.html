 132  	function addField($colName, $type, $method, $reference, $default = ""){
 133          if (!isset($this->columns[$colName])) {
 134              $this->columns[$colName] = array();
 135          }
 136          $this->columns[$colName]['type'] = $type;
 137          $this->columns[$colName]['method'] = $method;
 138          $this->columns[$colName]['reference'] = $reference;
 139          $this->columns[$colName]['default'] = $default;
 140  
 141          if ($method == 'VALUE') {
 142              $this->columns[$colName]['value'] = $reference;
 143              return;
 144          }
 145          if ($method == 'EXPRESSION') {
 146              $this->columns[$colName]['method'] = "VALUE";
 147          }
 148          if ($default != '') {
 149              $this->columns[$colName]['value'] = KT_DynamicData($default, $this->tNG, '');
 150          } else {
 151              $this->columns[$colName]['value'] = KT_DynamicData($reference, $this->tNG, '');
 152          }
 153          if ($type == 'DATE_TYPE') {
 154              $this->columns[$colName]['value'] = KT_formatDate2DB($this->columns[$colName]['value']);
 155          }
 156      }

 167  	function getValues() {
 168          $values = array();
 169          
 170          $fkReference = $this->fkReference;
 171          $idxReference = "";
 172          if (isset($this->tNG->multipleIdx)) {
 173              $idxReference = '_' . $this->tNG->multipleIdx;
 174              $idxReference = preg_quote($idxReference, '/');
 175          }
 176          $fkReference = preg_quote($fkReference, '/');
 177          $keys = array_keys($_POST);
 178          foreach ($keys as $idx => $key) {
 179              if (preg_match('/^' . $fkReference . '_(\d+)'.$idxReference.'$/', $key, $matches)) {
 180                  array_push($values, $matches[1]);
 181              }
 182          }
 183          return $values;
 184      }

 190  	function getOldValues() {
 191          $ret = array();
 192          $pk_value = $this->tNG->getPrimaryKeyValue();
 193          $pk_type = $this->tNG->getColumnType($this->tNG->getPrimaryKey());
 194          $pk_value = KT_escapeForSql($pk_value, $pk_type);
 195          $sql = "SELECT ".KT_escapeFieldName($this->fkName)." FROM " . $this->table . " WHERE " . KT_escapeFieldName($this->pkName) . " = " . $pk_value;
 196          $rs = $this->tNG->connection->Execute($sql);
 197          if ($rs === false) {
 198              return new tNG_error('TRIGGER_MESSAGE__MTM_SQL_ERROR', array(), array($this->tNG->connection->ErrorMsg(), $sql));
 199          }
 200          while (!$rs->EOF) {
 201              $ret[] = $rs->Fields($this->fkName);
 202              $rs->MoveNext();
 203          }
 204          return $ret;

 206      /**

 207       * Main method of the class. Execute the code.

 208       * Insert the new values and delete the old values

 209       * @return mix null or error object

 210       * @access public

 211       */
 212  	function Execute() {
 213          if ($this->fkReference == '') {
 214              return new tNG_error('TRIGGER_MESSAGE__MTM_NO_REFERENCE', array(), array());
 215          }
 216          $pk_value = $this->tNG->getPrimaryKeyValue();
 217          $pk_type = $this->tNG->getColumnType($this->tNG->getPrimaryKey());
 218          $pk_value = KT_escapeForSql($pk_value, $pk_type);
 219          
 220          $values = $this->getValues();
 221          $oldValues = $this->getOldValues();
 222          if (is_object($oldValues)) {// Returned error message
 223              return $oldValues;
 224          }
 225          if (count($oldValues) >0) {
 226              $deleteValues = array_diff($oldValues, $values);
 227              if (count($deleteValues) >0) {
 228                  $in_sql = "";
 229                  foreach ($deleteValues as $key => $value) {
 230                      if ($in_sql != '') {
 231                          $in_sql .= ",";
 232                      }
 233                      $in_sql .= KT_escapeForSql($value,$pk_type);                    
 234                  }
 235                  $sql = "DELETE FROM " . $this->table . " WHERE " . KT_escapeFieldName($this->pkName) . " = " . $pk_value." AND ".KT_escapeFieldName($this->fkName)." IN (".$in_sql.")";
 236                  $this->tNG->connection->Execute($sql);
 237                  if ($this->tNG->connection->ErrorMsg()!='') {
 238                      return new tNG_error('TRIGGER_MESSAGE__MTM_SQL_ERROR', array(), array($this->tNG->connection->ErrorMsg(), $sql));
 239                  }
 240              }
 241              // update existing values;

 242              $updateValues = array_intersect($oldValues, $values);
 243              if (count($updateValues) > 0) {
 244                  foreach ($updateValues as $key => $value) {
 245                      $arrExtraColumns = $this->getExtraColumnsValues($value, $updateValues);
 246                      if (isset($arrExtraColumns['validate']) && is_object($arrExtraColumns['validate'])) {
 247                          return $arrExtraColumns['validate'];
 248                      }
 249                      if (isset($arrExtraColumns['update']) && count($arrExtraColumns['update']) > 0) {
 250                          $sql = "UPDATE " . $this->table . " SET ". implode(', ', $arrExtraColumns['update'])  ." WHERE ". KT_escapeFieldName($this->pkName) . " = " . $pk_value." AND ". KT_escapeFieldName($this->fkName)." = ". $value;
 251                          $this->tNG->connection->Execute($sql);
 252                          if ($this->tNG->connection->ErrorMsg()!='') {
 253                              return new tNG_error('TRIGGER_MESSAGE__MTM_SQL_ERROR', array(), array($this->tNG->connection->ErrorMsg(), $sql));
 254                          }
 255                      }
 256                  }
 257              }
 258          }
 259          if (count($values)>0) {
 260              $insertValues = array_diff($values, $oldValues);
 261              if (count($insertValues) >0) {
 262                  foreach ($insertValues as $key => $value) {
 263                      $arrExtraColumns = $this->getExtraColumnsValues($value, $insertValues);
 264                      if (isset($arrExtraColumns['validate']) && is_object($arrExtraColumns['validate'])) {
 265                          return $arrExtraColumns['validate'];
 266                      }
 267                      $extraCols = '';
 268                      $extraVals = '';
 269                      if (count($arrExtraColumns) > 0 && count($arrExtraColumns['cols']) > 0 && count($arrExtraColumns['values']) > 0) {
 270                          $extraCols = ', ' . implode(', ', $arrExtraColumns['cols']);
 271                          $extraVals = ', ' . implode(', ', $arrExtraColumns['values']);
 272                      }
 273                      $value = KT_escapeForSql($value, $pk_type);
 274                      $sql = "INSERT INTO " . $this->table . " ( " . KT_escapeFieldName($this->pkName) . " , " . KT_escapeFieldName($this->fkName) . $extraCols . ") VALUES (" . $pk_value . " , " . $value . $extraVals . ")";
 275                      $ret = $this->tNG->connection->Execute($sql);
 276                      if ($ret === false) {
 277                          return new tNG_error('TRIGGER_MESSAGE__MTM_SQL_ERROR', array(), array($this->tNG->connection->ErrorMsg(), $sql));
 278                      }
 279                  }
 280              }
 281          }
 282          return null;
 283      }

 286       * Return the values for extra columns to use in insert/update SQL;

 287       * Only for PRO version

 293  	function getExtraColumnsValues($fk, $insertValues) {
 294          $arr = array();
 295          if (!in_array($fk, $insertValues)) {
 296              return $arr;    
 297          }    
 298          if (count($this->columns) > 0) {
 299              $arr['cols'] = array();
 300              $arr['values'] = array();
 301              $arr['update'] = array();
 302              $fkReference = $this->fkReference;
 303              $idxReference = "";
 304              if (isset($this->tNG->multipleIdx)) {
 305                  $idxReference = '_' . $this->tNG->multipleIdx;
 306                  $idxReference = preg_quote($idxReference, '/');
 307              }
 308              $fkReference = preg_quote($fkReference, '/');
 309  
 310              foreach ($this->columns as $colName => $arrTmp) {
 311                  $arr['cols'][] = KT_escapeFieldName($colName);
 312                  if ($arrTmp['method'] == 'VALUE') {
 313                      $arr['values'][] = KT_escapeForSql($arrTmp['value'], $arrTmp['type'], false);
 314                      $arr['update'][] =  KT_escapeFieldName($colName) . '=' . $arr['values'][count($arr['values'])-1];
 315                  } else {
 316                      $found = false;
 317                      foreach ($_POST as $key => $val) {
 318                          if (preg_match('/^' . $fkReference .'_'.$colName. '_'.$fk.$idxReference.'$/', $key)) {
 319                              if ($arrTmp['type'] == 'DATE_TYPE') {
 320                                  $val = KT_formatDate2DB($val);
 321                              }
 322                              $arr['values'][] = KT_escapeForSql($val, $arrTmp['type'], false);
 323                              $arr['update'][] =  KT_escapeFieldName($colName) . '=' . $arr['values'][count($arr['values'])-1];
 324                              $found = true;
 325                              break;
 326                          }
 327                      }
 328                      if (!$found && $this->columns[$colName]['default'] != '') {
 329                          $val = KT_DynamicData($this->columns[$colName]['default'], null);
 330                          if ($this->columns[$colName]['type'] == 'DATE_TYPE') {
 331                              $val = KT_formatDate2DB($val);
 332                          }
 333                          $arr['values'][] = KT_escapeForSql($val, $arrTmp['type'], false);
 334                          $arr['update'][] =  KT_escapeFieldName($colName) . '=' . $arr['values'][count($arr['values'])-1];
 335                      }
 336                  }                
 337              }            
 338          }
 339          
 340          return $arr;
 341      }
