  41     	function tNG_FieldCompare(&$tNG) { 
  42          $this->tNG = &$tNG; 
  43          $this->fields = array();
  44      }
  45      /**

  46       * Setter. Sets the information for one comparison

  47       * @param string field 1

  48       * @param string field 2

  49       * @param string operation 

  50       * @param string error message

  51       * @access public

  52       */    
  53  	function addField($name1, $name2, $operation, $error) {
  54          $value1 = $this->tNG->getColumnValue($name1);
  55          $value2 = KT_DynamicData($name2, null);
  56          if ($value2 == $name2) {
  57              $value2 = KT_DynamicData($name2, $this->tNG);
  58          }
  59          if ($this->tNG->getColumnType($name1) == 'DATE_TYPE') {
  60              $value1 = KT_convertDate($value1,  $GLOBALS['KT_db_date_format'].' ' .$GLOBALS['KT_db_time_format_internal'], "yyyy-mm-dd HH:ii:ss");
  61              $value2 = KT_convertDate($value2,  $GLOBALS['KT_db_date_format'].' ' .$GLOBALS['KT_db_time_format_internal'], "yyyy-mm-dd HH:ii:ss");
  62              $value1 = strtotime($value1);
  63              $value2 = strtotime($value2);
  64          } else if ($this->tNG->getColumnType($name1) == 'NUMERIC_TYPE' || $this->tNG->getColumnType($name1) == 'DOUBLE_TYPE') {
  65              $value1 = (float)$value1;    
  66              $value2 = (float)$value2;    
  67          }
  68          $this->fields[] = array(
  69                          '0' => array('name' => $name1, 'value' => $value1, 'type' => $this->tNG->getColumnType($name1)),
  70                          '1' => array('name' => $name2, 'value' => $value2, 'type' => $this->tNG->getColumnType($name1)),
  71                          'operation' => $operation,
  72                          'error' => KT_DynamicData($error, $this->tNG)
  73                          );    
  74      }
  75      /**

  76       * Main class method. Compare the fields and return error object if any

  77       * @return mixt null or tNG error object

  78       * @access public

  79       */    
  80  	function Execute() {
  81          $ret = array();
  82          foreach ($this->fields as $k => $arr) {
  83              switch ($arr['operation']) {
  84                  case '==':
  85                      if ($arr['0']['value'] != $arr['1']['value']) {
  86                          $ret[] = $arr['error'];
  87                      }
  88                      break;
  89                  case '<>':
  90                  case '!=':
  91                      if ($arr['0']['value'] == $arr['1']['value']) {
  92                          $ret[] = $arr['error'];
  93                      }
  94                      break;
  95                  case '<':
  96                      if ($arr['0']['value'] >= $arr['1']['value']) {
  97                          $ret[] = $arr['error'];
  98                      }
  99                      break;
 100                  case '<=':
 101                      if ($arr['0']['value'] > $arr['1']['value']) {
 102                          $ret[] = $arr['error'];
 103                      }
 104                      break;
 105                  case '>':
 106                      if ($arr['0']['value'] <= $arr['1']['value']) {
 107                          $ret[] = $arr['error'];
 108                      }
 109                      break;
 110                  case '>=':
 111                      if ($arr['0']['value'] < $arr['1']['value']) {
 112                          $ret[] = $arr['error'];
 113                      }
 114                      break;
 115                  default :
 116                      break;
 117              }
 118          }
 119          if (count($ret) > 0) {
 120              $ret = new tNG_error("%s", array(implode('<br/>', $ret)), array(''));
 121          } else {
 122              $ret = null;    
 123          }
 124          return $ret;        
 125      }
 126  }
