 123      /**

 124      * Main Class method. Sets the action: remove|replace|block.

 125      * @return mixt object in case of errors or null

 126      * @access public

 127      */
 128  	function Execute() {
 129          $ret = null;
 130          $arr = $this->tNG->columns;
 131          $columns = array();
 132          foreach ($arr as $colName=>$colDetails) {
 133              if ($colDetails['type'] == 'STRING_TYPE') {
 134                  $columns[$colName] = $colDetails['value'];
 135              }
 136          }
 137          if (count($columns) == 0) {
 138              return ;
 139          }
 140          $words = $this->getWords();
 141          if (is_object($this->error)) {
 142              return $this->error;
 143          }
 144          if (count($words) == 0) {
 145              return ;
 146          }
 147          $fieldWithErrors = array();
 148          foreach ($columns as $colName => $value) {
 149              if (!$this->checkValue($colName, $value, $words)) {
 150                  $error = true;
 151                  $fieldWithErrors[] = $colName;
 152              }
 153          }
 154          // action block

 155          if (isset($error) && $this->action == 'block') {
 156              if ($this->errorMsg == '') {
 157                  $ret = new tNG_error('TRIGGER_MESSAGE__CHECK_FORBIDDEN_WORDS', array(implode(', ', $fieldWithErrors)), array());
 158              } else {
 159                  $ret = new tNG_error('%s', array($this->errorMsg), array(''));
 160              }
 161              $errorMsg = KT_getResource('FORBIDDEN_FIELD_ERROR', 'tNG', array());
 162              foreach($fieldWithErrors as $colName) {
 163                  // set field error to $errorMsg

 164                  $ret->setFieldError($colName, '%s', array($errorMsg));
 165                  if ($this->tNG->columns[$colName]['method'] != 'POST') {
 166                      // set composed message as user error

 167                      $ret->addDetails('%s', array($errorMsg), array(''));
 168                  }
 169              }     
 170          // action remove/replace

 171          } else if (isset($error) && $this->action != 'block') {
 172              foreach ($this->replacements as $colName => $value) {
 173                  $this->tNG->setColumnValue($colName, $value);
 174              }
 175          }
 176          return $ret;
 177      }    
 178      /**

 179      * Verify if the given string has forbidden words.

 180      * If this.action is different from 'block', replace|remove all the occurence.

 181      * @param string column name from transaction

 182      * @param string string to test

 183      * @param array forbidden words

 184      * @return boolean true if no forbidden words was found

 185      * @access public

 187  	function checkValue($colName, $value, $words) {
 188          $ret = true;
 189          reset($words);
 190          $regexp = "/\b(?:". implode('|', $words) .")\b/ims";
 191          if (preg_match($regexp, $value, $m)) {
 192              $ret = false;
 193              if ($this->action == 'block') {
 194                  return $ret;
 195              } else if ($this->action == 'replace') {
 196                  $replacement = '$1' . $this->replaceWith . '$2';
 197                  $value = preg_replace($regexp, $replacement, $value);
 198              // remove    

 199              } else {
 200                  $replacement = '$1$2';
 201                  $value = preg_replace($regexp, $replacement, $value);
 202              }
 203          }
 204          $this->replacements[$colName] = $value;
 205          return $ret;
 206      }

 208      /**

 209      * Read the forbidden words file and return the words in an array.

 210      * @return array of forbidden words

 211      * @access public

 212      */
 213  	function getWords() {
 214          $arr = array();
 215          if ($this->table != '' && $this->field != '') {
 216              $sql = 'SELECT '.KT_escapeFieldName($this->field).' AS myfield FROM '.$this->table; 
 217              $rs = $this->tNG->connection->Execute($sql);
 218              if ($this->tNG->connection->errorMsg()!='') {
 219                  $this->error = new tNG_error('BADWORDS_SQL_ERROR', array(), array($this->tNG->connection->errorMsg(), $sql));
 220                  return $arr;
 221              }
 222              while (!$rs->EOF) {
 223                  $arr[] = trim($rs->Fields('myfield'));
 224                  $rs->MoveNext();
 225              }
 226              $rs->Close();
 227          } else {
 228              $file = dirname(__FILE__) . DIRECTORY_SEPARATOR . 'tNG_ChkForbiddenWords.txt';
 229                 if (!file_exists($file)) {
 230                     $this->error = new tNG_error('BADWORDS_FILE_ERROR', array(), array($file));
 231                  return $arr;
 232                 }
 233                 if ($fd = @fopen($file, 'rb')) {
 234                  while (!feof ($fd)) {
 235                       $tmp = fgets($fd, 4096);
 236                       $tmp = addcslashes($tmp, '/.()[]{}|^$');
 237                       if (trim($tmp) != '') {
 238                            $arrTmp = explode(',', $tmp);
 239                            foreach ($arrTmp as $k => $v) {
 240                                 $arr[] = trim($v);
 241                            }
 242                       }
 243                  }
 244                  fclose ($fd);
 245                 } else {
 246                  $this->error = new tNG_error('BADWORDS_FILE_ERROR', array(), array($file));
 247                  return $arr;
 248                 }
 249          }
 250          return $arr;
 251      }
