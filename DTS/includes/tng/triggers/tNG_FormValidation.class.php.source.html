 113  	function mask2regexp($txt) {
 114          $txt = preg_replace('/([-\/\[\]()\*\+])/', "\\1", $txt);
 115          $txt = preg_quote($txt);
 116          $txt = str_replace('\?', '?', $txt);
 117          $txt = str_replace('?', '.', $txt);
 118          $txt = str_replace('9', '\d', $txt);
 119          $txt = str_replace('X', '\w', $txt);
 120          $txt = str_replace('A', '[A-Za-z]', $txt);
 121          $txt = str_replace('/', '\/', $txt);
 122          $txt = '/^' . $txt . '$/';
 123          return $txt;
 124      }
