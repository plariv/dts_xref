 295      /**

 296       * main class method. Make the download or display the appropiate error.

 297       * call the garbagecollector;

 298       * @return nothing

 299       * @access public

 300       */ 
 301      function Execute() {
 302          $this->garbageCollector();        
 303          if (!$this->isDownload()) {
 304              return;
 305          }  
 306          
 307          //check counter    

 308          if (!$this->checkCounter()) {
 309              return $this->error;
 310          }    
 311          
 312          $fileHash = $this->downloadHash;
 313          $folder =  $fileHash['folder'];
 314          if (substr($folder, 0, strlen($this->relPath))==$this->relPath) {
 315              $folder = substr($folder, strlen($this->relPath));
 316          }    
 317          $fileName = $fileHash['fileName'];
 318          $folder = KT_realpath('../../../' . $folder);
 319          $absPath = KT_realpath($folder . $fileName, false);
 320          
 321          $fd = @fopen($absPath, "rb");    
 322          if (!$fd || !is_file($absPath)) {
 323              $this->setError(new tNG_error('ERR_DOWNLOAD_FILE_NO_READ', array($fileName), array($absPath)));
 324              return $this->error;
 325          }
 326          
 327          //increment counter;

 328          $this->incrementCounter();
 329          if ($this->hasError()) {
 330              return $this->error;
 331          }
 332          
 333          $nameForDownload = KT_pathInfo($fileName);
 334          $mime_type = (function_exists('mime_content_type'))? mime_content_type($absPath): 'application/octet-stream';
 335          header('Content-type: '.$mime_type);
 336          header('Cache-control: private');
 337          header('Content-Length: ' . filesize($absPath));
 338          header('Content-disposition: attachment; filename="' . $nameForDownload['basename'] . '";');
 339          if (!$fd) {
 340              $this->setError(new tNG_error('ERR_DOWNLOAD_FILE_NO_READ', array($fileName), array($absPath)));
 341              return $this->error;
 342          }
 343          do { 
 344                 echo fread($fd, 8192); 
 345                 flush();
 346              usleep(10000);
 347          } while(!feof($fd) && connection_status()==0); 
 348          fclose($fd);
 349          exit;
 350      }

 352       * Check the counter

 353       * @return boolean true if the download can be done

 356      function checkCounter() {
 357          if (($this->maxCounterFieldMtm == '' && $this->maxCounterValueMtm == '') || $this->counterFieldMtm == '' || count($this->fkMtm)==0 || count($this->pkMtm)==0 || $this->maxCounterValueMtm == -1) {
 358              return true;
 359          }
 360          $fileHash = $this->downloadHash;
 361                  
 362          if (!isset($fileHash['fkMtm']) || $fileHash['fkMtm'] == '') {
 363              $this->setError(new tNG_error('INCREMENTER_ERROR_FK', array(), array($this->fkMtm['field'])));    
 364              return null;
 365          }
 366          $this->fkMtm['value'] = $fileHash['fkMtm'];
 367          if (!isset($fileHash['pkMtm']) || $fileHash['pkMtm'] == '') {
 368              $this->setError(new tNG_error('INCREMENTER_ERROR_PK', array(), array($this->pkMtm['field'])));    
 369              return null;
 370          }
 371          $this->pkMtm['value'] = $fileHash['pkMtm'];
 372              
 373          $sql = 'SELECT '.KT_escapeFieldName($this->counterFieldMtm).' AS currval ';
 374          if ($this->maxCounterFieldMtm != '') {
 375              $sql .= ', '.KT_escapeFieldName($this->maxCounterFieldMtm).' AS maxval ';
 376          }
 377          $sql .= ' FROM '.$this->tableMtm.' WHERE '.KT_escapeFieldName($this->pkMtm['field']).' = '.KT_escapeForSql($this->pkMtm['value'], $this->pkMtm['type'], false);
 378          $sql .= ' AND '. KT_escapeFieldName($this->fkMtm['field']).' = '.KT_escapeForSql($this->fkMtm['value'], $this->fkMtm['type'], false);
 379          
 380          $rs = $this->conn->Execute($sql);
 381          if ($rs === false) {
 382              $this->setError(new tNG_error('CHECK_COUNTER_ERROR', array(), array($this->conn->ErrorMsg(), $sql)));
 383              return ;
 384          }
 385          $maxCounter = $this->maxCounterValueMtm;
 386          if ($this->maxCounterFieldMtm != '') {
 387              $maxCounter = (int)$rs->Fields('maxval');
 388              if ($maxCounter == -1) {
 389                  return true;
 390              }
 391          }
 392          if ( (int)$rs->Fields('currval') >=  $maxCounter) {
 393              $this->setError(new tNG_error('CHECK_COUNTER_ERROR_MAX', array($maxCounter), array($maxCounter)));    
 394              return;        
 395          }
 396          return true;
 397      }

 410              $ret = $this->conn->Execute($sql);

SET '.KT_escapeFieldName($this->counterFieldMtm).' = '.KT_escapeFieldName($this->counterFieldMtm).'+ 1 WHERE '.KT_escapeFieldName($this->pkMtm['field']).' = '.KT_escapeForSql($this->pkMtm['value'], $this->pkMtm['type'], false) . ' AND ' . KT_escapeFieldName($this->fkMtm['field']).' = '.KT_escapeForSql($this->fkMtm['value'], $this->fkMtm['type'], false);

 525      function garbageCollector() {
 526          if (!isset($_SESSION['tng_download'])) {
 527              return ;
 528          }
 529          // clear old session values;

 530          foreach ($_SESSION['tng_download'] as $id => $hash) {
 531              if (isset($hash['properties']) && $hash['properties']['time'] < time()-60*5) {
 532                  unset($_SESSION['tng_download'][$id]);
 533              }
 534          }
