 136  	function Execute() {
 137          if ($this->tNG->getTransactionType() == "_import") {
 138              $this->tNG->uploadObj = &$this;
 139          }
 140          $ret = null;
 141          if ($this->dbFieldName != '') {
 142              $oldFileName = $this->tNG->getSavedValue($this->dbFieldName);
 143              $saveFileName = $this->tNG->getColumnValue($this->dbFieldName);
 144              if ($this->tNG->getColumnType($this->dbFieldName) != 'FILE_TYPE') {
 145                  $errObj = new tNG_error('FILE_UPLOAD_WRONG_COLTYPE', array(), array($this->dbFieldName));
 146                  $errObj->addFieldError($this->dbFieldName, 'FILE_UPLOAD_WRONG_COLTYPE_D', array($this->dbFieldName));
 147                  return $errObj;
 148              }
 149          } else {
 150              $oldFileName = KT_DynamicData($this->renameRule, $this->tNG, '', true);
 151              if (isset($this->tNG->multipleIdx)) {
 152                  $saveFileName = @$_FILES[$this->formFieldName."_".$this->tNG->multipleIdx]['name'];
 153              } else {
 154                  $saveFileName = @$_FILES[$this->formFieldName]['name'];
 155              }
 156          }
 157          $this->dynamicFolder = KT_DynamicData($this->folder, $this->tNG, '', false);
 158          
 159          $arrArgs = array();
 160          $autoRename = false;
 161          switch ($this->rename) {
 162              case 'auto':
 163                  $autoRename = true;
 164                  break;
 165              case 'none':
 166                  break;
 167              case 'custom':
 168                  $path_info = KT_pathinfo($saveFileName);
 169                  $arrArgs = array('KT_name' => $path_info['filename'], 'KT_ext' => $path_info['extension']);
 170                  $saveFileName = KT_DynamicData($this->renameRule, $this->tNG, '', false, $arrArgs);
 171                  break;
 172              default:
 173                  die('INTERNAL ERROR: Unknown upload rename method.');
 174          }
 175          
 176                  if(tNG_isFileInsideBaseFolder($this->folder, $saveFileName) === false) {
 177                    $baseFileName = dirname(KT_realPath($this->dynamicFolder . $saveFileName, false));
 178                    return new tNG_error("FOLDER_DEL_SECURITY_ERROR", array(), array($baseFileName, tNG_getBaseFolder($this->folder)));
 179                  }
 180          
 181          // Upload File

 182          $fileUpload = new KT_fileUpload();
 183          if (isset($this->tNG->multipleIdx)) {
 184              $fileUpload->setFileInfo($this->formFieldName."_".$this->tNG->multipleIdx);
 185          } else {
 186              $fileUpload->setFileInfo($this->formFieldName);
 187          }
 188          $fileUpload->setFolder($this->dynamicFolder);
 189          $fileUpload->setRequired(false);
 190          $fileUpload->setAllowedExtensions($this->allowedExtensions);
 191          $fileUpload->setAutoRename($autoRename);
 192          $fileUpload->setMaxSize($this->maxSize);
 193          $this->uploadedFileName = $fileUpload->uploadFile($saveFileName, $oldFileName);
 194          
 195          $updateDB = basename($this->uploadedFileName);
 196          if ($fileUpload->hasError()) {
 197              $arrError = $fileUpload->getError();
 198              $errObj = new tNG_error('FILE_UPLOAD_ERROR', array($arrError[0]), array($arrError[1]));
 199              if ($this->dbFieldName != '') {
 200                  $errObj->addFieldError($this->dbFieldName, '%s', array($arrError[0]));
 201              }
 202              $ret = $errObj;
 203          } else {
 204              $this->dynamicFolder = KT_realpath($this->dynamicFolder);
 205              if ($this->uploadedFileName == "") {
 206                  //Check if for update we need to rename file

 207                  if ($this->rename == "custom") {
 208                      $path_info = KT_pathinfo($oldFileName);
 209                      $arrArgs['KT_ext'] = $path_info['extension'];
 210                  }
 211                  $tmpFileName = KT_DynamicData($this->renameRule,$this->tNG,'',false, $arrArgs);
 212                  if ($tmpFileName != "" && $oldFileName != "" && $tmpFileName != $oldFileName) {
 213                      if (file_exists($this->dynamicFolder.$oldFileName)) {
 214                          if (@rename($this->dynamicFolder.$oldFileName, $this->dynamicFolder.$tmpFileName) === true) {
 215                              $this->uploadedFileName = $tmpFileName;
 216                              $updateDB = basename($this->uploadedFileName);
 217                          } else {
 218                              $ret = new tNG_error('FILE_UPLOAD_RENAME', array(), array($this->dynamicFolder.$oldFileName, $this->dynamicFolder.$tmpFileName));
 219                          }
 220                      }
 221                  }
 222              }
 223  
 224              if ($ret === null) {
 225                  if ($this->tNG->getTransactionType() == "_insert" || $this->tNG->getTransactionType() == "_multipleInsert") {
 226                      $this->tNG->registerTrigger('ERROR', 'Trigger_Default_RollBack', 1, $this);
 227                  }
 228                  
 229                  $this->deleteThumbnails($this->dynamicFolder .'thumbnails'.DIRECTORY_SEPARATOR, $oldFileName);
 230                  if ($this->uploadedFileName != '') {
 231                      $this->deleteThumbnails($this->dynamicFolder.'thumbnails'.DIRECTORY_SEPARATOR, $this->uploadedFileName);
 232                  }
 233  
 234                  if ($this->dbFieldName != '' && $this->uploadedFileName != "") {
 235                      $ret = $this->tNG->afterUpdateField($this->dbFieldName, $updateDB);
 236                  }
 237              }
 238              if ($ret === null && $this->dbFieldName != "") {
 239                  $this->tNG->setRawColumnValue($this->dbFieldName,$updateDB);
 240              }
 241          }
 242          $this->errObj = $ret;
 243          return $ret;
 244      }
