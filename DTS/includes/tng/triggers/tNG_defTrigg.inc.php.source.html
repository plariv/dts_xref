 106  /**

 107   * Login trigger. Check if the user has the proper credentials or autolog on the user based on the cookie;

 108   * @param object tNG transaction object

 109   * @return mix null or error object

 110   * @access public

 111   */
 112  function Trigger_Login_CheckLogin(&$tNG) {
 113      // check user_name

 114      if (!is_object($tNG->transactionResult)) {
 115          $errObj = new tNG_error("LOGIN_FAILED", array(), array());
 116          $errObj->setFieldError("kt_login_user", "LOGIN_INVALID_user_name", array());
 117          return $errObj;
 118      }
 119      if ($tNG->transactionResult->RecordCount() > 1) {
 120          $errObj = new tNG_error("LOGIN_FAILED_MANYRECORDS", array(), array());
 121          $errObj->setFieldError("kt_login_user", "LOGIN_FAILED_MANYRECORDS_FIELDERR", array());
 122          return $errObj;
 123      }
 124  
 125      // check password 

 126      switch ($tNG->loginType) {
 127          case 'form':
 128              $db_password = $tNG->transactionResult->Fields('kt_login_password');
 129              $password_enc = $tNG->getColumnValue('kt_login_password');
 130              if ($GLOBALS['tNG_login_config']['password_encrypt'] == "true") {
 131                  $password_enc = tNG_encryptString($password_enc);
 132              }
 133              if ($db_password != $password_enc) {
 134                  $errObj = new tNG_error("LOGIN_FAILED", array(), array());
 135                  $errObj->setFieldError("kt_login_password", "LOGIN_INVALID_PASSWORD", array());
 136                  return $errObj;
 137              }
 138              break;
 139          case 'cookie':
 140              $db_password_enc  = tNG_encryptString($tNG->transactionResult->Fields('kt_login_password'));
 141              $password_enc_cookie = $tNG->getColumnValue('kt_login_test');
 142              if ($db_password_enc != $password_enc_cookie) {
 143                  return new tNG_error("LOGIN_FAILED", array(), array());
 144              }
 145              break;    
 146          case 'activation':
 147              $random_key_trans = $tNG->getColumnValue('kt_login_random');
 148              $random_key_db = $tNG->transactionResult->Fields($GLOBALS['tNG_login_config']['randomkey_field']);
 149              if ($random_key_trans != $random_key_db) {
 150                  return new tNG_error("LOGIN_FAILED", array(), array());
 151              }
 152              break;
 153      }
 154      return null;
 155  }

 185  function Trigger_Login_AddDynamicFields(&$tNG) {
 186      // register all the columns from the recordset as transaction columns (to be available later)

 187      $rs = $tNG->transactionResult;
 188  
 189      $tNG->addColumn("log_userkey", "STRING_TYPE", "VALUE", $rs->Fields("log_userkey"));        
 190      $tNG->addColumn("kt_login_user", "STRING_TYPE", "VALUE", $rs->Fields("kt_login_user"));

 193          $tNG->addColumn("kt_login_level", "STRING_TYPE", "VALUE", $rs->Fields($GLOBALS['tNG_login_config']['level_field']));        
 194      }    
 195      
 196      // must add {kt_login_redirect}

 197      $login_redirect = '';
 198      switch ($tNG->loginType) {
 199          case 'form':
 200          case 'activation':
 201              $login_redirect = "";
 202              if (isset($_SESSION['KT_denied_pageuri']) && is_array($_SESSION['KT_denied_pagelevels'])) {
 203                  // if restrict using levels is used

 204                  if ($GLOBALS['tNG_login_config']['level_field'] != "" ) {
 205                      $level_column = $GLOBALS['tNG_login_config']['level_field'];
 206                      $level_value = $tNG->transactionResult->Fields($level_column);
 207                      
 208                      $arr_allowed_levels = $_SESSION['KT_denied_pagelevels'];
 209                      // check if the current user can be redirected to previously denied page

 210                      if (count($arr_allowed_levels) > 0) {
 211                          if (in_array($level_value, $arr_allowed_levels)) {
 212                              $login_redirect = $_SESSION['KT_denied_pageuri'];
 213                          } else {
 214                              // redirect to the denied page will result into another denied page, so don't redirect

 215                          }
 216                      } else {
 217                          // levels array has no elements - acccess is allowed to all logged users

 218                          $login_redirect = $_SESSION['KT_denied_pageuri'];
 219                      }
 220                  } else {
 221                      // no levels restriction is used, so we can redirect to previously denied page

 222                      $login_redirect = $_SESSION['KT_denied_pageuri'];    
 223                  }
 224                  
 225                  unset($_SESSION['KT_denied_pageuri']);
 226                  KT_unsetSessionVar('KT_denied_pageuri');
 227                  unset($_SESSION['KT_denied_pagelevels']);
 228                  KT_unsetSessionVar('KT_denied_pagelevels');
 229              }
 230              
 231              if ($login_redirect == "") {
 232                  $relPath = '';
 233                  if (isset($tNG->dispatcher) && isset($tNG->dispatcher->relPath)) {
 234                      $relPath = KT_makeIncludedURL($tNG->dispatcher->relPath);
 235                  }
 236                  if ($GLOBALS['tNG_login_config']['level_field'] != "" ) {
 237                      $level_column = $GLOBALS['tNG_login_config']['level_field'];
 238                      $level_value = $tNG->transactionResult->Fields($level_column);
 239                      if (is_array($GLOBALS['tNG_login_config_redirect_success']) && isset($GLOBALS['tNG_login_config_redirect_success'][$level_value]) AND $GLOBALS['tNG_login_config_redirect_success'][$level_value] != "") {
 240                          $login_redirect = $relPath. $GLOBALS['tNG_login_config_redirect_success'][$level_value];
 241                      } else {
 242                          $login_redirect = $relPath. $GLOBALS['tNG_login_config']['redirect_success'];
 243                      }
 244                  } else {
 245                      $login_redirect =  $relPath. $GLOBALS['tNG_login_config']['redirect_success'];
 246                  }
 247              }    
 248              break;
 249          case 'cookie':
 250              // cookie login doesn't use redirect    

 251              break;
 252      }
 253      $tNG->addColumn("kt_login_redirect", "STRING_TYPE", "VALUE", $login_redirect);
 254      return null;
 255  }

 491      if ($rs->RecordCount() == 0) {
 492          return null;    
 493      }
 494      $sql = 'UPDATE '.$table.' SET '. KT_escapeFieldName($dateout) .' =  '.KT_escapeForSql($now, 'DATE_TYPE', false).' WHERE ' . KT_escapeFieldName($pk) .'='. $rs->Fields('pk');

 497          $ret = new tNG_error("LOGIN_LOGGER_ERROR", array(), array($tNG->connection->ErrorMsg(), $sql));

 689  function Trigger_Activation_Check(&$tNG) {
 690      if ($GLOBALS['tNG_login_config']['activation_field'] == "") {
 691          return new tNG_error("ACTIVATION_NOT_ENABLED", array(), array());
 692      }
 693      if ($GLOBALS['tNG_login_config']['email_field'] == "") {    
 694          return new tNG_error("ACTIVATION_NO_EMAIL", array(), array());
 695      }
 696      if ($tNG->getTable() != $GLOBALS['tNG_login_config']['table']) {
 697          return new tNG_error("ACTIVATION_WRONG_TABLE", array(), array());
 698      }
 699      if ($tNG->getPrimaryKey() != $GLOBALS['tNG_login_config']['pk_field']) {
 700          return new tNG_error("ACTIVATION_WRONG_PK", array(), array());
 701      }
 702      if (!isset($tNG->columns[$GLOBALS['tNG_login_config']['activation_field']])) {
 703          return new tNG_error("ACTIVATION_NO_ACTIVE_FIELD", array(), array());
 704      }
 705      // build the sql string to check 

 706      if ($GLOBALS['tNG_login_config']['randomkey_field']!="") {
 707          $random_key = KT_getRealValue("GET","kt_login_random");
 708          if($random_key=="") {    
 709              return new tNG_error("ACTIVATION_NO_PARAM_RANDOM", array(), array());
 710          }
 711          $random_key = KT_escapeForSql($random_key, "STRING_TYPE");
 712          $pk_value = KT_escapeForSql($tNG->getPrimaryKeyValue(), $GLOBALS['tNG_login_config']['pk_type']);
 713          $sql = "SELECT ". KT_escapeFieldName($tNG->getPrimaryKey()) . ", ".  KT_escapeFieldName($GLOBALS['tNG_login_config']['activation_field']) . " FROM " .  $tNG->getTable() . " WHERE " .  KT_escapeFieldName($tNG->getPrimaryKey()) . "=" .  $pk_value . " AND ".  KT_escapeFieldName($GLOBALS['tNG_login_config']['randomkey_field']) . "=" . $random_key ;
 714          $rs = $tNG->connection->Execute($sql);
 715          if (!is_object($rs)) {
 716              return new tNG_error("LOGIN_RECORDSET_ERR", array(), array());
 717          }
 718      } else {
 719          $email_value = KT_getRealValue("GET","kt_login_email");
 720          if($email_value=="") {
 721              return new tNG_error("ACTIVATION_NO_PARAM_EMAIL", array(), array());
 722          }
 723          $email_value = KT_escapeForSql($email_value, "STRING_TYPE");
 724          $pk_value = KT_escapeForSql($tNG->getPrimaryKeyValue(), $GLOBALS['tNG_login_config']['pk_type']);
 725          $sql = "SELECT ".  KT_escapeFieldName($tNG->getPrimaryKey()) . ", ".  KT_escapeFieldName($GLOBALS['tNG_login_config']['activation_field']) . " FROM " .  $tNG->getTable() . " WHERE " .  KT_escapeFieldName($tNG->getPrimaryKey()) . "=" . $pk_value . " AND ".  KT_escapeFieldName($GLOBALS['tNG_login_config']['email_field']) . "=" . $email_value ;
 726          $rs = $tNG->connection->Execute($sql);
 727          if (!is_object($rs)) {
 728              return new tNG_error("LOGIN_RECORDSET_ERR", array(), array());
 729          }
 730      }
 731      if ($rs->RecordCount() == 0 ) {
 732          return new tNG_error("ACTIVATION_NO_RECORDS", array(), array());
 733      }
 734      if ($rs->RecordCount() != 1 ) {
 735          return new tNG_error("ACTIVATION_TOOMANY_RECORDS", array(), array());
 736      }
 737      // check for inactive

 738      if ($rs->Fields($GLOBALS['tNG_login_config']['activation_field']) != 0) {
 739          return new tNG_error("ACTIVATION_ALREADY_ACTIVE", array(), array());
 740      }
 741      // register the AFTER trigger

 742      $tNG->registerTrigger("AFTER", "Trigger_Activation_Login", -1);
 743      return null;
 744  }

1093   * Only for PRO version

1095   * @param string reference for the key in session for the rename rule

1096   * @return mix null or error object

1097   * @access private

1099  function Trigger_MultipleUploadRename(&$tNG, $uploadObj) {
1100      $reference = $uploadObj->reference;
1101      $ret = null;
1102      if (!isset($_SESSION['tng_upload'][$reference]) || count($_SESSION['tng_upload'][$reference]) == 0) {
1103          return $ret;    
1104      }
1105      $pk = $tNG->getPrimaryKeyValue();
1106      
1107      // make the rename on insert    

1108      if ($tNG->transactionType == '_insert') {
1109          $hash = $_SESSION['tng_upload'][$reference]['properties'];
1110          if (isset($tNG->multipleIdx) && $tNG->multipleIdx > 0 ) {
1111                $hashFile = $_SESSION['tng_upload'][$reference]['files'][$tNG->multipleIdx];
1112          } else {
1113                $hashFile = $_SESSION['tng_upload'][$reference]['files'][1];
1114          }
1115          if (isset($hash['pkRule']) && isset($hashFile['pk']) && isset($hash['folderRule'])) {
1116              $folderExistent = str_replace('{'.$hash['pkRule'].'}', $hashFile['pk'], $hash['folderRule']);
1117              if (!file_exists($folderExistent)) {
1118                  if (isset($tNG->multipleIdx) && $tNG->multipleIdx > 0 ) {
1119                              $_SESSION['tng_upload'][$reference]['files'][$tNG->multipleIdx] = null;
1120                  } else {
1121                              $_SESSION['tng_upload'][$reference]['files'][1] = null;
1122                  }
1123                  return $ret;    
1124              }
1125              
1126              $arr = explode('/', $folderExistent);
1127              if (substr($folderExistent, -1, 1) == '/' || substr($folderExistent, -1, 1) == '\\') {
1128                  array_pop($arr);
1129              }
1130              $folderExistent = array_pop($arr);
1131      
1132              $folder = str_replace('{'.$hash['pkRule'].'}', $pk, $hash['folderRule']);
1133              $arr = explode('/', $folder);
1134              if (substr($folder, -1, 1) == '/' || substr($folder, -1, 1) == '\\') {
1135                  array_pop($arr);
1136              }
1137              $new = array_pop($arr);
1138              $f = implode('/', $arr) . '/'; 
1139              
1140              $fld = new KT_folder();
1141              $fld->renameFolder($f, $folderExistent, $new);
1142              if ($fld->hasError()) {
1143                  $err = $fld->getError();
1144                  return new tNG_error("%s", array($err[0]), array($err[1]));
1145              } else {

1147                              $_SESSION['tng_upload'][$reference]['files'][$tNG->multipleIdx] = null;
1148                  } else {
1149                              $_SESSION['tng_upload'][$reference]['files'][1] = null;
1150                  }
1151              }
1152          }
1153      }
1154      // cleanup session on update;

1155      if ($tNG->transactionType == '_update') {
1156          if (isset($tNG->multipleIdx) && $tNG->multipleIdx > 0 ) {
1157                $_SESSION['tng_upload'][$reference]['files'][$tNG->multipleIdx] = null;
