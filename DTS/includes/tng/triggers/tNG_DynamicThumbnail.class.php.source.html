 299  	function getPopupLink() {
 300          $this->garbageCollector();
 301          $ret = "#";
 302          if ($this->popupWidth == '' || $this->popupHeight == '') {
 303              return $ret;
 304          }
 305          $fileName = $this->getFileName();
 306          if ($fileName !== false) {
 307              $folder = $this->folder;
 308              $folder = KT_TransformToUrlPath($folder, true);
 309              if (!isset($_SESSION['tng_popup'])) {
 310                  $_SESSION['tng_popup'] = array();
 311              }
 312              if (!isset($_SESSION['tng_popup'][$this->id])) {
 313                  $_SESSION['tng_popup'][$this->id] = array();
 314              }
 315              if (!isset($_SESSION['tng_popup'][$this->id]['files'])) {
 316                  $_SESSION['tng_popup'][$this->id]['files'] = array();
 317              }
 318              
 319              $ret = $this->relpath . 'includes/tng/pub/popup_image.php?id=' . rawurlencode($this->id) . '&n=' . count($_SESSION['tng_popup'][$this->id]['files']);
 320              
 321              $siteRootFilename = $folder . $fileName;
 322              //$siteRootFilename = substr($siteRootFilename, strlen($this->relpath));

 323              $siteRootFilename = substr($this->currentfolder, strlen(KT_getSiteRoot().'/'))  . $siteRootFilename;
 324                  
 325              if (!$this->isSetted) {    
 326                  $arrP = array();
 327                  if ($this->watermarkImage != '') {
 328                      $arrP['watermark'] = KT_realpath($this->watermarkImage,false);
 329                      $arrP['watermarkAlpha'] = $this->watermarkAlpha;
 330                      $arrP['watermarkAlignment'] = $this->watermarkAlignment;
 331                      $arrP['watermarkResize'] = $this->watermarkResize;
 332                  }
 333                  if ($this->popupWidth != '' && $this->popupHeight != '') {
 334                      $arrP['popupwidth'] = $this->popupWidth;
 335                      $arrP['popupheight'] = $this->popupHeight;
 336                      $arrP['fitToWindow'] = $this->fitToWindow;
 337                  }
 338                  $arrP['popupNavigation'] = $this->popupNavigation;
 339                  $arrP['popupWatermark'] = $this->popupWatermark;
 340                  $arrP['time'] = time();
 341                  $_SESSION['tng_popup'][$this->id]['properties'] = $arrP;
 342                  $this->isSetted = true;
 343              }
 344              $_SESSION['tng_popup'][$this->id]['files'][]= array('fullfilename' => $siteRootFilename, 'filename' =>$fileName);
 345          }
 346          return $ret;
 347      }
 348      /**

 349       * Garbage collector. Clean the hash from session where the creation time is older than 30 minutes;

 350       * @return nothing

 351       * @access public

 352       */ 
 353      function garbageCollector() {
 354          if (!isset($_SESSION['tng_popup'])) {
 355              return ;
 356          }
 357          // clear old session values;

 358          foreach ($_SESSION['tng_popup'] as $id => $hash) {
 359              if (isset($hash['properties']) && $hash['properties']['time'] < time()-60*5) {
 360                  unset($_SESSION['tng_popup'][$id]);
 361              }
 362          }    
 363      }

 390  	function getFileName() {
 391          $ret = false;
 392          $relpath = $this->relpath;
 393          $folder = KT_TransformToUrlPath($this->folder);
 394          $fileName = KT_DynamicData($this->renameRule, null);
 395          $fileName = KT_TransformToUrlPath($fileName, false);
 396          $fullFileName = KT_realpath($folder . $fileName, false);
 397          // security

 398          $base = KT_realpath($folder, true);
 399          if (substr($fullFileName, 0, strlen($base)) != $base) {
 400              return false;
 401          }
 402          if(file_exists($fullFileName) && is_file($fullFileName)) {
 403              $ret = $fileName;
 404              if (substr($ret,0,1) == '/') {
 405                  $ret = substr($ret,1);
 406              }
 407          }
 408          return $ret;
