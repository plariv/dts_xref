 202       /**

 203       * Getter. Gets the link to the upload page

 204       * sets in session the neccesary info

 205       * @return string

 206       * @access public

 207       */
 208      function getUploadLink() {
 209          if (!$this->checkSecurity()) {
 210              return '';    
 211          }        
 212          
 213          $siteroot = KT_realpath($this->relPath, true);
 214          $uploadFolder = KT_realpath($this->baseFolder, true);
 215          $this->baseFolder = $this->relPath . substr($uploadFolder, strlen($siteroot));
 216          $url = '';
 217          if (!isset($_SESSION['tng_upload'])) {
 218              $_SESSION['tng_upload'] = array();
 219          }
 220          $this->noOfCalls++;
 221          if ($this->noOfCalls == 1) {
 222              $this->garbageCollector(); 
 223              if (!isset($_POST[$this->reference.'_'.$this->noOfCalls]) || !isset($_SESSION['tng_upload'][$this->reference]['files'])) {
 224                  $_SESSION['tng_upload'][$this->reference] = array();
 225                  $_SESSION['tng_upload'][$this->reference]['properties'] = array();
 226                  $_SESSION['tng_upload'][$this->reference]['properties']['maxSize'] = $this->maxSize;
 227                  $_SESSION['tng_upload'][$this->reference]['properties']['maxFiles'] = $this->maxFiles;
 228                  $_SESSION['tng_upload'][$this->reference]['properties']['allowedExtensions'] = $this->allowedExtensions;
 229                  $_SESSION['tng_upload'][$this->reference]['properties']['relPath'] = $this->relPath;
 230                  $_SESSION['tng_upload'][$this->reference]['properties']['connName'] = $this->connName;
 231                  $_SESSION['tng_upload'][$this->reference]['properties']['time'] = time();      
 232                  $_SESSION['tng_upload'][$this->reference]['files'] = array();
 233              }
 234              if (isset($_SESSION['tng_upload'][$this->reference]['files'])){
 235                  $tmpArr = array();
 236                  for($i=1;$i<=count($_SESSION['tng_upload'][$this->reference]['files']);$i++) {
 237                      if ($_SESSION['tng_upload'][$this->reference]['files'][$i] !== null) {
 238                          $tmpArr[count($tmpArr)+1] = $_SESSION['tng_upload'][$this->reference]['files'][$i];
 239                      }
 240                  }
 241                  $_SESSION['tng_upload'][$this->reference]['files'] = $tmpArr;
 242              }
 243          }
 244          
 245          $pk = KT_DynamicData($this->pkRenameRule, null);
 246          if ($pk == $this->pkRenameRule) {
 247              $pk = '';
 248          }
 249          if (!isset($_POST[$this->reference.'_'.$this->noOfCalls]) || !isset($_SESSION['tng_upload'][$this->reference]['files'][$this->noOfCalls])) {
 250            $_SESSION['tng_upload'][$this->reference]['files'][$this->noOfCalls] = array();
 251            if ($pk == '') {
 252              $_SESSION['tng_upload'][$this->reference]['files'][$this->noOfCalls]['pk'] = str_replace('.', '_', uniqid("", true));
 253              if ($this->noOfCalls == 1) {
 254                $_SESSION['tng_upload'][$this->reference]['properties']['pkRule'] = $this->pk;
 255                $_SESSION['tng_upload'][$this->reference]['properties']['folderRule'] = $this->baseFolder . $this->folder;
 256              }
 257            } else {
 258              $_SESSION['tng_upload'][$this->reference]['files'][$this->noOfCalls]['pk'] = $pk;    
 259            }
 260            $_SESSION['tng_upload'][$this->reference]['files'][$this->noOfCalls]['folder'] = str_replace('{'.$this->pk.'}', $_SESSION['tng_upload'][$this->reference]['files'][$this->noOfCalls]['pk'], $this->baseFolder . $this->folder);
 261          }
 262          
 263          $url = $this->relPath . 'includes/tng/pub/multiple_upload.php';
 264          $url = KT_addReplaceParam($url, $this->reference, $this->noOfCalls);
 265          return $url;
 266      }

 272      function checkSecurity() {
 273          // security

 274          $base = KT_realpath($this->baseFolder, true);
 275          $fullFolder = KT_realpath(str_replace('{'.$this->pk.'}', 1, $this->baseFolder . $this->folder));
 276          if (substr($fullFolder, 0, strlen($base)) != $base) {
 277              return false;
 278          }
 279          if (strpos($fullFolder, '{') !== false) {
 280              return false;
 281          }
 282          return true;    
 283      }
 284       /**

 285       * garbage collector. remove from session entries older than 30 minutes;

 286       * @return string

 287       * @access public

 288       */
 289      function garbageCollector() {
 290           // cleanup orphan folders

 291           $arr = explode('/', $this->baseFolder . $this->folder);
 292          if (substr($this->folder, -1, 1) == '/' || substr($this->folder, -1, 1) == '\\') {
 293              array_pop($arr);
 294          }
 295          $new = array_pop($arr);
 296          $f = implode('/', $arr) . '/'; 
 297          
 298          $fld = new KT_folder();
 299          $arr = $fld->readFolder($f, false);
 300          //print_r($arr); 

 301          if (isset($arr['folders']) && count($arr['folders'])) {
 302              foreach ($arr['folders'] as $entry) {
 303                  if (preg_match("/^(.*)\w{14}_\w{8}(.*)/is", $entry['name'])) {
 304                      $fld = new KT_folder();
 305                      $files = $fld->readFolder($f . $entry['name']);
 306                      if (isset($files['files']) && count($files['files']) > 0) {
 307                          foreach ($files['files'] as $file) {
 308                              if (filectime($f . $entry['name'] . '/' . $file['name']) < time()-60*30 ) {
 309                                  $fld->deleteFolderNR($f . $entry['name']);
 310                                  break;    
 311                              }
 312                          }                        
 313                      } else {
 314                          $fld->deleteFolderNR($f . $entry['name']);
 315                      }                
 316                  }
 317              }
 318          }                
 319          if (!isset($_SESSION['tng_upload'])) {
 320              return ;    
 321          }
 322          // clear old session values;

 323          foreach ($_SESSION['tng_upload'] as $id => $hash) {
 324              if (isset($hash['properties']) && $hash['properties']['time'] < time()-60*30) {
 325                    unset($_SESSION['tng_upload'][$id]);
 326              }
 327          }
 328      }
