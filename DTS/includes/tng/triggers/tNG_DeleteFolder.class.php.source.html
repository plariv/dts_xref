  69          function setBaseFolder($baseFolder) {
  70            $pos = strpos($baseFolder, '{');
  71            if ($pos !== false) {

  94          /**

  95           * Method that checks if all dynamic data placeholders have been replaced.

  96           *

  97           * @param text  The string containing the dynamic data placeholders.

  98           * 

  99           * @returns true or false depending on whether all placeholders have been replaced.

 100           */
 101          function isValidReplacement($text) {
 102            if(preg_match_all('/\{([\w\d\.\s\(\)]+)\}/', $text, $matches)) {
 103              if(isset($matches[1]) && is_array($matches[1])) {
 104                foreach($matches[1] as $key=>$placeholder) {
 105                  $value = KT_DynamicData('{' . $placeholder . '}', $this->tNG, '', false, array(), false);
 106                  if(empty($value)) {
 107                    return false;
 108                  }
 109                }
 110              }
 111            }
 112  
 113            return true;
 114          }

 118           * @return mixt null or error object in case of error;

 119           * @access public

 120           */
 121          function Execute() {
 122            if($this->validReplacement === false) {
 123              $ret = new tNG_error("FOLDER_DEL_SECURITY_ERROR", array(), array($this->fullFolder, $this->baseFolder));
 124              return $ret;
 125            }
 126  
 127            $this->fullFolder = KT_realpath($this->baseFolder. $this->folder);
 128            // security

 129            if (substr($this->fullFolder, 0, strlen($this->baseFolder)) != $this->baseFolder) {
 130              $ret = new tNG_error("FOLDER_DEL_SECURITY_ERROR", array(), array($this->fullFolder, $this->baseFolder));
 131              return $ret;
 132            }
 133            
 134            $ret = null;
 135            if (!file_exists($this->fullFolder)) {
 136              return $ret;    
 137            }
 138            $folder = new KT_Folder();
 139            // delete thumbnails;

 140            $folder->deleteFolderNR($this->fullFolder);
 141            if ($folder->hasError()) {
 142              $arr = $folder->getError();
 143              $ret = new tNG_error("FOLDER_DEL_ERROR", array($arr[0]), array($arr[1]));
 144              return $ret;
 145            }
 146            
 147            return $ret;    
 148      }
