 121                  $pk_value = KT_escapeForSql($pk_value, $pk_type);
 122                  $sql .= " AND " . $pk . " <> " . $pk_value;
 123              }
 124              $ret = $this->tNG->connection->Execute($sql);
 125              if ($ret === false) {
 126                  return new tNG_error('CHECK_TF_SQL_ERROR', array(), array($this->tNG->connection->ErrorMsg(), $sql));
 127              }
 128              if (!$ret->EOF) {
 129                  $useSavedData = false;
 130                  if (in_array($this->tNG->transactionType, array('_delete', '_multipleDelete'))) {
 131                      $useSavedData = true;
 132                  }
 133                  $this->errorMsg = KT_DynamicData($this->errorMsg, $this->tNG, '', $useSavedData);
 134                  if ($GLOBALS['tNG_debug_mode'] == 'DEVELOPMENT') {
 135                      $err = new tNG_error('TRIGGER_MESSAGE__CHECK_UNIQUE',  array(implode(', ', $this->field)), array());
 136                  } else {

 138                  }
 139                  if (count($this->field)==1 && isset($this->tNG->columns[$this->field[$first]])) {
