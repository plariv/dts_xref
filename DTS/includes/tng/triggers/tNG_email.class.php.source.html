 273      /**

 274       * Setter. Sets the attachmetns 

 275           * Only for PRO version

 276       * @param string rename rule

 277       * @access public

 278       */
 279  	function setAttachmentRenameRule($renameRule) {
 280          if ($this->type == 'recordset' && $this->error == '') {
 281              if (!isset($GLOBALS[$this->rsName])) {
 282                  $this->error = new tNG_error('EMAIL_ERROR_RECORDSET', array(), array($this->rsName));
 283                  return;
 284              }
 285              $recordset = &$GLOBALS[$this->rsName];
 286              if (is_resource($recordset)) {
 287                  $rs = new KT_Recordset($recordset); 
 288              } else {
 289                  $rs = &$recordset;
 290              }
 291              
 292              $rs->MoveFirst();
 293              while (!$rs->EOF) {
 294                  $GLOBALS["row_".$this->rsName] = $rs->fields;
 295                  $renameRule2 = KT_DynamicData($renameRule, null, null, false, array());
 296                  
 297                  // security

 298                  if (substr(KT_realpath($this->folder . $renameRule2, false), 0, strlen($this->folder)) != $this->folder) {
 299                      $this->error = new tNG_error("EMAIL_ERROR_FOLDER", array(), array(KT_realpath($this->folder . $renameRule2, false), $this->folder));        
 300                      break;    
 301                  } else {
 302                      if (is_file($this->folder . $renameRule2)) {
 303                          $this->attachments[] = $this->folder . $renameRule2;
 304                      }
 305                  }    
 306                  $rs->MoveNext();
 307              }    
 308              $rs->MoveFirst();        
 309          } else if ($this->type == 'custom' && $this->error == '') {
 310              $renameRule = KT_DynamicData($renameRule, $this->getTng(), $this->escapeMethod, $this->getUseSavedData(), array());
 311              // security

 312              if (substr(KT_realpath($this->folder . $renameRule, false), 0, strlen($this->folder)) != $this->folder) {
 313                  $this->error = new tNG_error("EMAIL_ERROR_FOLDER", array(), array(KT_realpath($this->folder . $renameRule, false), $this->folder));            
 314              } else {
 315                  if (is_file($this->folder . $renameRule)) {
 316                      $this->attachments[] = $this->folder . $renameRule;
 317                  }
 318              }    
 319          }
 320          $this->type = '';
 321          $this->folder = '';
 322          $this->rsName = '';
 323      }

 512      /**

 513       * the main method, execute the code of the class;

 514       * @access public

 515       */
 516  	function Execute() 
 517      {
 518          if (is_object($this->error)) {
 519              return $this->error;
 520          }
 521          if ($this->contentFile!='' && file_exists($this->contentFile) && $fp = fopen($this->contentFile, 'r')) {
 522              $this->content = fread($fp, filesize($this->contentFile));
 523              fclose($fp);
 524          } else if ($this->contentFile!='') {
 525              return new tNG_error('EMAIL_NO_TEMPLATE', array(), array());
 526          }
 527          
 528          $email = new KT_email();
 529          $email->setPriority($this->importance);
 530          foreach ($this->attachments as $filename) {
 531              $email->addAttachment($filename);
 532          }
 533          $email->sendemail($GLOBALS['tNG_email_host'], $GLOBALS['tNG_email_port'], $GLOBALS['tNG_email_user'], $GLOBALS['tNG_email_password'], $this->getFrom(), $this->getTo(), $this->getCc(), $this->getBcc(), $this->getSubject(), $this->getEncoding(), $this->getTextBody(), $this->getHtmlBody());
 534          if ($email->hasError()) {
 535              $arr = $email->getError();
 536              return new tNG_error('EMAIL_FAILED', array(''), array($arr[1]));
 537          } 
 538      }
