 135      * Main Class method. Sets the action: remove|replace|block.

 136      * @return mixt object in case of errors or null

 139  	function Execute() {
 140          $ret = null;
 141          $arr = $this->tNG->columns;
 142          $columns = array();
 143          foreach ($arr as $colName=>$colDetails) {
 144              if (in_array($colName, $this->columns) && $colDetails['type'] == 'STRING_TYPE') {
 145                  $columns[$colName] = $colDetails['value'];
 146              }
 147          }
 148          if (count($columns) == 0) {
 149              return ;
 150          }
 151          $fieldWithErrors = array();
 152          foreach ($columns as $colName => $value) {
 153              $return = $this->execTidy($colName, $value);
 154              if ($return === false) {
 155                  $error = true;
 156                  $fieldWithErrors[] = $colName;
 157              } else {
 158                  $columns[$colName] = $this->tidiedValues[$colName];
 159              }
 160          }
 161          // write the values back in tNG

 162          foreach($columns as $name => $value) {
 163              $this->tNG->setColumnValue($name, $value);
 164          }
 165          if (isset($error)) {
 166              if ($this->errorMsg == '') {
 167                  $ret = new tNG_error('ERROR_TIDY_CONTENT', array(), array(implode(', ', $fieldWithErrors)));
 168              } else {
 169                  $ret = new tNG_error('%s', array($this->errorMsg), array($this->errorMsg));
 170              }                 
 171          }
 172          return $ret;
 173      }

 194          $shell->execute($loc, $arg_test);
 195          if ($shell->hasError()) {

 198              return false;
 199          }
 200          $execPath = $shell->getExecutedCommand();
 201          if (!isset($_SESSION['TidyContent']['ExecPath']) && $execPath != '') {
 202              $_SESSION['TidyContent']['ExecPath'] = $execPath;
 203              $loc = array($execPath);
 204          }
 205          
 206          $tidyEncoding = 'raw';
 207          if (strtolower($this->outEncoding) == 'iso-8859-1') {
 208              $tidyEncoding = 'ascii';
 209          }
 210          if (strpos(strtolower($this->outEncoding), 'utf-8') !== false) {
 211              $tidyEncoding = 'utf8';
 212          }
 213  
 214          $string = $value;
 215          $string = str_replace("&amp;nbsp;", "&amp;amp;nbsp;", $string);
 216          $string = str_replace("&nbsp;", "&amp;nbsp;", $string);
 217  
 218          if (!file_exists($this->folderName)) {
 219              $folder = new KT_folder();
 220              $folder->createFolder($this->folderName);

 225              }
 226          }

 229          if ($f === false) {
