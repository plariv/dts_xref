  23       * possible level values

  42      /**

  44       * @param string 

  45       * @access public

  47  	function addLevel($level) {
  48          array_push($this->levels, $level);
  49      }
  50      /**

  51       * Main method of the class. 

  52       * If the user is not log in, call tNG_cookieLogin which will try to autologin based on the cookies;

  53       * verify if the user is logged in and have the proper levels;

  54       * @return boolean true if the user has the rights and is loggedin;

  55       * @access public

  56       */
  57  	function Execute() {
  58          tNG_cookieLogin($this->connection);
  59          // access denied defaults to "redirect_failed" specified in Login Config

  60          $grantAccess = false;
  61          
  62          tNG_clearSessionVars();
  63          
  64          if (isset($_SESSION['kt_login_user'])) {
  65              if (count($this->levels) > 0) {
  66                  if (isset($_SESSION['kt_login_level'])) {
  67                      if (in_array($_SESSION['kt_login_level'], $this->levels) ) {
  68                          $grantAccess = true;
  69                      }
  70                  }
  71              } else {
  72                  // no levels are required for this page access

  73                  // the user is logged in, so grant the access

  74                  $grantAccess = true;
  75              }
  76          }
  77          return $grantAccess;
  78      }
  79  }
