 110  	function getTextBody()
 111      {    
 112          $content = $this->removeScript($this->content);
 113          $content = $this->removeStyle($content);
 114          $content = strip_tags($content);
 115          $content = @html_entity_decode($content, ENT_QUOTES, $this->getEncoding());
 116          if (strtoupper($this->getEncoding()) == 'UTF-8') {
 117              $content = @utf8_encode($content);
 118          }
 119          return trim($content);
 120      }

 255      /**

 256       * Parse the content of the $file and keep any CSS found in;

 257       * If the encoding is not setted and found one in the text, it will use it.

 258       * @access protected

 259       */
 260  	function searchCss()
 261      {
 262          if ($this->format=='html/text' && $this->file!='' 
 263                  && file_exists($this->file) 
 264                  && $fp=fopen($this->file, "r")) {
 265              $content = fread($fp, filesize($this->file));
 266              fclose($fp);
 267              
 268              // find the css: <link ...>;

 269              preg_match_all("/<\s*link([^>]*)>/ims", $content, $matches);
 270              if (isset($matches[1]) && count($matches[1])>0) {
 271                  foreach ($matches[1] as $key=>$val) {
 272                      if (preg_match("/stylesheet/ims", $val)) {
 273                          preg_match("/href\s*=(.*)/ims", $val, $match);
 274                          if (isset($match[1])) {
 275                              $this->css .= '<link rel="stylesheet" type="text/css" href='.$match[1].' />'."\n";
 276                          }
 277                      }
 278                  }
 279              }
 280              // find the css: <style>...</style>

 281              preg_match_all("/<style\s*(type=\"text\/css\")?>([.\s\S]*)<\/style>/ims", $content, $matches);
 282              if (isset($matches[0]) && count($matches[0])>0) {
 283                  $this->css .= implode("\n", $matches[0]);
 284              }
 285              
 286              // set the encoding

 287              if ($this->encoding=='') {
 288                  preg_match("/<\s*meta([\s\S][^>]*)>/ims", $content, $matches);
 289                  if (isset($matches[1]) && $matches[1]!='') {
 290                      preg_match("/charset=(.[^>\"\']*)/ims", $content, $matches);
 291                      if (isset($matches[1]) && $matches[1]!='') {
 292                          $this->encoding = $matches[1];
 293                      }
 294                  }
 295              }
 296          }    
 297      }
