  75  	function getRecordset($tableName) {
  76  
  77              $method_where = -1;
  78              $method = -1;
  79              /*

  80                  -1 = unknown

  81                   1 = default values

  82                   2 = from recordset

  83                   3 = submitted values

  84              */
  85              for ($i=0;$i<$this->n;$i++) {
  86                  if ($this->tNGs[$i]->getTable() == $tableName) {
  87                      switch($this->tNGs[$i]->getTransactionType()) {
  88                          case '_login':
  89                          case '_custom':
  90                          case '_insert':
  91                          case '_multipleInsert':
  92                              if ($method < 1) {
  93                                  $method = 1;
  94                                  $method_where = $i;
  95                              }
  96                              break;
  97                          case '_update':
  98                          case '_multipleUpdate':
  99                              $pkv = $this->tNGs[$i]->getPrimaryKeyValue();
 100                              if (isset($pkv)) {
 101                                  if ($method < 2) {
 102                                      $method = 2;
 103                                      $method_where = $i;
 104                                  }
 105                              }
 106                              break;
 107                      }
 108                      if ($this->tNGs[$i]->isStarted()) {
 109                          if ($this->tNGs[$i]->exportsRecordset()) {
 110                              $tmpArrErr = $this->tNGs[$i]->getErrorMsg();
 111                              if ($tmpArrErr[1] != '' || $tmpArrErr[2] != '') {
 112                                  $method = 3;
 113                                  $method_where = $i;
 114                              }
 115                          }
 116                      }
 117                  }
 118              }
 119              if ($method_where == -1) {
 120                  $ret = $this->tNGs[0]->getRecordset(); 
 121                  //die('tNG_dispatcher.getRecordset:<br/>The transaction that handles the curent recordset is not properly configured.<br>Check the settings from the user interface that generated the code. Possible reasons for this failure: you have specified URL parameter for the Primary key column in the interface and you are running the page without any parameter or the incorect parameter.');

 122              } else {
 123                  $ret = $this->tNGs[$method_where]->getRecordset();
 124          }
 125  
 126          return $ret;
 127      }

 249  	function prepareValidation(&$uniVal) {
 250          $this->UnivalCount++;
 251          if (!is_array($uniVal->columns) || count($uniVal->columns) == 0) {
 252              return;
 253          }
 254          foreach ($uniVal->columns as $columnName => $column) {
 255              // Set unival JS div errors

 256              // here we set the least restrictive required prop

 257              $required = ($column['required']===true)?'true':'false';
 258              if (array_key_exists($columnName, $this->UnivalProps)) {
 259                  if ($this->UnivalProps[$columnName]['required'] != $required) {
 260                      $required = 'false';
 261                  }
 262              }
 263              $this->UnivalProps[$columnName]['required'] = $required;
 264              $this->UnivalProps[$columnName]['type'] = $column['type'];
 265              $this->UnivalProps[$columnName]['format'] = $column['format'];
 266              $this->UnivalProps[$columnName]['additional_params'] = $column['additional_params'];
 267              $this->UnivalProps[$columnName]['min'] = $column['min_cs'];
 268              $this->UnivalProps[$columnName]['max'] = $column['max_cs'];
 269              @$this->UnivalProps[$columnName]['count']++;
 270              $this->UnivalErrors[$columnName] = $column['message'];
 271              // Set field Hints

 272              if ($column['type'] == 'regexp') {
 273                  $this->fieldHints[$columnName]  = '';
 274              } elseif ($column['type'] == 'mask') {
 275                  $this->fieldHints[$columnName]  = '(' . $column['additional_params'] . ')';
 276              } elseif ($column['type'] == 'date' && $column['format'] != '') {
 277                  $this->fieldHints[$columnName]  = '(' . $uniVal->genericValidationMessages['date_' . $column['format']] . ' ' . $column['date_screen_format'] . ')';
 278              } elseif ($column['format'] != '') {
 279                  $this->fieldHints[$columnName]  = '(' . $uniVal->genericValidationMessages[$column['type'] . '_' . $column['format']] . ')';
 280              }
 281          }
 282      }
