 160      /**

 161       * Executes all sub-transactions

 162       * @access protected

 163       */
 164  	function prepareSQL() {
 165          tNG_log::log('tNG_import', 'prepareSQL', 'begin');
 166          
 167          $ret = $this->prepareData();
 168          if ($ret === null) {
 169              $this->noSuccess = 0;
 170              $this->noSkip = 0;
 171              $failed = false;
 172              $line = $this->lineStart;
 173              
 174              $tNGindex = 1;
 175              
 176              for ($k = 0; $k < count($this->data); $k++) {
 177                  $dataarr = $this->data[$k];
 178                  $skipped = false;
 179                  $line++;
 180                  
 181                  /*

 182                  if ( !is_array($dataarr) || count($dataarr) < 1 || (count($dataarr) == 1 && reset($dataarr) == '') ) {

 183                      // skip empty lines

 184                      continue;

 185                  }

 186                  */
 187                  
 188                  // exports the values line to be available for KT_getRealValue and KT_DynamicData

 189                  unset($GLOBALS[$this->importReference]);
 190                  $GLOBALS[$this->importReference] = $dataarr;
 191                  
 192                  unset($GLOBALS[$this->importReference . '_LINE']);
 193                  $GLOBALS[$this->importReference . '_LINE'] = $line;
 194                  
 195                  $isInsert = true;
 196                  $uniqueColName = $this->uniqueKey;
 197                  if ($uniqueColName != '') {
 198                      $uniqueColDetails = $this->computeMultipleValues($this->columns[$uniqueColName], $tNGindex);
 199                      if ($uniqueColDetails['value'] != '') {
 200                          $sql = 'SELECT ' . KT_escapeFieldName($uniqueColName) . ' FROM ' . $this->getTable() . ' WHERE ' . KT_escapeFieldName($uniqueColName) . ' = ' . KT_escapeForSql($uniqueColDetails['value'], $uniqueColDetails['type']);
 201                          $rs = $this->connection->Execute($sql);
 202                          if ($rs === false) {
 203                              $failed = true;
 204                              $ret = new tNG_error('IMPORT_SQL_ERROR', array(), array($sql, $this->connection->ErrorMsg()));
 205                              tNG_log::log('KT_ERROR');
 206                              break;
 207                          }
 208                          if ($rs->recordCount() >= 1) {
 209                              // duplicates found

 210                              if ($this->handleDuplicates == "SKIP") {
 211                                  // ignore case

 212                                  $isInsert = false;
 213                                  $this->noSkip++;
 214                                  continue;
 215                              }
 216                              if ($this->handleDuplicates == "UPDATE") {
 217                                  // update case

 218                                  $isInsert = false;
 219                                  $this->multTNGs[$tNGindex-1] = new tNG_update($this->connection);
 220                              }
 221                              if ($this->handleDuplicates == "SKIPWITHERROR") {
 222                                  // throw error case

 223                                  $isInsert = false;
 224                                  $skipped = true;
 225                                  $this->noSkip++;
 226                                  $this->multTNGs[$tNGindex-1] = new tNG_insert($this->connection);
 227                                  $this->multTNGs[$tNGindex-1]->setError(new tNG_error($this->importType . '_IMPORT_DUPLICATE_ERROR', array($line, $uniqueColDetails['value'], $uniqueColName), array()));
 228                              }
 229                          }
 230                      }
 231                  }
 232                  
 233                  if ($isInsert) {
 234                      $this->multTNGs[$tNGindex-1] = new tNG_insert($this->connection);
 235                  }
 236                  
 237                  $this->multTNGs[$tNGindex-1]->setDispatcher($this->dispatcher);
 238                  $this->multTNGs[$tNGindex-1]->multipleIdx = $tNGindex;
 239                  // register triggers

 240                  for ($j=0;$j<sizeof($this->multTriggers);$j++) {
 241                      call_user_func_array(array(&$this->multTNGs[$tNGindex-1], "registerConditionalTrigger"), $this->multTriggers[$j]);
 242                  }
 243                  $this->multTNGs[$tNGindex-1]->setTable($this->table);
 244                  // add columns

 245                  
 246                  foreach($this->columns as $colName=>$colDetails) {
 247                      $colDetails = $this->computeMultipleValues($colDetails, $tNGindex);
 248                      $this->columns[$colName]['value'] = $colDetails['value'];
 249                      if ($this->multTNGs[$tNGindex-1]->transactionType == '_update') {
 250                          if ($colName != $uniqueColName) {
 251                              $this->multTNGs[$tNGindex-1]->addColumn($colName, $colDetails['type'], $colDetails['method'], $colDetails['reference']);
 252                          }
 253                      } else {
 254                          $this->multTNGs[$tNGindex-1]->addColumn($colName, $colDetails['type'], $colDetails['method'], $colDetails['reference'], $colDetails['default']);
 255                      }
 256                  }
 257                  
 258                  if ($this->multTNGs[$tNGindex-1]->transactionType == '_update') {
 259                      $this->multTNGs[$tNGindex-1]->setPrimaryKey($uniqueColName, $uniqueColDetails['type'], 'VALUE', $uniqueColDetails['value']);
 260                  } else {
 261                      $this->multTNGs[$tNGindex-1]->setPrimaryKey($this->primaryKey, $this->primaryKeyColumn['type']);
 262                  }
 263                  $this->multTNGs[$tNGindex-1]->compileColumnsValues();
 264                  
 265                  if ($this->getError()) {
 266                      $this->multTNGs[$tNGindex-1]->setError($this->getError());
 267                  }
 268                  
 269                  $this->multTNGs[$tNGindex-1]->setStarted(true);
 270                  $this->multTNGs[$tNGindex-1]->doTransaction();
 271                  
 272                  if (!$skipped) {
 273                      if ($this->multTNGs[$tNGindex-1]->getError()) {
 274                          $err = $this->multTNGs[$tNGindex-1]->getError();
 275                          $tmp_all_errmsg = '';
 276                          $tmp_unique_details = '';
 277                          if ($uniqueColName != '') {
 278                              if ($uniqueColDetails['value'] != '') {
 279                                  $tmp_unique_details = ' (' . $uniqueColName . ' = ' . $uniqueColDetails['value'] . ')';
 280                              }
 281                          }
 282                          foreach($err->fieldErrors as $tmp_col => $tmp_errmsg) {
 283                              $tmp_all_errmsg .= "\n<br />&nbsp;&nbsp;&nbsp;- ".$tmp_col." : ".$tmp_errmsg;
 284                          }
 285                          if ($tmp_all_errmsg == '') {
 286                              $tmp_all_errmsg = $err->getDetails();
 287                          }
 288                          $lineErr = $line . $tmp_unique_details;
 289                          $newErr = new tNG_error($this->importType . '_IMPORT_LINE_ERROR',array($lineErr, $tmp_all_errmsg), array());
 290                          $this->multTNGs[$tNGindex-1]->setError($newErr);
 291                          $failed = true;
 292                      } else {
 293                          $this->noSuccess++;
 294                          if ($this->getPrimaryKey() == $this->multTNGs[$tNGindex-1]->getPrimaryKey()) {
 295                              $this->primaryKeyColumn['value'] = $this->multTNGs[$tNGindex-1]->getPrimaryKeyValue();
 296                          }
 297                      }
 298                  }
 299                  
 300                  $tNGindex++;
 301              }
 302              
 303              if (!$failed) {
 304                  for ($i=0;$i<sizeof($this->multTNGs);$i++) {
 305                      if ($this->multTNGs[$i]->getError()) {
 306                          $failed = true;
 307                          $ret = new tNG_error('IMPORT_SKIPPED', array(), array());
 308                          tNG_log::log('KT_ERROR');
 309                          break;
 310                      }
 311                  }
 312              }
 313          
 314              if ($failed) {
 315                  if ($ret === null) {
 316                      $ret = new tNG_error('IMPORT_ERROR', array(), array());
 317                      tNG_log::log('KT_ERROR');
 318                  }
 319                  if ($this->executeSubSets === false) {
 320                      for ($i=0;$i<sizeof($this->multTNGs);$i++) {
 321                          if (!$this->multTNGs[$i]->getError()) {
 322                              $this->multTNGs[$i]->setError($ret);
 323                              $this->multTNGs[$i]->executeTriggers('ERROR');
 324                          }
 325                      }
 326                  }
 327              }
 328              
 329              if ($this->executeSubSets === false) {
 330                  $this->noSuccess = 0;
 331              }
 332              
 333          } else {
 334              tNG_log::log('KT_ERROR');
 335          }
 336          
 337          tNG_log::log('tNG_import', 'prepareSQL', 'end');
 338          return $ret;
 339      }

 341      /**

 342       * Sets the column details corresponding to its method and current transaction index

 343       * @param array $colDetails Column details (one element of the $column array)

 344       * @param integer $tNGindex The current transaction's index

 345       * @return array $colDetails

 346       * @access private

 347       */
 348  	function computeMultipleValues($colDetails, $tNGindex) {
 349          if ($colDetails['method'] == 'VALUE') {
 350              $reference = $colDetails['reference'];
 351              $value = KT_getRealValue($colDetails['method'], $colDetails['reference']);
 352          } elseif ($colDetails['method'] == $this->importType) {
 353              $value = KT_getRealValue($colDetails['method'], $this->headers[$colDetails['reference']]);
 354              $reference = $this->headers[$colDetails['reference']];
 355          } else {
 356              $reference = $colDetails['reference'] . '_' . $tNGindex;
 357              $value = KT_getRealValue($colDetails['method'], $colDetails['reference'] . '_' . $tNGindex);
 358              if (!isset($value)) {
 359                  $reference = $colDetails['reference'];
 360                  $value = KT_getRealValue($colDetails['method'], $colDetails['reference']);
 361              }
 362          }
 363          $colDetails['value'] = $value;
 364          $colDetails['reference'] = $reference;
 365          return $colDetails;
 366      }

 369       * Adds a column to the transaction

 370       * Calls the parent addColumn method then sets the default value.

 371       * @param string $colName The column name

 372       * @param string $type The column type (NUMERYC_TYPE, STRING_TYPE, etc)

 373       * @param string $method The request method (GET, POST, FILE, COOKIE, SESSION)

 374       * @param string $reference The submitted variable name (if method=GET and reference=test, value=$_GET['test'])

 375       * @param string $defaultValue The default value for the current column

 376       * @access public

 377       */
 378  	function addColumn($colName, $type, $method, $reference, $defaultValue = '') {
 379          parent::addColumn($colName, $type, $method, $reference);
 380          if ($method == 'VALUE') {
 381              $this->columns[$colName]['default'] = $reference;
 382          } else {
 383              if ($method == $this->importType) {
 384                  $this->headers[$reference] = $reference;
 385              }
 386              $this->columns[$colName]['default'] = $defaultValue;
 387          }
 388      }

 404          $ret[0] = trim($ret[0]);
 405          return $ret;
