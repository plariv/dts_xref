  21  class tNG_delete extends tNG_fields {
  22  
  23      /**

  24       * Constructor. Sets the connection, the database name and other default values.

  25       * Also sets the transaction type.

  26       * @param object KT_Connection &$connection the connection object

  27       * @access public

  28       */
  29  	function tNG_delete(&$connection) {
  30          parent::tNG_fields($connection);
  31          $this->transactionType = '_delete';
  32          $this->registerTrigger("BEFORE", "Trigger_Default_saveData", -1);
  33      }
  34      
  35      /**

  36       * Prepares the delete SQL query to be executed

  37       * @access protected

  38       */
  39  	function prepareSQL() {
  40          tNG_log::log('tNG_delete', 'prepareSQL', 'begin');
  41          parent::prepareSQL();
  42          // check if we have a valid primaryKey

  43          if (!$this->primaryKey) {
  44              $ret = new tNG_error('DEL_NO_PK_SET', array(), array());
  45          }
  46  
  47          // check the primary key value

  48          if (!isset($this->primaryKeyColumn['value'])) {
  49              $ret = new tNG_error('DEL_NO_PK_VAL', array(), array());
  50          }
  51  
  52          $ret = null;
  53          $sql = 'DELETE FROM ' . $this->table . ' WHERE ' . KT_escapeFieldName($this->primaryKey) . ' = ';
  54          $sql .= KT_escapeForSql($this->primaryKeyColumn['value'], $this->primaryKeyColumn['type']);
  55          $this->setSQL($sql);
  56          tNG_log::log('tNG_delete', 'prepareSQL', 'end');
  57          return $ret;
  58      }
  59  
  60      /**

  61       * This function exits because the current class does not export a recordset.

  62       * @access protected

  63       */
  64  	function getLocalRecordset() {
  65          $this->setError(new tNG_error('DEL_NO_RS', array(), array()));
  66      }
  67  }
