  36       * Overwrites tNG method in order to register some triggers specific to Update Users Transaction

  37       * @access protected

  39  	function doTransaction() {
  40          $table = $this->getTable();
  41          if (isset($GLOBALS['tNG_login_config']['table']) && $GLOBALS['tNG_login_config']['table'] == $table ) {
  42              // BEFORE triggers

  43              if (isset($this->columns[$GLOBALS['tNG_login_config']['password_field']])) { 
  44                  if ($GLOBALS['tNG_login_config']['password_encrypt'] == "true") {
  45                      $this->registerTrigger("BEFORE", "Trigger_UpdatePassword_EncryptPassword", 200);
  46                  }
  47                  $this->registerTrigger("BEFORE", "Trigger_UpdatePassword_RemovePassword", 210);
  48                  
  49                  $this->registerTrigger("AFTER", "Trigger_UpdatePassword_AddPassword", -100);
  50                  
  51                  if ($GLOBALS['tNG_login_config']['password_encrypt'] == "true") {
  52                      $this->registerTrigger("AFTER", "Trigger_UpdatePassword_RestorePassword", -90);
  53                  }
  54                  
  55                  $this->registerTrigger("ERROR", "Trigger_UpdatePassword_AddPassword", -100);
  56                  $this->registerTrigger("ERROR", "Trigger_UpdatePassword_RemoveOldPassword", -90);
  57              }
  58              $this->registerTrigger("AFTER", "Trigger_Update_RefreshSession", 200);
  59          }
  60          parent::doTransaction();

  62      /**

  63       * Prepares the update SQL query to be executed

  64       * @access protected

  66  	function prepareSQL() {
  67          tNG_log::log('tNG_update', 'prepareSQL', 'begin');
  68          parent::prepareSQL();
  69          // check if we have a valid primaryKey

  70          if ($this->primaryKey == '') {
  71              return new tNG_error('UPD_NO_PK_SET', array(), array());
  72          }
  73  
  74          // check the primary key value

  75          if (!isset($this->primaryKeyColumn['value'])) {
  76              return new tNG_error('UPD_NO_PK_SET', array(), array());
  77          }
  78  
  79          // begin the SQL generator

  80          $sql = 'UPDATE ' . $this->table . ' SET ';
  81          $KT_sp = false;
  82  
  83          foreach($this->columns as $colName=>$colDetails) {
  84              $colType = $this->columns[$colName]['type'];
  85              $colValue = $this->columns[$colName]['value'];
  86              $colMethod = $this->columns[$colName]['method'];
  87              $sep = ($KT_sp) ? ', ' : ''; // set the separator ',' (first time will be none)

  88              if ($colType == 'FILE_TYPE' || $colMethod == 'CURRVAL') {
  89                  $sql .= $sep . KT_escapeFieldName($colName) . '=' . KT_escapeFieldName($colName);
  90              } else {
  91                  // if we handle a hidden field, we should not use it in the update SQL.

  92                  // add the column to the SQL string

  93                  $sql .= $sep . KT_escapeFieldName($colName) . '=' . KT_escapeForSql($colValue, $colType);
  94              }
  95              $KT_sp = true;
  96          }
  97  
  98          if (!$KT_sp) {
  99              // no column was actually added

 100              return new tNG_error('UPD_NO_FIELDS', array(), array());
 101          }
 102  
 103          // add the where clause

 104          $sql .= ' WHERE ' . KT_escapeFieldName($this->primaryKey) . ' = ';
 105          $sql .= KT_escapeForSql($this->primaryKeyColumn['value'], $this->primaryKeyColumn['type']);
 106          $this->setSQL($sql);
 107          tNG_log::log('tNG_update', 'prepareSQL', 'end');
 108          return null;
 109      }

 116  	function getLocalRecordset() {
 117          tNG_log::log('tNG_update', 'getLocalRecordset');
 118          $sql = '';
 119          $tmpArr = $this->columns;
 120          $tmpArr[$this->primaryKey]['type'] = $this->primaryKeyColumn['type'];
 121          $tmpArr[$this->primaryKey]['method'] = $this->primaryKeyColumn['method'];
 122          $tmpArr[$this->primaryKey]['reference'] = $this->primaryKeyColumn['reference'];
 123          foreach($tmpArr as $colName=>$colDetails) {
 124              if ($sql != '') {
 125                  $sql .=',';
 126              }
 127              $sql.=KT_escapeFieldName($colName);
 128          }
 129          $sql .= ', ' . KT_escapeFieldName($this->primaryKey) . ' as ' . KT_escapeFieldName($this->pkName);
 130          $sql = 'SELECT '.$sql.' FROM '.$this->table;
 131          $sql = $sql . ' WHERE ' . KT_escapeFieldName($this->primaryKey) . ' =';
 132          $pkValue = KT_getRealValue($this->primaryKeyColumn['method'], $this->primaryKeyColumn['reference']);
 133          $sql = $sql . KT_escapeForSql($pkValue, $this->primaryKeyColumn['type']);
 134          if (isset($this->connection->servermodel)) {
 135              $rs = $this->connection->MySQL_Execute($sql);
 136          } else {
 137              $rs = $this->connection->Execute($sql);
 138          }
 139          if (!$rs) {
 140              tNG_log::log('KT_ERROR');
 141              $this->setError(new tNG_error('UPD_RS', array(), array($this->connection->ErrorMsg(), $sql)));
 142              echo $this->dispatcher->getErrorMsg();
 143              exit;
 144          }
 145          return $rs;
 146      }
