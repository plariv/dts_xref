 125  FlashTag.prototype.toString = function() {
 126      var ie = (navigator.appName.indexOf ("Microsoft") != -1) ? 1 : 0;
 127      var flashTag = new String();
 128      if (ie)
 129      {
 130          flashTag += '<object classid="clsid:D27CDB6E-AE6D-11cf-96B8-444553540000" ';
 131          if (this.id != null)
 132          {
 133              flashTag += 'id="'+this.id+'" ';
 134          }
 135          flashTag += 'codebase="http://download.macromedia.com/pub/shockwave/cabs/flash/swflash.cab#version='+this.version+'" ';
 136          flashTag += 'width="'+this.width+'" ';
 137          flashTag += 'height="'+this.height+'">';
 138          flashTag += '<param name="movie" value="'+this.src+'"/>';
 139          flashTag += '<param name="quality" value="high"/>';
 140          flashTag += '<param name="bgcolor" value="#'+this.bgcolor+'"/>';
 141          flashTag += '<param name="allowScriptAccess" value="always"/>';
 142          if(this.wmode) {
 143            flashTag += '<param name="wmode" value="'+this.wmode+'" />';
 144      }
 145          if (this.flashVars != null)
 146          {
 147              flashTag += '<param name="flashvars" value="'+this.flashVars+'"/>';
 148          }
 149          flashTag += '</object>';
 150      }
 151      else
 152      {
 153          flashTag += '<embed src="'+this.src+'" allowScriptAccess="always" ';
 154          flashTag += 'quality="high" '; 
 155          flashTag += 'bgcolor="#'+this.bgcolor+'" ';
 156          flashTag += 'width="'+this.width+'" ';
 157          flashTag += 'height="'+this.height+'" ';
 158          flashTag += 'type="application/x-shockwave-flash" ';
 159          if(this.wmode) {
 160            flashTag += 'wmode="'+this.wmode+'" ';
 161      }
 162          if (this.flashVars != null)
 163          {
 164              flashTag += 'flashvars="'+this.flashVars+'" ';
 165          }
 166          if (this.id != null)
 167          {
 168              flashTag += 'name="'+this.id+'" ';
 169          }
 170          flashTag += 'pluginspage="http://www.macromedia.com/go/getflashplayer">';
 171          flashTag += '</embed>';
 172      }
 173      return flashTag;
 174  }

 431  getPlayerVersion = function(){
 432      var PlayerVersions = new PlayerVersion(0,0,0);
 433      if(navigator.plugins && navigator.mimeTypes.length){
 434          var x = navigator.plugins["Shockwave Flash"];
 435          if(x && x.description) {
 436              PlayerVersions = new PlayerVersion(x.description.replace(/([a-z]|[A-Z]|\s)+/, "").replace(/(\s+r|\s+b[0-9]+)/, ".").split("."));
 437          }
 438      } else if (window.ActiveXObject){
 439         try {
 440            var axo = new ActiveXObject("ShockwaveFlash.ShockwaveFlash");
 441             PlayerVersions = new PlayerVersion(axo.GetVariable("$version").split(" ")[1].split(","));
 442         } catch (e) {}
 443      }
 444      return PlayerVersions;
 445  };

 447  PlayerVersion = function(arrVersion){
 448      this.major = parseInt(arrVersion[0]) || 0;
 449      this.minor = parseInt(arrVersion[1]) || 0;
 450      this.rev = parseInt(arrVersion[2]) || 0;
 451  };
 452  
 453  PlayerVersion.prototype.versionIsValid = function(fv){
 454      if(this.major < fv.major) return false;
 455      if(this.major > fv.major) return true;
 456      if(this.minor < fv.minor) return false;
 457      if(this.minor > fv.minor) return true;
 458      if(this.rev < fv.rev) return false;
 459      return true;
 460  };
