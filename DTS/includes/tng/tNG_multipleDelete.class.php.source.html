  21          /**

  22           * Constructor. Sets the connection, the database name and other default values.

  27  		function tNG_multipleDelete(&$connection) {
  28              parent::tNG_multiple($connection);
  29              $this->transactionType = '_multipleDelete';
  30          }
  31          /**

  32           * Executes all sub-transactions

  33           * @param none

  34           * @access protected

  35           */
  36  		function prepareSQL() {
  37              tNG_log::log('tNG_multipleDelete', 'prepareSQL', 'begin');
  38              $failed = false;
  39              $ret = null;
  40              for ($i=1;true;$i++) {
  41                  $tmp = KT_getRealValue("POST", $this->pkName . "_" . $i);
  42                  if (!isset($tmp)) {
  43                      break;
  44                  }
  45                  $this->multTNGs[$i-1] = new tNG_delete($this->connection);
  46                  $this->multTNGs[$i-1]->setDispatcher($this->dispatcher);
  47                  $this->multTNGs[$i-1]->multipleIdx = $i;
  48                  // register triggers

  49                  $this->multTNGs[$i-1]->registerTrigger("STARTER", "Trigger_Default_Starter", 1, "VALUE", true);
  50                  for ($j=0;$j<sizeof($this->multTriggers);$j++) {
  51                      call_user_func_array(array(&$this->multTNGs[$i-1], "registerConditionalTrigger"), $this->multTriggers[$j]);
  52                  }
  53                  // add columns

  54                  $this->multTNGs[$i-1]->setTable($this->table);
  55                  foreach($this->columns as $colName => $colDetails) {
  56                      $this->multTNGs[$i-1]->addColumn($colName, $colDetails['type'], $colDetails['method'], $colDetails['reference'] . "_" . $i);
  57                  }
  58                  $this->multTNGs[$i-1]->setPrimaryKey($this->primaryKey, $this->primaryKeyColumn['type'], "POST", $this->pkName . "_" . $i);
  59  
  60                  $this->multTNGs[$i-1]->executeTransaction();
  61                  
  62                  if ($this->multTNGs[$i-1]->getError()) {
  63                      $failed = true;
  64                  }
  65              }
  66              if ($failed) {
  67                  $ret = new tNG_error('MDEL_ERROR', array(), array());
  68              }
  69              tNG_log::log('tNG_multipleDelete', 'prepareSQL', 'end');
  70              return $ret;
  71          }
