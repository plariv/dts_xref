 128      /**

 129       * register a trigger to the current transaction

 130       * @param string $triggerType the trigger type (STARTER, AFTER, BEFORE, ERROR or END)

 131       * @param string $callBackFunction the function name to callback when the trigger is executed

 132       * @param integer $priority the trigger priority

 133       * @access private

 134       */
 135  	function registerConditionalTrigger($condition, $triggerType, $callBackFunction, $priority) {
 136          // check if the trigger type is valid

 137          if (in_array($triggerType,array("STARTER","AFTER","BEFORE","END","ERROR"))) {
 138              $params = array();
 139              if (func_num_args() > 4) {
 140                  $params = array_slice(func_get_args(),4);
 141              }
 142              $this->triggers[$triggerType][] = array($condition, $callBackFunction,$priority,$params);
 143              if ($callBackFunction == "Trigger_Default_FormValidation" && isset($this->dispatcher->UnivalProps) && $condition !== true) {
 144                  $this->dispatcher->UnivalProps = array();
 145              }
 146          } else {
 147              $this->setError(new tNG_error('UNKNOWN_TRIGGER', array(), array($triggerType)));
 148          }
 149      }

 173      /**

 174       * Executes the Transaction

 175       * Tests the STARTER triggers, executes the BEFORE triggers, the transaction SQL, the AFTER triggers and the END triggers. If error occurs, it executes the ERROR triggers and exits.

 176       * @access public

 177       */
 178  	function executeTransaction() {
 179          tNG_log::log('tNG' . $this->transactionType, 'executeTransaction', 'begin');
 180          if ($this->started) {
 181              tNG_log::log('tNG' . $this->transactionType, 'executeTransaction', 'end');
 182              return false;
 183          }
 184  
 185          //calling the starter triggers and terminate execution if we had an error

 186          $ret = $this->executeTriggers("STARTER");
 187          if (is_object($ret)) {
 188              $this->setError($ret);
 189              $this->executeTriggers("ERROR");
 190              tNG_log::log('tNG' . $this->transactionType, 'executeTransaction', 'end');
 191              return false;
 192          }
 193  
 194          if (!$this->started) {
 195              tNG_log::log('tNG' . $this->transactionType, 'executeTransaction', 'end');
 196              return false;
 197          }
 198  
 199          $ret = $this->doTransaction();
 200          tNG_log::log('tNG' . $this->transactionType, 'executeTransaction', 'end');
 201  
 202          return $ret;
 203      }

 205      /**

 206       * Executes the registered triggers that matches the specified type

 207       * @param string $triggerType (STARTER, AFTER, BEFORE, ERROR and END)

 208       * @access protected

 209       */
 210  	function executeTriggers($triggerType) {
 211          if (isset($this->triggers[$triggerType]) && is_array($this->triggers[$triggerType])) {
 212              uasort($this->triggers[$triggerType],array('tNG','compareTriggers'));
 213              foreach ($this->triggers[$triggerType] as $key => $trigger) {
 214                  $run = KT_DynamicData($trigger[0], $this, 'expression');
 215                  $runTrigger = false;
 216                  $ok = false;
 217                  @eval('$runTrigger = ('.$run.');$ok = true;');
 218                  if ($ok !== true) {
 219                      die('Internal Error.Invalid boolean expression: '.$run);
 220                  }
 221                  if ($runTrigger) {
 222                      $callBackFunction = $trigger[1];
 223                      $tempParam = array_reverse($trigger[3]);
 224                      $tempParam[] = &$this;
 225                      $tempParam = array_reverse($tempParam, true);
 226                      tNG_log::log($triggerType, $callBackFunction, 'begin');
 227                      if (is_string($callBackFunction) && function_exists($callBackFunction)) {
 228                      $ret = call_user_func_array($callBackFunction,$tempParam);
 229                      } else {
 230                          die('Internal Error. Trigger "'.$callBackFunction.'" does not exist.');
 231                      }
 232                      // call all triggers for ERROR and STARTER tiggers

 233                      if ($triggerType != "ERROR") {
 234                          if (is_object($ret)) {
 235                              tNG_log::log('KT_ERROR');
 236                              tNG_log::log($triggerType, $callBackFunction, 'end');
 237                              return $ret;
 238                          }
 239                      }
 240                      tNG_log::log($triggerType, $callBackFunction, 'end');
 241                  }
 242              }
 243          }
 244          return null;
 245      }

 247      /**

 248       * compares two trigger objects (by their priority)

 249       * @param array $tr1 first trigger

 250       * @param array $tr2 second trigger

 251       * @return integer

 252       *         0  if same priority

 253       *         -1 if tr1.priority < tr2.priority

 254       *         1  if tr1.priority > tr2.priority

 255       * @access private

 256       */
 257  	function compareTriggers($tr1 , $tr2) {
 258          if ($tr1[2] == $tr2[2]) {
 259              return 0;
 260          }
 261          return ($tr1[2] < $tr2[2]) ? -1 : 1;
 262      }

 362      /**

 363       * executing the transaction (triggers, prepare SQL)

 364       * @access protected

 365       */
 366  	function doTransaction() {
 367          tNG_log::log('tNG' . $this->transactionType, 'doTransaction', 'begin');
 368          
 369          $tmp = $this->getError();
 370          if (is_object($tmp)) {
 371              $this->setError($tmp);
 372              $this->executeTriggers("ERROR");
 373              tNG_log::log('tNG' . $this->transactionType, 'doTransaction', 'end');
 374              return false;
 375          }
 376          
 377          //calling the before triggers and terminate execution if we had an error

 378          $ret = $this->executeTriggers("BEFORE");
 379          if (is_object($ret)) {
 380              $this->setError($ret);
 381              $this->executeTriggers("ERROR");
 382              tNG_log::log('tNG' . $this->transactionType, 'doTransaction', 'end');
 383              return false;
 384          }
 385          
 386          //process the SQL for eventual auto-generation

 387          $ret = $this->prepareSQL();
 388          if (is_object($ret)) {
 389              tNG_log::log('KT_ERROR');            
 390              $this->setError($ret);
 391              $this->executeTriggers("ERROR");
 392              tNG_log::log('tNG' . $this->transactionType, 'doTransaction', 'end');
 393              return false;
 394          }
 395  
 396          $ret = $this->getError();
 397          if (is_object($ret)) {
 398              $this->executeTriggers("ERROR");
 399              tNG_log::log('tNG' . $this->transactionType, 'doTransaction', 'end');
 400              return false;
 401          }
 402          
 403          //executing the transaction

 404          if ($this->sql != '') {
 405              tNG_log::log('tNG' . $this->transactionType, 'executeTransaction', 'execute sql');
 406              if (!is_array($this->sql)) {
 407                  $this->transactionResult = $this->connection->Execute($this->sql);
 408              } else {
 409                  for ($i=0;$i<sizeof($this->sql);$i++) {
 410                      $this->transactionResult = $this->connection->Execute($this->sql[$i], $this->connection);
 411                      if ($this->transactionResult === false) {
 412                          break;
 413                      }
 414                  }
 415              }
 416  
 417              //check if the transaction has been done OK

 418              if (!$this->transactionResult) {
 419                  $ret = $this->parseSQLError($this->sql, $this->connection->ErrorMsg());
 420                  $this->setError($ret);
 421                  tNG_log::log('KT_ERROR');
 422                  $this->executeTriggers("ERROR");
 423                  tNG_log::log('tNG' . $this->transactionType, 'doTransaction', 'end');
 424                  return false;
 425              }
 426          }
 427  
 428          $ret = $this->postExecuteSql();
 429          if (is_object($ret)) {
 430              $this->setError($ret);
 431              $this->executeTriggers("ERROR");
 432              tNG_log::log('tNG' . $this->transactionType, 'doTransaction', 'end');
 433              return false;
 434          }
 435                  
 436          //if the SQL is a SELECT statement

 437          if (is_object($this->transactionResult)) {
 438              if ($this->transactionResult->RecordCount() == 0) {
 439                  $this->transactionResult = null;
 440              }
 441          }
 442                      
 443          //calling the after triggers

 444          $ret = $this->executeTriggers("AFTER");
 445          if (is_object($ret)) {
 446              $this->setError($ret);
 447              $this->executeTriggers("ERROR");
 448              tNG_log::log('tNG' . $this->transactionType, 'doTransaction', 'end');
 449              return false;
 450          }
 451          
 452          $ret = $this->executeTriggers("END");
 453          if (is_object($ret)) {
 454              $this->setError($ret);
 455              $this->executeTriggers("ERROR");
 456              tNG_log::log('tNG' . $this->transactionType, 'doTransaction', 'end');
 457              return false;
 458          }
 459          tNG_log::log('tNG' . $this->transactionType, 'doTransaction', 'end');
 460          return true;
 461      }
