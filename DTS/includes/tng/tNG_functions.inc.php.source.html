 150  function tNG_prepareValues(&$colDetails) {
 151      $type2alt = array(
 152          'CHECKBOX_1_0_TYPE'=>'1',
 153          'CHECKBOX_-1_0_TYPE'=>'-1',
 154          'CHECKBOX_YN_TYPE'=>"Y",
 155          'CHECKBOX_TF_TYPE'=>"t",
 156      );
 157      if (isset($colDetails['method']) && isset($colDetails['reference']) && isset($colDetails['type'])) {
 158          $colValue = KT_getRealValue($colDetails['method'], $colDetails['reference']);
 159          if ($colDetails['method'] == 'VALUE') {
 160              $colValue = KT_DynamicData($colValue, null);
 161              if (isset($colDetails['default'])) {
 162                  $colDetails['default'] = $colValue;
 163              }
 164          } elseif (isset($colDetails['default'])) {
 165              $colDetails['default'] = KT_DynamicData($colDetails['default'], null);
 166          }
 167          switch ($colDetails['type']) {
 168              case 'CHECKBOX_YN_TYPE':
 169              case 'CHECKBOX_1_0_TYPE':
 170              case 'CHECKBOX_-1_0_TYPE':
 171              case 'CHECKBOX_TF_TYPE':
 172                  $colValue = !isset($colValue) ?  '' : $type2alt[$colDetails['type']];
 173                  break;
 174              case 'DATE_TYPE':
 175              case 'DATE_ACCESS_TYPE':
 176                  $colValue = KT_formatDate2DB($colValue);
 177                  if (isset($colDetails['default'])) {
 178                      $colDetails['default'] = KT_formatDate2DB($colDetails['default']);
 179                  }
 180                  break;
 181          }
 182      } else {
 183          $colValue = "";
 184      }
 185      $colDetails['value'] =  $colValue;
 186  }

 198      $absPath_running_script = str_replace("\\", "/", $_SERVER['PATH_TRANSLATED']);
 199      $absPath_running_script = str_replace('//', '/', $absPath_running_script);
 200  
 201      // try to match $tNGinc_path into $absPath_running_script and get the remainings 

 202      // ( remainings = the relative path of the current file to the root of the site)

 203  
 204      $pos = strpos(strtolower($absPath_running_script), strtolower($tNGinc_path));
 205      if ($pos === false) {
 206          $valability_path = "/";
 207      } else {
 208          // build the relPath_running_script as the remaining after removing $tNGinc_path from $absPath_running_script

 209          $relPath_running_script = substr($absPath_running_script, $pos + strlen($tNGinc_path));
 210          
 211          $url_running_script = $_SERVER['PHP_SELF'];
 212          // to get valability path must remove $relPath_running_script from $url_running_script

 213          $pos = strpos(strtolower($url_running_script), strtolower($relPath_running_script));
 214          if ($pos === false) {
 215              $valability_path = "/";
 216          } else {
 217              $valability_path = substr($url_running_script, 0, $pos);
 218          }
 219      }
 220      $parts = explode("/",$valability_path);
 221      $partsURL = array_map("rawurlencode",$parts);
 222      $valability_path = implode("/", $partsURL);

 343   * @param string $file the image file name

 344   * @param string $md5 md5 string

 345   * @return nothing

 348  function tNG_deleteThumbnails($folder, $fileName, $md5) {
 349      if ($fileName != '') {
 350          $path_info = KT_pathinfo($fileName);
 351          if ($md5 == '') {
 352              $regexp = '/^'.preg_quote($path_info['filename'],'/').'_\d+x\d+(_w_(\w+))?';
 353          } else {
 354              $regexp = '/^'.preg_quote($path_info['filename'],'/').'(_w_(\w+))?';
 355          }
 356          if ($path_info['extension'] != "") {
 357              $regexp    .= '\.'.preg_quote($path_info['extension'],'/');
 358          }
 359          $regexp    .= '$/i';
 360              
 361          $folderObj = new KT_folder();
 362          $entry = $folderObj->readFolder($folder, false); 
 363          if (!$folderObj->hasError()) {
 364              foreach($entry['files'] as $key => $fDetail) {
 365                  if (preg_match($regexp, $fDetail['name'], $matches)) {
 366                      if ($md5 != '') {
 367                          if (isset($matches[2]) && $matches[2] != $md5) {
 368                              @unlink($folder . $fDetail['name']);
 369                          }
 370                      } else {
 371                          @unlink($folder . $fDetail['name']);
 372                      }                    
 373                  }
 374              }
 375          }        
 376      }
 377  }
