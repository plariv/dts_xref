  78  function UNI_mask2regexp(txt) {
  79      txt = txt.replace(/([-\/\[\]()\*\+])/g, '\\$1');
  80      txt = txt.replace(/9/g, '\\d');
  81      txt = txt.replace(/\?/g, '.');
  82      txt = txt.replace(/X/g, '\\w');
  83      txt = txt.replace(/A/g, '[A-Za-z]');
  84      var re = new RegExp('^' + txt + '$');
  85      return re;
  86  }

  88  function UNI_regexp2regexp(txt) {
  89      var sep = txt.substring(0, 1);
  90      var pos = txt.lastIndexOf(sep);
  91  
  92      var reg = txt.substring(1, pos);
  93      var mods = '';
  94      if (pos+1 <= txt.length-1) {
  95          var mods = txt.substring(pos+1, txt.length);
  96      }
  97      var re = new RegExp(reg, mods);
  98      return re;
  99  }

 101  function UNI_init_error_elements(el) {
 102      var ret = {};
 103      var el_name = el.name;
 104  
 105      var allLabels = document.getElementsByTagName("LABEL");
 106      for(var i=0; i < allLabels.length; i++) {
 107          if (allLabels[i].htmlFor == el.id) {
 108              ret['label'] = allLabels[i];
 109              break;
 110          }
 111      }
 112      var is_widget = el.getAttribute("wdg:type")!=null;
 113  
 114      if (is_widget) {
 115          var zel = el;
 116          while(zel) {
 117              if(typeof(zel.className) != "undefined" && zel.className.indexOf("widget_container")>=0) {
 118                  break;
 119              }
 120              zel = zel.parentNode;
 121          }
 122          if (zel) {
 123              el = zel;
 124          }
 125      }
 126  
 127      ret['container'] = el.parentNode;
 128      while(ret['container'].nodeType != 1) {
 129          ret['container'] = ret['container'].parentNode;
 130      }
 131  
 132      var dv = document.getElementById(el_name + '_error_element');
 133      if (dv != null) {
 134          ret['error_element'] = dv;
 135      } else {
 136          ret['error_element'] = utility.dom.createElement("DIV", {
 137              'id': el_name + '_error_element'
 138          });
 139          var server_side = utility.dom.getElementsByClassName(el.parentNode, $UNI_CLASSNAME_ERROR_SS);
 140          if (server_side.length > 0) {
 141              el.parentNode.removeChild(server_side[0]);
 142          }
 143          var spn = utility.dom.getElementsByClassName(el.parentNode, 'KT_field_hint')[0];
 144          if (typeof spn == 'undefined') {
 145              spn = el;
 146              if (el.type) {
 147                  if (el.type.toString().toLowerCase() == 'radio') {
 148                      try {
 149                          spn = el.parentNode;
 150                      } catch(e) { spn = el; }
 151                  }
 152              }
 153          }
 154          //find the last element in the container element to insert the error_element after (it should be the last)

 155          while (spn.nextSibling) {
 156              spn = spn.nextSibling;
 157          }
 158          ret['error_element'] = utility.dom.insertAfter(ret['error_element'], spn);
 159      }
 160  
 161      return ret;
 162  }

 288  function UNI_dateBuilder(year, month, day, hour, minutes, seconds) {
 289      var month_length = [31,28,31,30,31,30,31,31,30,31,30,31];
 290      if (! (parseInt(year) > 0)) { return false; }
 291      if (! (parseInt(month) > 0 && parseInt(month) <= 12)) { return false; }
 292      if ((
 293              (parseInt(year) % 4 == 0) 
 294              && 
 295              (parseInt(year) % 100 != 0)
 296          ) 
 297          || 
 298              (parseInt(year) % 400 == 0)
 299          ) {
 300              month_length[1] = 29;
 301      }
 302      if (! (parseInt(day) > 0 && parseInt(day) <= month_length[parseInt(month)-1])) { return false; }
 303  
 304      month_length[1] = 28;
 305      if (! (parseInt(hour) >= 0 && parseInt(hour) <= 23)) { return false; }
 306      if (! (parseInt(minutes) >= 0 && parseInt(minutes) <= 59)) { return false; }
 307      if (! (parseInt(seconds) >= 0 && parseInt(seconds) <= 59)) { return false; }
 308  
 309      year = utility.math.zeroPad(year, 4);
 310      month = utility.math.zeroPad(month, 2);
 311      day = utility.math.zeroPad(day, 2);
 312      hour = utility.math.zeroPad(hour, 2);
 313      minutes = utility.math.zeroPad(minutes, 2);
 314      seconds = utility.math.zeroPad(seconds, 2);
 315  
 316      return year + month + day + hour + minutes + seconds;
 317  }

 328          var newDate = UNI_dateBuilder(o['year'], o['month'], o['day'], o['hour'], o['minutes'], o['seconds']);
 329          if (newDate == false) {
 330              toret = false;
 331          }
 332      }
 333

 716  function UNI_workOnElement(el) {
 717      var toret = true;
 718  
 719      if (el && el.type && el.type.toLowerCase() == 'hidden') {
 720          if (!UNI_isktml(el)) {
 721              return true;
 722          }

 737      }
 738      if (!toret) {
 739          return toret;

 742      // Validation Step 2 : FORMAT UNI_validate_[type]_format

 743      //UNI_validate_text_ip, UNI_validate_text_cc_generic

 744          //if they don't exist, try to exec UNI_validate_text

 748          var functor_type = window['UNI_validate_format_' + props['type']];
 749          var functor_generic = UNI_validate_generic;
 750          if (typeof functor_specific == 'function') {
 751              toret = toret && functor_specific(el, props);
 752          } else if (typeof functor_type == 'function') {
 753              toret = toret && functor_type(el, props);
 754          } else {
 755              toret = toret && functor_generic(el, props);
 756          }
 757      }
 758      if (!toret) {
 759          return toret;
 760      }
 761      //Validation Step 3 : min, max

 762      if (el.value != '') {
 763          if (props['min'] != '' || props['max'] != '') {
 764              var functor_specific = window['UNI_validate_minmax_' + props['type']];
 765              var functor_generic = UNI_validate_minmax;
 766              if (typeof functor_specific == 'function') {
 767                  toret = toret && functor_specific(el, props);
 768              } else if (typeof functor_generic == 'function') {
 769                  toret = toret && functor_generic(el, props);
 770              }
 771          }
 772      }
 773      if (!toret) {
 774          return toret;
 775      }
 776  
 777      //Validation Step 4 : delete errors from old fields that have them

 778      if (toret == true) {
 779          UNI_fieldok_action(el, props);
 780      }
 781      return toret;
 782  }
