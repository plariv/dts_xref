  17   * This class keep the log of the execution of a transaction;

  18   * @access public

  19   */
  20  class tNG_log {
  21      /**

  22       * Constructor. Set the parameters in a global array

  23       * @param string $className the name of the class

  24       * @param string $methodName 

  25       * @param string $message

  26       * @access public

  27       */
  28  	function log($className, $methodName=NULL, $message=null) {
  29          $GLOBALS['KT_logArray'][] = array($className, $methodName, $message);
  30      }
  31      /**

  32       * getter. get trace of the transaction

  33       * @param string $mode if is "text" return the log with html tags striped of

  34       * @access public

  35       */
  36  	function getResult($mode, $uniq='') {
  37          $ret = '';
  38          $alt = '';
  39          $ret .= '<ul id="KT_tngtrace_details'.$uniq.'" style="display:none;">';
  40          $depth = 2;
  41          for ($i=0;$i<count($GLOBALS['KT_logArray']);$i++) {
  42              if (isset($GLOBALS['KT_logArray'][$i+1][0]) && $GLOBALS['KT_logArray'][$i+1][0] == 'KT_ERROR') {
  43                  $alt = ' style="color:red"';
  44              }
  45              if ($GLOBALS['KT_logArray'][$i][0] == 'KT_ERROR') {
  46                  $alt = '';
  47                  continue;
  48              }
  49              if ($GLOBALS['KT_logArray'][$i][2] == 'begin') {
  50                  $ret .= str_repeat(' ', $depth) . "<li$alt>" . $GLOBALS['KT_logArray'][$i][0] . '.' . $GLOBALS['KT_logArray'][$i][1] . ($alt!=''?'*':'') . '</li>' . "\r\n";
  51                  $ret .= "<ul>";
  52                  $depth+=2;
  53              } elseif ($GLOBALS['KT_logArray'][$i][2] == 'end') {
  54                  $ret .= "</ul>";
  55                  $depth-=2;
  56              } else {
  57                  if (!is_null($GLOBALS['KT_logArray'][$i][2])) {
  58                      $ret .= str_repeat(' ', $depth) . "<li$alt>" . $GLOBALS['KT_logArray'][$i][0] . '.' . $GLOBALS['KT_logArray'][$i][1] . ' - ' . $GLOBALS['KT_logArray'][$i][2] . ($alt!=''?'*':'') . '</li>' . "\r\n";
  59                  } else {
  60                      $ret .= str_repeat(' ', $depth) .  "<li$alt>" . $GLOBALS['KT_logArray'][$i][0] . '.' . $GLOBALS['KT_logArray'][$i][1] . ($alt!=''?'*':'') . '</li>' . "\r\n";
  61                  }
  62              }
  63          }
  64          $ret .= "</ul>";
  65          if ($mode == 'text') {
  66              $ret = strip_tags($ret);
  67          }
  68          return $ret;
  69      }
  70  }
