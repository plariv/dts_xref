  48       * Register a trigger to the current transaction

  49       * Or to the multiple transactions. The STARTER and the END triggers are registered to self, and the other triggers are sent to the multiple transactions.

  50       * @param string $triggerType the trigger type (STARTER, AFTER, BEFORE, ERROR or END)

  51       * @param string $triggerName the function name to callback when the trigger is executed

  52       * @param integer $priority the trigger priority

  53       * @return boolean

  54       *         true if the trigger was registered, 

  55       *         false if there is an error (unknown trigger type)

  56       * @access public

  58  	function registerTrigger($triggerType, $callBackFunction, $priority) {
  59          $ret = true;
  60          // check if the trigger type is valid

  61          if (in_array($triggerType,array("STARTER","END"))) {
  62              // copy from parent class

  63              $params = func_get_args();
  64              array_unshift($params,true);
  65              $ret = call_user_func_array(array(&$this, "registerConditionalTrigger"), $params);
  66              // end copy from parent class

  67          } elseif (in_array($triggerType,array("AFTER","BEFORE","ERROR"))) {
  68              $tmp = func_get_args();
  69              array_unshift($tmp,true);
  70              $this->multTriggers[] = $tmp;
  71          } else {
  72              $this->setError(new tNG_error('UNKNOWN_TRIGGER', array(), array($triggerType)));
  73              $ret = false;
  74          }
  75          return $ret;
  76      }

  78      /**

  79       * Register a trigger to the current transaction

  80       * Or to the multiple transactions. The STARTER and the END triggers are registered to self, and the other triggers are sent to the multiple transactions.

  81       * @param string $condition 

  82       * @param string $triggerType the trigger type (STARTER, AFTER, BEFORE, ERROR or END)

  83       * @param string $callBackFunction the function name to callback when the trigger is executed

  84       * @param integer $priority the trigger priority

  86       *         true if the trigger was registered, 

  87       *         false if there is an error (unknown trigger type)

  88       * @access public

  90  	function registerConditionalTrigger($condition, $triggerType, $callBackFunction, $priority) {
  91          $ret = true;
  92          // check if the trigger type is valid

  93          if (in_array($triggerType,array("STARTER","END"))) {
  94              // copy from parent class

  95              $params = array();
  96              if (func_num_args() > 4) {
  97                  $params = array_slice(func_get_args(),4);
  98              }
  99              $this->triggers[$triggerType][] = array($condition, $callBackFunction, $priority, $params);
 100              // end copy from parent class

 101          } elseif (in_array($triggerType,array("AFTER","BEFORE","ERROR"))) {
 102              $tmp = func_get_args();
 103              $this->multTriggers[] = $tmp;
 104              if ($callBackFunction == "Trigger_Default_FormValidation" && isset($this->dispatcher->UnivalProps) && $condition !== true) {
 105                  $this->dispatcher->UnivalProps = array();
 106              }
 107          } else {
 108              $this->setError(new tNG_error('UNKNOWN_TRIGGER', array(), array($triggerType)));
 109              $ret = false;
 110          }
 111          return $ret;
 112      }

 118  	function compileError() {
 119          if (!isset($this->errorWasCompiled)) {
 120              $errObj = &$this->getError();
 121              for ($i=0;$i<sizeof($this->multTNGs);$i++) {
 122                  if ($this->multTNGs[$i]->getError()) {
 123                      $tmp = &$this->multTNGs[$i]->getError();
 124                      $errObj->addDetails('%s', array($tmp->getDetails()), array($tmp->getDeveloperDetails()));
 125                  }
 126              }
 127              $this->errorWasCompiled = true;
 128          }
