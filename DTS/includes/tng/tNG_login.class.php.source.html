  80       * Prepares the custom SQL query to be executed

  83  	function prepareSQL() {
  84          tNG_log::log('tNG_login', 'prepareSQL', 'begin');
  85          
  86          $table = $GLOBALS['tNG_login_config']['table'];
  87          $pk_column = $this->getPrimaryKey();
  88          $user_column = $GLOBALS['tNG_login_config']['user_field'];
  89          $password_column = $GLOBALS['tNG_login_config']['password_field'];
  90  
  91          $sql = "SELECT ".$table.".*, ". KT_escapeFieldName($pk_column) ." AS log_userkey, ". KT_escapeFieldName($user_column) ." AS kt_login_user, ". KT_escapeFieldName($password_column)." AS kt_login_password ";
  92          if (isset($GLOBALS['tNG_login_config']['max_tries']) && isset($GLOBALS['tNG_login_config']['max_tries_field']) && isset($GLOBALS['tNG_login_config']['max_tries_disableinterval']) && isset($GLOBALS['tNG_login_config']['max_tries_disabledate_field']) && $GLOBALS['tNG_login_config']['max_tries']!='' && $GLOBALS['tNG_login_config']['max_tries_field']!='' && $GLOBALS['tNG_login_config']['max_tries_disableinterval']!='' && $GLOBALS['tNG_login_config']['max_tries_disabledate_field']!='') {
  93              $sql .= ', '.KT_escapeFieldName($GLOBALS['tNG_login_config']['max_tries_field']).' AS kt_login_maxtries, '.KT_escapeFieldName($GLOBALS['tNG_login_config']['max_tries_disabledate_field']).' AS kt_login_maxtriesdate ';
  94          }
  95          if (isset($GLOBALS['tNG_login_config']['registration_date_field']) && isset($GLOBALS['tNG_login_config']['expiration_interval_field']) && isset($GLOBALS['tNG_login_config']['expiration_interval_default']) && $GLOBALS['tNG_login_config']['registration_date_field']!='' && $GLOBALS['tNG_login_config']['expiration_interval_default']!='' && $GLOBALS['tNG_login_config']['expiration_interval_field']!='') {
  96              $sql .= ', '.KT_escapeFieldName($GLOBALS['tNG_login_config']['expiration_interval_field']).' AS kt_login_expiration_interval, '.KT_escapeFieldName($GLOBALS['tNG_login_config']['registration_date_field']).' AS kt_login_regdate ';
  97          }
  98          $sql .= " FROM ". $table;
  99          if ($this->loginType == 'form') {
 100              $sql.= " WHERE ".KT_escapeFieldName($user_column). "={kt_login_user}";
 101          } else {
 102              $sql.= " WHERE ".KT_escapeFieldName($pk_column). "={log_userkey}";
