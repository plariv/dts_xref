  15  function nxt_list_collect_checked(form) {
  16      var res = [];
  17      var checkboxes_found = false;
  18      Array_each(utility.dom.getElementsByTagName(form, 'INPUT'), function(input, i){
  19          if (input.type == "checkbox") {
  20              if (input.name.match(/^kt_pk/)) {
  21                  checkboxes_found = true;
  22              }
  23              if (input.name.match(/^kt_pk/) && input.checked) {
  24                  Array_push(res, [input.name, input.value, input]);
  25              }
  26          }
  27      });
  28      if (!checkboxes_found) {
  29          res = null;
  30      }
  31      return res;
  32  }

  56  function nxt_list_submit_inputs(inputs, options, raw_inputs) {
  57      var default_options = {
  58          action: '',
  59          method: 'POST', // form method
  60          rename: true,   // wether to rename inputs before adding
  61          skip_rename: 0  // number of items to skip from renaming at the start of the array
  62      }
  63      // put defaults options in if they were not passed

  64      for (var x in default_options) {
  65          if (typeof options[x] == 'undefined') {
  66              options[x] = default_options[x];
  67          }
  68      }
  69      if (typeof raw_inputs == 'undefined') raw_inputs = [];
  70      // make form

  71      var frm = utility.dom.createElement(
  72          "FORM", 
  73          {action: options.action, method: options.method, style: "display: none"}
  74      );
  75      Array_each(inputs, function(input, i){
  76          var iname = input[0];
  77          if (options.rename && i >= options.skip_rename) {
  78              iname += '_' + (i - options.skip_rename + 1);
  79          }
  80          frm.appendChild(utility.dom.createElement( 'INPUT', { 'type': 'hidden', 'name': iname,  'id': iname, 'value': input[1] }));
  81      });
  82      Array_each(raw_inputs, function(input, i){
  83          frm.appendChild(utility.dom.createElement('INPUT', {'type': 'hidden', 'id': input[0], 'name': input[0], 'value': input[1]}));
  84      });
  85      // post form

  86      document.body.appendChild(frm);
  87      return frm;
  88  }

 101  function nxt_list_edit_link_base(form, page_url) {
 102      if (!nxt_list_check_changed()) {
 103          return false;
 104      }
 105      list_has_changed = false;
 106      // collect

 107      var inputs = nxt_list_collect_checked(form);

 109          return false;
 110      }
 111      if (inputs.length == 0) {
 112          alert(utility.string.getInnerText(NXT_Messages['please_select_record']));
 113          return true;
 114      }
 115  
 116      var ajax_panel = typeof PanelForm_overrideSubmit == 'function';
 117      var res = [];

 119      var variables = [];
 120      if (!ajax_panel) {
 121          variables.push(['KT_back', '1']);

 124      Array_each(qs.keys, function(key, i) {
 125          if (key != 'KT_back') {
 126              Array_push(variables, [key, qs.values[i]]);

 128      });

 131          var hidden = utility.dom.getChildrenByTagName(check[2].parentNode, 'input');

 142          variables
 143      );

 172  function nxt_list_delete_link_base(form) {
 173      if (!nxt_list_check_changed()) {
 174          return false;
 175      }
 176      list_has_changed = false;
 177      // collect

 178      var inputs = nxt_list_collect_checked(form);
 179      if (inputs == null) {
 180          return false;
 181      }
 182  
 183      if (inputs.length == 0) {
 184          alert(utility.string.getInnerText(NXT_Messages['please_select_record']));
 185          return false;
 186      }
 187  
 188      if (!confirm(utility.string.getInnerText(NXT_Messages['are_you_sure_delete']))) {
 189          return false;
 190      }
 191  
 192      //get the hidden id_... from the form

 193      var res = [];
 194      Array_each(inputs, function(check, i) {
 195          var hidden = utility.dom.getChildrenByTagName(check[2].parentNode, 'input');
 196          Array_push(res, [hidden[1].name, hidden[1].value, hidden[1]]);
 197      });
 198  
 199      //build the query string

 200      var res_action = form.form_action;
 201      Array_each(res, function(input, i){
 202          var iname = input[0];
 203          if (i >= 1) {
 204              iname += '_' + (i);
 205          }
 206          res_action += ((res_action.indexOf('?') >= 0) ? '&' : '?') + iname + '=' + input[1];
 207      });
 208  
 209      var ajax_panel = typeof PanelForm_overrideSubmit == 'function';
 210      //if we are inside an ajax panel we must avoid the server side redirect from KT_back 

 211      //so we'll submit directly to the delete transaction; do not add the KT_back parameter

 212      if (!ajax_panel) {
 213          res_action += ((res_action.indexOf('?') >= 0) ? '&' : '?') + 'KT_back=1';
 214      }
 215      var variables = [['KT_Delete1', '1']];
 216      var qs = new QueryString(form.param_name);
 217      Array_each(qs.keys, function(key, i) {
 218          if (key != 'KT_back') {
 219              res_action += ((res_action.indexOf('?') >= 0) ? '&' : '?') + key + '=' + qs.values[i];
 220          }
 221      });
 222  
 223      var frm = nxt_list_submit_inputs(
 224          inputs, 
 225          {action: res_action, method: 'POST'}, 
 226          variables
 227      );
 228      // GET: id_usr=1&id_usr_1=2&KT_back=1

 229      // POST: kt_pk_user_usr_1=1&kt_pk_user_usr_2=2&KT_Delete1=1

 230      if (ajax_panel) {
 231          frm.submit = PanelForm_overrideSubmit;
 232      }
 233      frm.submit();
 234  
 235      return false;
 236  }

 329  function nxt_list_move_link_row(obj, direction) {
 330      // autoselect current

 331      //nxt_list_select_cbx_from_link(obj);

 332      var form = utility.dom.getParentByTagName(obj, 'FORM');
 333      var table = utility.dom.getParentByTagName(obj, 'table');
 334      var row = utility.dom.getParentByTagName(obj, 'tr');
 335  
 336      var row_length = table.getElementsByTagName('tbody')[0].getElementsByTagName('tr').length;
 337      var has_filter = utility.dom.getElementsByClassName(table, "KT_row_filter").length;
 338      row_length = row_length + has_filter;
 339  
 340      var tbody = utility.dom.getParentByTagName(obj, 'tbody');
 341  
 342      row.style.backgroundColor = $nxt_tr_movehighlight_color;
 343      row.setAttribute('moved', '1');
 344  
 345      var first_data_row = nxt_list_first_data_row(table);
 346  
 347      var    index1 = row.rowIndex;
 348      var    index2 = index1 + direction;
 349  
 350      if (direction == 1 && index1 == ( table.rows.length-1 ) ) {
 351          return false;
 352      }
 353      if (direction == -1 && index1 <= first_data_row) {
 354          return false;
 355      }
 356  
 357      list_has_changed = true;
 358  
 359      var r2 = table.rows[index2];
 360      var r1 = table.rows[index1];
 361  
 362      if ( direction == -1 ) {
 363          tbody.insertBefore(tbody.removeChild(r1),r2);
 364      }
 365  
 366      if ( direction == 1 ) {
 367          tbody.insertBefore(tbody.removeChild(r2),r1);
 368      }
 369  
 370      var checked_attr = [];
 371      Array_each([r1, r2], function(row, row_index) {
 372          Array_each(utility.dom.getElementsByTagName(row, 'input'), function(el) {
 373              if (el.type.toLowerCase() == 'checkbox') {
 374                  checked_attr[row_index] = el.checked;
 375              }
 376          });
 377      });
 378      //update the hidden fields : field1, field2 and the collecter

 379      var input1 = utility.dom.getElementsByClassName(r1, 'KT_orderhidden');
 380      var input2 = utility.dom.getElementsByClassName(r2, 'KT_orderhidden');
 381      if (input1.length == 1 && input2.length == 1) {
 382          input1 = input1[0];
 383          input2 = input2[0];
 384          //old_order | new_order

 385          var tmp1 = input1.value.split('|');
 386          var tmp2 = input2.value.split('|');
 387  
 388          input1.value =  tmp1[0] + '|' + tmp2[1];
 389          input2.value = tmp2[0] + '|' + tmp1[1];
 390      }
 391      //set the class names

 392      if (Array_indexOf(utility.dom.getClassNames(r1), 'KT_even') >= 0) {
 393          if (r1.getAttribute('moved') != '1') { utility.dom.classNameRemove(r1, 'KT_even'); }
 394          if (r2.getAttribute('moved') != '1') { utility.dom.classNameAdd(r2, 'KT_even'); }
 395      } else {
 396          if (r1.getAttribute('moved') != '1') { utility.dom.classNameRemove(r2, 'KT_even'); }
 397          if (r2.getAttribute('moved') != '1') { utility.dom.classNameAdd(r1, 'KT_even'); }
 398      }
 399      if (typeof $NXT_LIST_SETTINGS.row_effects != 'undefined' && $NXT_LIST_SETTINGS.row_effects) {
 400          if (Array_indexOf(utility.dom.getClassNames(r1), 'KT_highlight') >= 0) {
 401              if (r1.getAttribute('moved') != '1') { utility.dom.classNameRemove(r1, 'KT_highlight'); }
 402          } else {
 403              if (r1.getAttribute('moved') != '1') { utility.dom.classNameRemove(r2, 'KT_highlight'); }
 404          }
 405      }
 406      Array_each([r1, r2], function(row, row_index) {
 407          Array_each(utility.dom.getElementsByTagName(row, 'input'), function(el) {
 408              if (el.type.toLowerCase() == 'checkbox') {
 409                  el.checked = checked_attr[row_index];
 410              }
 411          });
 412      });
 413      
 414      var first_row = r1.rowIndex > r2.rowIndex ? r2 : r1 ;
 415      var second_row = r1.rowIndex > r2.rowIndex ? r1 : r2 ;
 416  
 417      if (first_row.rowIndex == first_data_row) {
 418          //show all on second, hide on first

 419          nxt_hide_move_link(utility.dom.getElementsByClassName(first_row, 'KT_order')[0], 1);
 420          var tagname = '';
 421  
 422          if ($NXT_LIST_SETTINGS.show_as_buttons) { tagname = 'input';
 423          } else { tagname = 'a'; }
 424  
 425          Array_each(
 426              utility.dom.getElementsByClassName(second_row, 'KT_order')[0].getElementsByTagName(tagname), 
 427              function(el) { el.style.visibility = 'visible'; }
 428          );
 429      } else if (second_row.rowIndex == row_length) {
 430          //show all on first, hide on second_row

 431          nxt_hide_move_link(utility.dom.getElementsByClassName(second_row, 'KT_order')[0], 0);
 432  
 433          if ($NXT_LIST_SETTINGS.show_as_buttons) { tagname = 'input';
 434          } else { tagname = 'a'; }
 435  
 436          Array_each(
 437              utility.dom.getElementsByClassName(first_row, 'KT_order')[0].getElementsByTagName(tagname), 
 438              function(el) { el.style.visibility = 'visible'; }
 439          );
 440      }
 441      if (row_length == 2) {
 442          //hide second on first row, hide first on second row

 443          nxt_hide_move_link(utility.dom.getElementsByClassName(first_row, 'KT_order')[0], 1);    
 444          nxt_hide_move_link(utility.dom.getElementsByClassName(second_row, 'KT_order')[0], 0);
 445      }
 446      var op_link = utility.dom.getElementsBySelector('a.KT_move_op_link');
 447      for (var i =0; i < op_link.length; i++) {
 448          var to_op = op_link[i];
 449          var inp = utility.dom.getNextSiblingByTagName(to_op, 'input');
 450          if (inp != null) {
 451              to_op = inp;
 452          }
 453          to_op.style.display = '';
 454          var th = utility.dom.getParentByTagName(to_op, 'th');
 455          Array_each(th.getElementsByTagName('a'), function(el) {
 456              if (el != op_link[i]) {
 457                  el.style.display = 'none';
 458              }
 459          });
 460          utility.dom.classNameAdd(th, 'KT_order_selected');
 461  
 462      }
 463  
 464      return false;
 465  }

 467  function nxt_list_move_link_form(elem) {
 468      var form = utility.dom.getParentByTagName(elem, 'FORM');
 469  
 470      //collect and fill;

 471      var str = [];
 472      var at_least_one = true;
 473      Array_each(utility.dom.getElementsByTagName(form, 'INPUT'), function(input, i){
 474          if (input.type == "checkbox") {
 475              if (input.name.match(/^kt_pk/)) {
 476                  var tr = utility.dom.getParentByTagName(input, 'tr');
 477                  var other = utility.dom.getElementsByClassName(tr, 'KT_orderhidden');
 478                  if (other.length > 0) {
 479                      var tmp = other[0].value.split('|');
 480                      if (tmp[0] == '') {
 481                          tmp[0] = '0';
 482                      }
 483                      if (tmp[1] == '') {
 484                          tmp[1] = '0';
 485                      }
 486                      Array_push(str, input.value + '|' + tmp.join('|'));
 487                  }
 488              }
 489          }
 490      });
 491      str = str.join(',');
 492      if (at_least_one) {
 493          var ajax_panel = typeof PanelForm_overrideSubmit == 'function';
 494          if (ajax_panel) {
 495              var qs = new QueryString(window.location.hash.replace(/#/, ''));
 496          } else {
 497              var qs = new QueryString(window.location.search.replace(/\?/, ''));
 498          }
 499          var variables = [];
 500          var once = {};
 501          Array_each(qs.keys, function(key, i) {
 502              if (typeof once[key] == 'undefined') {
 503                  once[key] = 1;
 504                  Array_push(variables, key+'='+qs.values[i]);
 505              }
 506          });
 507          var action_url = window.location.protocol + '//' + window.location.host + window.location.pathname + '?' + variables.join('&');
 508          list_has_changed = false;
 509          if (!ajax_panel) {
 510              action_url += window.location.hash;
 511          }
 512  
 513          action_url = action_url.replace(/#$/i, '');
 514          action_url = action_url.replace(/\?$/i, '');
 515          var frm = nxt_list_submit_inputs(
 516              [], 
 517              {action: action_url, method: 'POST'}, 
 518              [[$NXT_MOVE_SETTINGS['orderfield'], str]]
 519          );
 520          if (ajax_panel) {
 521              frm.submit = PanelForm_overrideSubmit;
 522          }
 523          frm.submit();
 524      }
 525      return false;
 526  }
