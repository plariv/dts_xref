 334  function nxt_form_insertasnew(obj, var_name) {
 335      var frm = obj.form;
 336      if (is.ie && frm.action == '') {
 337          var action = window.location.href
 338      } else {
 339          var action = frm.action.toString();
 340      }
 341      parts = action.split("?");
 342      var qs = new QueryString(parts[1]); var new_qs = [];
 343      var re = new RegExp('^'+var_name, 'g');
 344      Array_each(qs.keys, function(key, i) {
 345          if (! key.match(re)) {
 346              Array_push(new_qs, key+'='+qs.values[i]);
 347          }
 348      });
 349      var new_part = new_qs.join('&');
 350      action = parts[0];
 351      if (new_part != '')
 352          action += '?' + new_part;
 353      frm.action = action;
 354      return true;
 355  }
