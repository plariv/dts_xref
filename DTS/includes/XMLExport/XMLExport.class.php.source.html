 148    /**

 149     * This method is the one that generates and exports the XML file.

 150     */
 151    function Execute() {
 152      if ( !isset($this->recordset) || !is_resource($this->recordset) ) {
 153        die('<strong>XMLExport Error.</strong><br/>Passed argument is not a valid recordset.');
 154        return;
 155      }
 156  
 157      if (count($this->columns) < 1) {
 158        die('<strong>XMLExport Error.</strong><br/>No columns defined!');
 159        return;
 160      }
 161          
 162      $document = '';
 163      $document .= '<'.'?xml version="1.0" encoding="' . $this->xmlEncoding . '"?'.'>';
 164      $document .= "\r\n";
 165      $document .= '<' . $this->rootNodeName . '>';
 166      $document .= "\r\n";
 167  
 168      $numRow = 1;
 169      if(mysql_num_rows($this->recordset) > 0) {
 170        mysql_data_seek($this->recordset, 0);
 171      }
 172      while($row_recordset = mysql_fetch_assoc($this->recordset)) {
 173        if ($this->maxRecords != 'ALL' && $this->maxRecords < $numRow) {
 174          break;
 175        }
 176          
 177        $row = "\t";
 178        $row .= '<' . $this->rowNodeName;
 179        if ($this->xmlFormat != 'ATTRIBUTES') {
 180          $row .= '>' . "\r\n";
 181        }
 182          
 183        foreach ($this->columns as $column => $details) {
 184          $colName = $column;
 185          if ($details['label'] != '') {
 186            $colName = $details['label'];
 187          }
 188          $value = $row_recordset[$column];
 189          if ($this->xmlEncoding != $this->dbEncoding) {
 190            if (!function_exists('mb_convert_encoding')) {
 191              die('<strong>XMLExport Error.</strong><br/>Could not perform characters conversion. Function <strong>mb_convert_encoding</strong> is not available! Please enable the <strong>mbstring</strong> extension!');
 192            }
 193            $value = mb_convert_encoding($value, $this->xmlEncoding, $this->dbEncoding);
 194          }
 195          $value = str_replace(array('&', '>', '<', '"'), array('&amp;', '&gt;', '&lt;', '&quot;'), $value);
 196  
 197          if ($this->xmlFormat == 'ATTRIBUTES') {
 198            $row .= ' ' . $colName . '="';
 199            $row .= $value;
 200            $row .= '"';
 201          } else {
 202            $row .= "\t\t";
 203            $row .= '<' . $colName . '>';
 204            $row .= $value;
 205            $row .= '</' . $colName . '>';
 206            $row .= "\r\n";
 207          }
 208        }
 209          
 210        if ($this->xmlFormat == 'ATTRIBUTES') {
 211          $row .= ' />';
 212        } else {
 213          $row .= "\t" . '</' . $this->rowNodeName . '>';
 214        }
 215          
 216        $document .= $row;
 217        $document .= "\r\n";
 218        $numRow++;
 219      }
 220      
 221      $document .= '</' . $this->rootNodeName . '>';
 222      $size = strlen($document);
 223      $this->sendHeaders($size);
 224      echo $document;
 225      exit;
 226    }

 240      header('Cache-control: private');
 241      header('Expires: -1');
 242      header('Content-Length: ' . $size);
 243    }
