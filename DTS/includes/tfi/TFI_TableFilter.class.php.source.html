 436          $dateArr = explode('-', $myDate);
 437          $year = $dateArr[0];
 438          $month = $dateArr[1];

 440          if ( $month < 1 || $month > 12 ) {

 442          }
 443          $maxday = KT_getDaysOfMonth($month, $year);
 444          if ( $day < 1 || $day > $maxday ) {
 445              $day = '01';
 446          }
 447          return array($year, $month, $day);

 450  	function getTimeParts($timePart, $format) {
 451          $myDate = '';
 452          $dateArr = array();
 453          $hour = '';
 454          $min = '';
 455          $sec = '';
 456          
 457          $myDate = KT_convertDate($timePart, $GLOBALS['KT_screen_time_format_internal'], $format);
 458          $dateArr = explode(':', $myDate);
 459          $hour = $dateArr[0];
 460          if (isset($dateArr[1])) {
 461              $min = $dateArr[1];
 462          }
 463          if (isset($dateArr[2])) {
 464              $sec = $dateArr[2];
 465          }
 466          if ( $format != 'HH:ii:ss' && preg_match('/p/i', $timePart) && $hour < 12) {
 467              $hour += 12;
 468          }
 469          return array($hour, $min, $sec);
