  23  	function TSO_TableSorter($rsName, $sorterName) {
  24          $this->rsName = $rsName;
  25          $this->sorterName = $sorterName;
  26          
  27          KT_SessionKtBack(KT_getFullUri());

  31  	function addColumn($colName) {
  32          if (!isset($this->columns[$colName])) {
  33              $this->columns[$colName] = true;
  34          }
  35      }
  36  
  37  	function setDefault($defaultColumn) {
  38          $this->defaultColumn = $defaultColumn;
  39          $sorter_reference = "sorter_" . $this->sorterName;
  40          if (!isset($_SESSION[$sorter_reference])) {
  41              $_SESSION[$sorter_reference] = $defaultColumn;
  42          }

  45  	function Execute() {
  46          $sorter_reference = "sorter_" . $this->sorterName;
  47          
  48          if (isset($_GET[$sorter_reference])) {
  49              $sorterString = $_GET[$sorter_reference];
  50              $columnName = str_replace(" DESC", "", $sorterString);
  51              if (isset($this->columns[$columnName])) {
  52                  $_SESSION[$sorter_reference]=$_GET[$sorter_reference];
  53                  $url = KT_addReplaceParam(KT_getFullUri(), $sorter_reference);
  54                  KT_redir($url);
  55              }
  56          }
  57      }

  59      // Get Current Sort

  60  	function getCurrentSort() {
  61          $value = $this->defaultColumn;
  62          $sorter_reference = "sorter_" . $this->sorterName;
  63        if (isset($_SESSION[$sorter_reference])) {
  64              $value = $_SESSION[$sorter_reference];
  65        }
  66        return $value;
  67      }

  70  	function getSortIcon($column){
  71        $value = $this->getCurrentSort();

  73        if ($value == $column) {
  74              return 'KT_asc';
  75        } elseif ($value == $column.' DESC') {
  76              return 'KT_desc';
  77        }
  78      }
