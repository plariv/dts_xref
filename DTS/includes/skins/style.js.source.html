  28      if (typeof add_event == 'undefined') {
  29          add_event = false;
  30      }
  31      var elnou = utility.dom.createElement('input', {
  32          'type' : 'button', 
  33          'value': el.innerHTML
  34      });
  35  
  36      el.style.display = 'none';
  37      elnou = utility.dom.insertAfter(elnou, el);
  38  
  39      if (add_event) {
  40          var onclick = el.onclick;
  41          elnou.onclick = onclick;
  42      }
  43  
  44      elnou.style.visibility = el.style.visibility;
  45      if (el.innerHTML == '') {
  46          elnou.style.display = 'none';
  47      }
  48  
  49      return elnou;
  50  }

 715  function nxt_style_set_globals() {
 716      $lists = [];
 717      var tmp = utility.dom.getElementsByClassName(document, 'KT_tng', 'div');
 718      for (var k = 0; k < tmp.length; k++) {
 719          var list_obj = {};
 720          var kt_styles_attached = tmp[k].getAttribute("kt_styles_attached");
 721          if (kt_styles_attached) {
 722              continue;
 723          }
 724          list_obj.kt_styles_attached = false;
 725          list_obj.name = tmp[k].id;
 726          list_obj.main = tmp[k];
 727          list_obj.inner = utility.dom.getElementsByClassName(tmp[k], 'KT_tnglist', 'div');
 728          if (typeof list_obj.inner == 'object' && list_obj.inner != null && list_obj.inner.length && list_obj.inner.length > 0) {
 729              list_obj.inner = list_obj.inner[0];
 730              if (is.mozilla) {
 731                  utility.dom.classNameAdd(list_obj.inner, 'fix_content_enlarge');
 732              }
 733          
 734              var frm = list_obj.inner.getElementsByTagName('form')[0];
 735              for (var i = 0; i < frm.childNodes.length; i++) {
 736                  if (frm.childNodes[i].nodeType == 1) {
 737                      var n = frm.childNodes[i];
 738                      var t = n.tagName.toLowerCase();
 739                      var c = n.className;
 740                      if (t == 'table') {
 741                          list_obj.table = n;
 742                      }
 743                      if (/KT_topbuttons/.test(c)) {
 744                          list_obj.topbuttons = n;
 745                      }
 746                      if (/KT_bottombuttons/.test(c)) {
 747                          list_obj.bottombuttons = n;
 748                      }
 749                      if (/KT_topnav/.test(c)) {
 750                          list_obj.topnav = n;
 751                          var divs = list_obj.topnav.getElementsByTagName('div');
 752                          for (var j = 0; j < divs.length; j++) {
 753                              if (/KT_textnav/.test(divs[j].className)) {
 754                                  list_obj.toptextnav = divs[j];
 755                                  break;
 756                              }
 757                          }
 758                      }
 759                      if (/KT_bottomnav/.test(c)) {
 760                          list_obj.bottomnav = n;
 761                          var divs = list_obj.bottomnav.getElementsByTagName('div');
 762                          for (var j = 0; j < divs.length; j++) {

 765                                  break;
 766                              }
