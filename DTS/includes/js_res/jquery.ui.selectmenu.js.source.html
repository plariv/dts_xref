 304          if (selectOptionData.length) {
 305              for (var i = 0; i < selectOptionData.length; i++) {
 306                  var thisLiAttr = { role : 'presentation' };
 307                  if ( selectOptionData[ i ].disabled ) {
 308                      thisLiAttr[ 'class' ] = this.namespace + '-state-disabled';

 310                  var thisAAttr = {
 311                      html: selectOptionData[i].text || '&nbsp;',
 312                      href : '#nogo',
 313                      tabindex : -1,
 314                      role : 'option',
 315                      'aria-selected' : false
 316                  };
 317                  if ( selectOptionData[ i ].disabled ) {
 318                      thisAAttr[ 'aria-disabled' ] = selectOptionData[ i ].disabled;
 319                  }

 321                      thisAAttr[ 'typeahead' ] = selectOptionData[ i ].typeahead;
 322                  }                
 323                  var thisA = $('<a/>', thisAAttr);
 324                  var thisLi = $('<li/>', thisLiAttr)    
 325                      .append(thisA)                
 326                      .data('index', i)
 327                      .addClass(selectOptionData[i].classes)
 328                      .data('optionClasses', selectOptionData[i].classes || '')
 329                      .bind("mouseup.selectmenu", function(event) {
 330                          if (self._safemouseup && !self._disabled(event.currentTarget) && !self._disabled($( event.currentTarget ).parents( "ul>li." + self.widgetBaseClass + "-group " )) ) {
 331                              var changed = $(this).data('index') != self._selectedIndex();
 332                              self.index($(this).data('index'));

 334                              if (changed) {

' ' + this.namespace + '-state-disabled" aria-disabled="true"' : '"' ) + '><span class="' + self.widgetBaseClass + '-group-label">' + selectOptionData[i].parentOptGroup.attr('label') + '</span><ul></ul></li> ')

 477      _typeAhead: function( code, eventType ) {
 478          var self = this,
 479              c = String.fromCharCode(code).toLowerCase(),
 480              matchee = null,
 481              nextIndex = null;
 482  
 483          // Clear any previous timer if present

 484          if ( self._typeAhead_timer ) {
 485              window.clearTimeout( self._typeAhead_timer );
 486              self._typeAhead_timer = undefined;
 487          }
 488  
 489          // Store the character typed

 490          self._typeAhead_chars = (self._typeAhead_chars === undefined ? "" : self._typeAhead_chars).concat(c);
 491  
 492          // Detect if we are in cyciling mode or direct selection mode

 493          if ( self._typeAhead_chars.length < 2 ||
 494               (self._typeAhead_chars.substr(-2, 1) === c && self._typeAhead_cycling) ) {
 495              self._typeAhead_cycling = true;
 496  
 497              // Match only the first character and loop

 498              matchee = c;
 499          }
 500          else {
 501              // We won't be cycling anymore until the timer expires

 502              self._typeAhead_cycling = false;
 503  
 504              // Match all the characters typed

 505              matchee = self._typeAhead_chars;
 506          }
 507  
 508          // We need to determine the currently active index, but it depends on

 509          // the used context: if it's in the element, we want the actual

 510          // selected index, if it's in the menu, just the focused one

 511          // I copied this code from _moveSelection() and _moveFocus()

 512          // respectively --thg2k

 513          var selectedIndex = (eventType !== 'focus' ?
 514              this._selectedOptionLi().data('index') :
 515              this._focusedOptionLi().data('index')) || 0;
 516  
 517          for (var i = 0; i < this._optionLis.length; i++) {
 518              var thisText = this._optionLis.eq(i).text().substr(0, matchee.length).toLowerCase();
 519  
 520              if ( thisText === matchee ) {
 521                  if ( self._typeAhead_cycling ) {
 522                      if ( nextIndex === null )
 523                          nextIndex = i;
 524  
 525                      if ( i > selectedIndex ) {
 526                          nextIndex = i;
 527                          break;
 528                      }
 529                  } else {
 530                      nextIndex = i;
 531                  }
 532              }
 533          }
 534  
 535          if ( nextIndex !== null ) {
 536              // Why using trigger() instead of a direct method to select the

 537              // index? Because we don't what is the exact action to do, it

 538              // depends if the user is typing on the element or on the popped

 539              // up menu

 540              this._optionLis.eq(nextIndex).find("a").trigger( eventType );
 541          }
 542  
 543          self._typeAhead_timer = window.setTimeout(function() {
 544              self._typeAhead_timer = undefined;
 545              self._typeAhead_chars = undefined;
 546              self._typeAhead_cycling = undefined;
 547          }, self.options.typeAhead);
 548      },

 560      open: function(event) {
 561          var self = this, o = this.options;
 562          if ( self.newelement.attr("aria-disabled") != 'true' ) {
 563              self._closeOthers(event);
 564              self.newelement.addClass('ui-state-active');
 565                  
 566              self.listWrap.appendTo( o.appendTo );
 567              self.list.attr('aria-hidden', false);            
 568              self.listWrap.addClass( self.widgetBaseClass + '-open' );
 569                          
 570              var selected = this._selectedOptionLi();
 571              if ( o.style == "dropdown" ) {
 572                  self.newelement.removeClass('ui-corner-all').addClass('ui-corner-top');
 573              } else {                
 574                  // center overflow and avoid flickering

 575                  this.list
 576                      .css("left", -5000)
 577                      .scrollTop( this.list.scrollTop() + selected.position().top - this.list.outerHeight()/2 + selected.outerHeight()/2 )
 578                      .css("left","auto");
 579              }
 580              
 581              self._refreshPosition();    
 582              
 583              var link = selected.find("a");
 584              if (link.length) link[0].focus();        
 585              
 586              self.isOpen = true;
 587              self._trigger("open", event, self._uiHash());
 588          }

 597              if ( this.options.style == "dropdown" ) {
 598                  this.newelement.removeClass('ui-corner-top').addClass('ui-corner-all');

 600              if ( retainFocus ) {

 602              }
 603              this.isOpen = false;
 604              this._trigger("close", event, this._uiHash());
 605          }
