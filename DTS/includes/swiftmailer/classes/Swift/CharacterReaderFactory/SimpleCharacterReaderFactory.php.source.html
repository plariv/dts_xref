 102    /**

 103     * Returns a CharacterReader suitable for the charset applied.

 104     * @param string $charset

 105     * @return Swift_CharacterReader

 106     */
 107    public function getReaderFor($charset)
 108    {
 109      $charset = trim(strtolower($charset));
 110      foreach (self::$_map as $pattern => $spec)
 111      {
 112        $re = '/^' . $pattern . '$/D';
 113        if (preg_match($re, $charset))
 114        {
 115          if (!array_key_exists($pattern, self::$_loaded))
 116          {
 117            $reflector = new ReflectionClass($spec['class']);
 118            if ($reflector->getConstructor())
 119            {
 120              $reader = $reflector->newInstanceArgs($spec['constructor']);
 121            }
 122            else
 123            {
 124              $reader = $reflector->newInstance();
 125            }
 126            self::$_loaded[$pattern] = $reader;
 127          }
 128          return self::$_loaded[$pattern];
 129        }
 130      }
 131    }
