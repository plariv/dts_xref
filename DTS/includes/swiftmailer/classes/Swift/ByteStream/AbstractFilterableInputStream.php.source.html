  13   * Provides the base functionality for an InputStream supporting filters.

  14   * @package Swift

  15   * @subpackage ByteStream

  16   * @author Chris Corbyn

  17   */
  18  abstract class Swift_ByteStream_AbstractFilterableInputStream
  19    implements Swift_InputByteStream, Swift_Filterable
  20  {
  21    
  22    /** Write sequence */

  23    private $_sequence = 0;
  24    
  25    /** StreamFilters */

  26    private $_filters = array();
  27    
  28    /** A buffer for writing */

  29    private $_writeBuffer = '';
  30    
  31    /** Bound streams */

  32    private $_mirrors = array();
  33    
  34    /**

  35     * Commit the given bytes to the storage medium immediately.

  36     * @param string $bytes

  37     * @access protected

  38     */
  39    abstract protected function _commit($bytes);
  40    
  41    /**

  42     * Flush any buffers/content with immediate effect.

  43     * @access protected

  44     */
  45    abstract protected function _flush();
  46    
  47    /**

  48     * Add a StreamFilter to this InputByteStream.

  49     * @param Swift_StreamFilter $filter

  50     * @param string $key

  51     */
  52    public function addFilter(Swift_StreamFilter $filter, $key)
  53    {
  54      $this->_filters[$key] = $filter;
  55    }
  56    
  57    /**

  58     * Remove an already present StreamFilter based on its $key.

  59     * @param string $key

  60     */
  61    public function removeFilter($key)
  62    {
  63      unset($this->_filters[$key]);
  64    }
  65    
  66    /**

  67     * Writes $bytes to the end of the stream.

  68     * @param string $bytes

  69     * @throws Swift_IoException

  70     */
  71    public function write($bytes)
  72    {
  73      $this->_writeBuffer .= $bytes;
  74      foreach ($this->_filters as $filter)
  75      {
  76        if ($filter->shouldBuffer($this->_writeBuffer))
  77        {
  78          return;
  79        }
  80      }
  81      $this->_doWrite($this->_writeBuffer);
  82      return ++$this->_sequence;
  83    }
  84    
  85    /**

  86     * For any bytes that are currently buffered inside the stream, force them

  87     * off the buffer.

  88     * 

  89     * @throws Swift_IoException

  90     */
  91    public function commit()
  92    {
  93      $this->_doWrite($this->_writeBuffer);
  94    }
  95    
  96    /**

  97     * Attach $is to this stream.

  98     * The stream acts as an observer, receiving all data that is written.

  99     * All {@link write()} and {@link flushBuffers()} operations will be mirrored.

 100     * 

 101     * @param Swift_InputByteStream $is

 102     */
 103    public function bind(Swift_InputByteStream $is)
 104    {
 105      $this->_mirrors[] = $is;
 106    }
 107    
 108    /**

 109     * Remove an already bound stream.

 110     * If $is is not bound, no errors will be raised.

 111     * If the stream currently has any buffered data it will be written to $is

 112     * before unbinding occurs.

 113     * 

 114     * @param Swift_InputByteStream $is

 115     */
 116    public function unbind(Swift_InputByteStream $is)
 117    {
 118      foreach ($this->_mirrors as $k => $stream)
 119      {
 120        if ($is === $stream)
 121        {
 122          if ($this->_writeBuffer !== '')
 123          {
 124            $stream->write($this->_filter($this->_writeBuffer));
 125          }
 126          unset($this->_mirrors[$k]);
 127        }
 128      }
 129    }
 130    
 131    /**

 132     * Flush the contents of the stream (empty it) and set the internal pointer

 133     * to the beginning.

 134     * @throws Swift_IoException

 135     */
 136    public function flushBuffers()
 137    {
 138      if ($this->_writeBuffer !== '')
 139      {
 140        $this->_doWrite($this->_writeBuffer);
 141      }
 142      $this->_flush();
 143      
 144      foreach ($this->_mirrors as $stream)
 145      {
 146        $stream->flushBuffers();
 147      }
 148    }
 149    
 150    // -- Private methods

 151    
 152    /** Run $bytes through all filters */

 153    private function _filter($bytes)
 154    {
 155      foreach ($this->_filters as $filter)
 156      {
 157        $bytes = $filter->filter($bytes);
 158      }
 159      return $bytes;
 160    }
 161    
 162    /** Just write the bytes to the stream */

 163    private function _doWrite($bytes)
 164    {
 165      $this->_commit($this->_filter($bytes));
 166      
 167      foreach ($this->_mirrors as $stream)
 168      {
 169        $stream->write($bytes);
 170      }
 171      
 172      $this->_writeBuffer = '';
 173    }
 174    
 175  }
