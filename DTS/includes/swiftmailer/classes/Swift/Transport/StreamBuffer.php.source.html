 109    {
 110      if (isset($this->_stream))
 111      {
 112        switch ($this->_params['type'])
 113        {
 114          case self::TYPE_PROCESS:
 115            fclose($this->_in);
 116            fclose($this->_out);
 117            proc_close($this->_stream);
 118            break;

 120          default:
 121            fclose($this->_stream);
 122            break;
 123        }
 124      }

 126      $this->_out = null;
 127      $this->_in = null;

 130    /**

 131     * Set an array of string replacements which should be made on data written

 132     * to the buffer.  This could replace LF with CRLF for example.

 133     * @param string[] $replacements

 134     */
 135    public function setWriteTranslations(array $replacements)
 136    {
 137      foreach ($this->_translations as $search => $replace)
 138      {
 139        if (!isset($replacements[$search]))
 140        {
 141          $this->removeFilter($search);
 142          unset($this->_translations[$search]);
 143        }
 144      }
 145      
 146      foreach ($replacements as $search => $replace)
 147      {
 148        if (!isset($this->_translations[$search]))
 149        {
 150          $this->addFilter(
 151            $this->_replacementFactory->createFilter($search, $replace), $search
 152            );
 153          $this->_translations[$search] = true;
 154        }
 155      }
 156    }

 242    /**

 243     * Establishes a connection to a remote server.

 244     * @access private

 245     */
 246    private function _establishSocketConnection()
 247    {
 248      $host = $this->_params['host'];
 249      if (!empty($this->_params['protocol']))
 250      {
 251        $host = $this->_params['protocol'] . '://' . $host;
 252      }
 253      $timeout = 15;
 254      if (!empty($this->_params['timeout']))
 255      {
 256        $timeout = $this->_params['timeout'];
 257      }
 258      $options = array();
 259      if (!empty($this->_params['sourceIp']))
 260      {
 261          $options['socket']['bindto']=$this->_params['sourceIp'].':0';
 262      }
 263      $this->_stream = @stream_socket_client($host.':'.$this->_params['port'], $errno, $errstr, $timeout, STREAM_CLIENT_CONNECT, stream_context_create($options));
 264      if (false === $this->_stream)
 265      {
 266        throw new Swift_TransportException(
 267          'Connection could not be established with host ' . $this->_params['host'] .
 268          ' [' . $errstr . ' #' . $errno . ']'
 269          );
 270      }
 271      if (!empty($this->_params['blocking']))
 272      {
 273        stream_set_blocking($this->_stream, 1);
 274      }
 275      else
 276      {
 277        stream_set_blocking($this->_stream, 0);
 278      }
 279      stream_set_timeout($this->_stream, $timeout);
 280      $this->_in =& $this->_stream;
 281      $this->_out =& $this->_stream;
 282    }

 284    /**

 285     * Opens a process for input/output.

 286     * @access private

 287     */
 288    private function _establishProcessConnection()
 289    {
 290      $command = $this->_params['command'];
 291      $descriptorSpec = array(
 292        0 => array('pipe', 'r'),
 293        1 => array('pipe', 'w'),
 294        2 => array('pipe', 'w')
 295        );
 296      $this->_stream = proc_open($command, $descriptorSpec, $pipes);
 297      stream_set_blocking($pipes[2], 0);
 298      if ($err = stream_get_contents($pipes[2]))
 299      {
 300        throw new Swift_TransportException(
 301          'Process could not be started [' . $err . ']'
 302          );
 303      }
 304      $this->_in =& $pipes[0];
 305      $this->_out =& $pipes[1];
 306    }

 309    private function _getReadConnectionDescription()
 310    {
 311      switch ($this->_params['type'])
 312      {
 313        case self::TYPE_PROCESS:
 314          return 'Process '.$this->_params['command'];
 315          break;
 316          
 317        case self::TYPE_SOCKET:
 318        default:
 319          $host = $this->_params['host'];
 320          if (!empty($this->_params['protocol']))
 321          {
 322            $host = $this->_params['protocol'] . '://' . $host;
 323          }
 324          $host.=':'.$this->_params['port'];
 325          return $host;
 326          break;
 327      }
 328    }
