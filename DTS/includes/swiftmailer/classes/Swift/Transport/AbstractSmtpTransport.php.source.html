 100     * Start the SMTP connection.

 102    public function start()
 103    {
 104      if (!$this->_started)
 105      {
 106        if ($evt = $this->_eventDispatcher->createTransportChangeEvent($this))
 107        {
 108          $this->_eventDispatcher->dispatchEvent($evt, 'beforeTransportStarted');
 109          if ($evt->bubbleCancelled())
 110          {
 111            return;
 112          }
 113        }
 114        
 115        try
 116        {
 117          $this->_buffer->initialize($this->_getBufferParams());
 118        }
 119        catch (Swift_TransportException $e)
 120        {
 121          $this->_throwException($e);
 122        }
 123        $this->_readGreeting();
 124        $this->_doHeloCommand();
 125        
 126        if ($evt)
 127        {
 128          $this->_eventDispatcher->dispatchEvent($evt, 'transportStarted');
 129        }
 130        
 131        $this->_started = true;
 132      }
 133    }

 145    /**

 146     * Send the given Message.

 147     * 

 148     * Recipient/sender data will be retrieved from the Message API.

 149     * The return value is the number of recipients who were accepted for delivery.

 150     * 

 151     * @param Swift_Mime_Message $message

 152     * @param string[] &$failedRecipients to collect failures by-reference

 153     * @return int

 154     */
 155    public function send(Swift_Mime_Message $message, &$failedRecipients = null)
 156    {
 157      $sent = 0;
 158      $failedRecipients = (array) $failedRecipients;
 159      
 160      if ($evt = $this->_eventDispatcher->createSendEvent($this, $message))
 161      {
 162        $this->_eventDispatcher->dispatchEvent($evt, 'beforeSendPerformed');
 163        if ($evt->bubbleCancelled())
 164        {
 165          return 0;
 166        }
 167      }
 168      
 169      if (!$reversePath = $this->_getReversePath($message))
 170      {
 171        throw new Swift_TransportException(
 172          'Cannot send message without a sender address'
 173          );
 174      }
 175      
 176      $to = (array) $message->getTo();
 177      $cc = (array) $message->getCc();
 178      $bcc = (array) $message->getBcc();
 179      
 180      $message->setBcc(array());
 181      
 182      try
 183      {
 184        $sent += $this->_sendTo($message, $reversePath, $to, $failedRecipients);
 185        $sent += $this->_sendCc($message, $reversePath, $cc, $failedRecipients);
 186        $sent += $this->_sendBcc($message, $reversePath, $bcc, $failedRecipients);
 187      }
 188      catch (Exception $e)
 189      {
 190        $message->setBcc($bcc);
 191        throw $e;
 192      }
 193      
 194      $message->setBcc($bcc);
 195      
 196      if ($evt)
 197      {
 198        if ($sent == count($to) + count($cc) + count($bcc))
 199        {
 200          $evt->setResult(Swift_Events_SendEvent::RESULT_SUCCESS);
 201        }
 202        elseif ($sent > 0)
 203        {
 204          $evt->setResult(Swift_Events_SendEvent::RESULT_TENTATIVE);
 205        }
 206        else
 207        {
 208          $evt->setResult(Swift_Events_SendEvent::RESULT_FAILED);
 209        }
 210        $evt->setFailedRecipients($failedRecipients);
 211        $this->_eventDispatcher->dispatchEvent($evt, 'sendPerformed');
 212      }
 213      
 214      $message->generateId(); //Make sure a new Message ID is used

 215      
 216      return $sent;
 217    }

 222    public function stop()
 223    {
 224      if ($this->_started)
 225      {
 226        if ($evt = $this->_eventDispatcher->createTransportChangeEvent($this))
 227        {
 228          $this->_eventDispatcher->dispatchEvent($evt, 'beforeTransportStopped');
 229          if ($evt->bubbleCancelled())
 230          {
 231            return;
 232          }
 233        }
 234        
 235        try
 236        {
 237          $this->executeCommand("QUIT\r\n", array(221));
 238        }
 239        catch (Swift_TransportException $e) {}
 240        
 241        try
 242        {
 243          $this->_buffer->terminate();
 244        
 245          if ($evt)
 246          {
 247            $this->_eventDispatcher->dispatchEvent($evt, 'transportStopped');
 248          }
 249        }
 250        catch (Swift_TransportException $e)
 251        {
 252          $this->_throwException($e);
 253        }
 254      }
 255      $this->_started = false;
 256    }

 286    /**

 287     * Run a command against the buffer, expecting the given response codes.

 288     * 

 289     * If no response codes are given, the response will not be validated.

 290     * If codes are given, an exception will be thrown on an invalid response.

 291     * 

 292     * @param string $command

 293     * @param int[] $codes

 294     * @param string[] &$failures

 295     * @return string

 296     */
 297    public function executeCommand($command, $codes = array(), &$failures = null)
 298    {
 299      $failures = (array) $failures;
 300      $seq = $this->_buffer->write($command);
 301      $response = $this->_getFullResponse($seq);
 302      if ($evt = $this->_eventDispatcher->createCommandEvent($this, $command, $codes))
 303      {
 304        $this->_eventDispatcher->dispatchEvent($evt, 'commandSent');
 305      }
 306      $this->_assertResponseCode($response, $codes);
 307      return $response;
 308    }

 365    /** Determine the best-use reverse path for this message */

 366    protected function _getReversePath(Swift_Mime_Message $message)
 367    {
 368      $return = $message->getReturnPath();
 369      $sender = $message->getSender();
 370      $from = $message->getFrom();
 371      $path = null;
 372      if (!empty($return))
 373      {
 374        $path = $return;
 375      }
 376      elseif (!empty($sender))
 377      {
 378        // Don't use array_keys

 379        reset($sender); // Reset Pointer to first pos

 380        $path = key($sender); // Get key

 381      }
 382      elseif (!empty($from))
 383      {
 384        reset($from); // Reset Pointer to first pos

 385        $path = key($from); // Get key

 386      }
 387      return $path;
 388    }

 407    /** Throws an Exception if a response code is incorrect */

 408    protected function _assertResponseCode($response, $wanted)
 409    {
 410      list($code) = sscanf($response, '%3d');
 411      $valid = (empty($wanted) || in_array($code, $wanted));
 412      
 413      if ($evt = $this->_eventDispatcher->createResponseEvent($this, $response,
 414        $valid))
 415      {
 416        $this->_eventDispatcher->dispatchEvent($evt, 'responseReceived');
 417      }
 418      
 419      if (!$valid)
 420      {
 421        $this->_throwException(
 422          new Swift_TransportException(
 423            'Expected response code ' . implode('/', $wanted) . ' but got code ' .
 424            '"' . $code . '", with message "' . $response . '"'
 425            )
 426          );
 427      }
 428    }

 430    /** Get an entire multi-line response using its sequence number */

 431    protected function _getFullResponse($seq)
 432    {
 433      $response = '';
 434      try
 435      {
 436        do
 437        {
 438          $line = $this->_buffer->readLine($seq);
 439          $response .= $line;
 440        }
 441        while (null !== $line && false !== $line && ' ' != $line{3});
 442      }
 443      catch (Swift_TransportException $e)
 444      {
 445        $this->_throwException($e);
 446      }
 447      return $response;
 448    }

 452    /** Send an email to the given recipients from the given reverse path */

 453    private function _doMailTransaction($message, $reversePath,
 454      array $recipients, array &$failedRecipients)
 455    {
 456      $sent = 0;
 457      $this->_doMailFromCommand($reversePath);
 458      foreach ($recipients as $forwardPath)
 459      {
 460        try
 461        {
 462          $this->_doRcptToCommand($forwardPath);
 463          $sent++;
 464        }
 465        catch (Swift_TransportException $e)
 466        {
 467          $failedRecipients[] = $forwardPath;
 468        }
 469      }
 470      
 471      if ($sent != 0)
 472      {
 473        $this->_doDataCommand();
 474        $this->_streamMessage($message);
 475      }
 476      else
 477      {
 478        $this->reset();
 479      }
 480      
 481      return $sent;
 482    }

 508    /** Send a message to all Bcc: recipients */

 509    private function _sendBcc(Swift_Mime_Message $message, $reversePath,
 510      array $bcc, array &$failedRecipients)
 511    {
 512      $sent = 0;
 513      foreach ($bcc as $forwardPath => $name)
 514      {
 515        $message->setBcc(array($forwardPath => $name));
 516        $sent += $this->_doMailTransaction(
 517          $message, $reversePath, array($forwardPath), $failedRecipients
 518          );
 519      }
 520      return $sent;

 523    /** Try to determine the hostname of the server this is run on */

 524    private function _lookupHostname()
 525    {
 526      if (!empty($_SERVER['SERVER_NAME'])
 527        && $this->_isFqdn($_SERVER['SERVER_NAME']))
 528      {
 529        $this->_domain = $_SERVER['SERVER_NAME'];
 530      }
 531      elseif (!empty($_SERVER['SERVER_ADDR']))
 532      {
 533        $this->_domain = sprintf('[%s]', $_SERVER['SERVER_ADDR']);

 537    /** Determine is the $hostname is a fully-qualified name */

 538    private function _isFqdn($hostname)
 539    {
 540      //We could do a really thorough check, but there's really no point

 541      if (false !== $dotPos = strpos($hostname, '.'))
 542      {
 543        return ($dotPos > 0) && ($dotPos != strlen($hostname) - 1);
 544      }
 545      else
 546      {
 547        return false;
 548      }
 549    }
