 200    /**

 201     * Run a command against the buffer, expecting the given response codes.

 202     * If no response codes are given, the response will not be validated.

 203     * If codes are given, an exception will be thrown on an invalid response.

 204     * @param string $command

 205     * @param int[] $codes

 206     * @param string[] &$failures

 207     * @return string

 208     */
 209    public function executeCommand($command, $codes = array(), &$failures = null)
 210    {
 211      $failures = (array) $failures;
 212      $stopSignal = false;
 213      $response = null;
 214      foreach ($this->_getActiveHandlers() as $handler)
 215      {
 216        $response = $handler->onCommand(
 217          $this, $command, $codes, $failures, $stopSignal
 218          );
 219        if ($stopSignal)
 220        {
 221          return $response;
 222        }
 223      }
 224      return parent::executeCommand($command, $codes, $failures);
 225    }

 230    public function __call($method, $args)
 231    {
 232      foreach ($this->_handlers as $handler)
 233      {
 234        if (in_array(strtolower($method),
 235          array_map('strtolower', (array) $handler->exposeMixinMethods())
 236          ))
 237        {
 238          $return = call_user_func_array(array($handler, $method), $args);
 239          //Allow fluid method calls

 240          if (is_null($return) && substr($method, 0, 3) == 'set')
 241          {
 242            return $this;
 243          }
 244          else
 245          {
 246            return $return;
 247          }
 248        }
 249      }
 250      trigger_error('Call to undefined method ' . $method, E_USER_ERROR);
 251    }

 304      $this->_setHandlerParams();
 305      foreach ($this->_getActiveHandlers() as $handler)
 306      {

 308      }

 343    /** Determine ESMTP capabilities by function group */

 344    private function _getCapabilities($ehloResponse)
 345    {
 346      $capabilities = array();
 347      $ehloResponse = trim($ehloResponse);
 348      $lines = explode("\r\n", $ehloResponse);
 349      array_shift($lines);
 350      foreach ($lines as $line)
 351      {
 352        if (preg_match('/^[0-9]{3}[ -]([A-Z0-9-]+)((?:[ =].*)?)$/Di', $line, $matches))
 353        {
 354          $keyword = strtoupper($matches[1]);
 355          $paramStr = strtoupper(ltrim($matches[2], ' ='));
 356          $params = !empty($paramStr) ? explode(' ', $paramStr) : array();
 357          $capabilities[$keyword] = $params;
 358        }
 359      }
 360      return $capabilities;
 361    }

 363    /** Set parameters which are used by each extension handler */

 364    private function _setHandlerParams()

 367      {

 375    /** Get ESMTP handlers which are currently ok to use */

 376    private function _getActiveHandlers()
 377    {
 378      $handlers = array();
 379      foreach ($this->_handlers as $keyword => $handler)
 380      {

 384        }
 385      }
