  13   * Processes bytes as they pass through a buffer and replaces sequences in it.

  14   * This stream filter deals with Byte arrays rather than simple strings.

  15   * @package Swift

  16   * @author Chris Corbyn

  17   */
  18  class Swift_StreamFilters_ByteArrayReplacementFilter
  19    implements Swift_StreamFilter
  20  {
  21  
  22    /** The needle(s) to search for */

  23    private $_search;
  24  
  25    /** The replacement(s) to make */

  26    private $_replace;
  27  
  28    /** The Index for searching */

  29    private $_index;
  30  
  31    /** The Search Tree */

  32    private $_tree = array();
  33  
  34    /**  Gives the size of the largest search */

  35    private $_treeMaxLen = 0;
  36    
  37    private $_repSize;

  39    /**

  40     * Create a new ByteArrayReplacementFilter with $search and $replace.

  41     * @param array $search

  42     * @param array $replace

  43     */
  44    public function __construct($search, $replace)
  45    {
  46      $this->_search = $search;
  47      $this->_index = array();
  48      $this->_tree = array();
  49      $this->_replace = array();
  50      $this->_repSize = array();
  51      
  52      $tree = null;
  53      $i = null;
  54      $last_size = $size = 0;
  55      foreach ($search as $i => $search_element)
  56      {
  57        if ($tree !== null)
  58        {
  59          $tree[-1] = min (count($replace) - 1, $i - 1);
  60          $tree[-2] = $last_size;
  61        }
  62        $tree = &$this->_tree;
  63        if (is_array ($search_element))
  64        {
  65          foreach ($search_element as $k => $char)
  66          {
  67            $this->_index[$char] = true;
  68            if (!isset($tree[$char]))
  69            {
  70              $tree[$char] = array();
  71            }
  72            $tree = &$tree[$char];
  73          }
  74          $last_size = $k+1;
  75          $size = max($size, $last_size);
  76        }
  77        else
  78        {
  79          $last_size = 1;
  80          if (!isset($tree[$search_element]))
  81          {
  82            $tree[$search_element] = array();
  83          }
  84          $tree = &$tree[$search_element];
  85          $size = max($last_size, $size);
  86          $this->_index[$search_element] = true;
  87        }
  88      }
  89      if ($i !== null)
  90      {
  91        $tree[-1] = min (count ($replace) - 1, $i);
  92        $tree[-2] = $last_size;
  93        $this->_treeMaxLen = $size;
  94      }
  95      foreach ($replace as $rep)
  96      {
  97        if (!is_array($rep))
  98        {
  99          $rep = array ($rep);
 100        }
 101        $this->_replace[] = $rep;
 102      }
 103      for ($i = count($this->_replace) - 1; $i >= 0; --$i)
 104      {
 105        $this->_replace[$i] = $rep = $this->filter($this->_replace[$i], $i);
 106        $this->_repSize[$i] = count($rep);
 107      }
 108    }

 110    /**

 111     * Returns true if based on the buffer passed more bytes should be buffered.

 112     * @param array $buffer

 115    public function shouldBuffer($buffer)
 116    {
 117      $endOfBuffer = end($buffer);
 118      return isset ($this->_index[$endOfBuffer]);
 119    }

 121    /**

 122     * Perform the actual replacements on $buffer and return the result.

 123     * @param array $buffer

 124     * @return array

 125     */
 126    public function filter($buffer, $_minReplaces = -1)
 127    {
 128      if ($this->_treeMaxLen == 0)
 129      {
 130        return $buffer;
 131      }
 132      
 133      $newBuffer = array();
 134      $buf_size = count($buffer);
 135      for ($i = 0; $i < $buf_size; ++$i)
 136      {
 137        $search_pos = $this->_tree;
 138        $last_found = PHP_INT_MAX;
 139        // We try to find if the next byte is part of a search pattern

 140        for ($j = 0; $j <= $this->_treeMaxLen; ++$j)
 141        {
 142          // We have a new byte for a search pattern

 143          if (isset ($buffer [$p = $i + $j]) && isset($search_pos[$buffer[$p]]))
 144          {
 145            $search_pos = $search_pos[$buffer[$p]];
 146            // We have a complete pattern, save, in case we don't find a better match later

 147            if (isset($search_pos[- 1]) && $search_pos[-1] < $last_found
 148              && $search_pos[-1] > $_minReplaces)
 149            {
 150              $last_found = $search_pos[-1];
 151              $last_size = $search_pos[-2];
 152            }
 153          }
 154          // We got a complete pattern

 155          elseif ($last_found !== PHP_INT_MAX)
 156          {
 157            // Adding replacement datas to output buffer

 158            $rep_size = $this->_repSize[$last_found];
 159            for ($j = 0; $j < $rep_size; ++$j)
 160            {
 161              $newBuffer[] = $this->_replace[$last_found][$j];
 162            }
 163            // We Move cursor forward

 164            $i += $last_size - 1;
 165            // Edge Case, last position in buffer

 166            if ($i >= $buf_size)
 167            {
 168              $newBuffer[] = $buffer[$i];
 169            }
 170            
 171            // We start the next loop

 172            continue 2;
 173          }
 174          else
 175          {
 176            // this byte is not in a pattern and we haven't found another pattern

 177            break;
 178          }
 179        }
 180        // Normal byte, move it to output buffer

 181        $newBuffer[] = $buffer[$i];
 182      }
 183      
 184      return $newBuffer;
 185    }
