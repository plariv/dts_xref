 111     */
 112    public function sendPerformed(Swift_Events_SendEvent $evt)
 113    {
 114      ++$this->_counter;
 115      if ($this->_counter >= $this->_threshold)
 116      {
 117        $transport = $evt->getTransport();
 118        $transport->stop();
 119        if ($this->_sleep)
 120        {
 121          $this->sleep($this->_sleep);
 122        }
 123        $transport->start();
 124        $this->_counter = 0;
 125      }
 126    }
