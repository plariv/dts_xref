  12  /**

  13   * Allows customization of Messages on-the-fly.

  14   * 

  15   * @package Swift

  16   * @subpackage Plugins

  17   * 

  18   * @author Chris Corbyn

  19   * @author Fabien Potencier

  20   */
  21  class Swift_Plugins_DecoratorPlugin
  22    implements Swift_Events_SendListener, Swift_Plugins_Decorator_Replacements
  23  {
  24  
  25    /** The replacement map */

  26    private $_replacements;
  27    
  28    /** The body as it was before replacements */

  29    private $_originalBody;
  30  
  31    /** The original headers of the message, before replacements */

  32    private $_originalHeaders = array();
  33  
  34    /** Bodies of children before they are replaced */

  35    private $_originalChildBodies = array();
  36  
  37    /** The Message that was last replaced */

  38    private $_lastMessage;

  40    /**

  41     * Create a new DecoratorPlugin with $replacements.

  42     * 

  43     * The $replacements can either be an associative array, or an implementation

  44     * of {@link Swift_Plugins_Decorator_Replacements}.

  45     * 

  46     * When using an array, it should be of the form:

  47     * <code>

  48     * $replacements = array(

  49     *  "address1@domain.tld" => array("{a}" => "b", "{c}" => "d"),

  50     *  "address2@domain.tld" => array("{a}" => "x", "{c}" => "y")

  51     * )

  52     * </code>

  53     * 

  54     * When using an instance of {@link Swift_Plugins_Decorator_Replacements},

  55     * the object should return just the array of replacements for the address

  56     * given to {@link Swift_Plugins_Decorator_Replacements::getReplacementsFor()}.

  57     * 

  58     * @param mixed $replacements

  59     */
  60    public function __construct($replacements)
  61    {
  62      if (!($replacements instanceof Swift_Plugins_Decorator_Replacements))
  63      {
  64        $this->_replacements = (array) $replacements;
  65      }
  66      else
  67      {
  68        $this->_replacements = $replacements;
  69      }
  70    }
  71  
  72    /**

  73     * Invoked immediately before the Message is sent.

  74     * 

  75     * @param Swift_Events_SendEvent $evt

  76     */
  77    public function beforeSendPerformed(Swift_Events_SendEvent $evt)
  78    {
  79      $message = $evt->getMessage();
  80      $this->_restoreMessage($message);
  81      $to = array_keys($message->getTo());
  82      $address = array_shift($to);
  83      if ($replacements = $this->getReplacementsFor($address))
  84      {
  85        $body = $message->getBody();
  86        $search = array_keys($replacements);
  87        $replace = array_values($replacements);
  88        $bodyReplaced = str_replace(
  89          $search, $replace, $body
  90          );
  91        if ($body != $bodyReplaced)
  92        {
  93          $this->_originalBody = $body;
  94          $message->setBody($bodyReplaced);
  95        }
  96  
  97        foreach ($message->getHeaders()->getAll() as $header)
  98        {
  99          $body = $header->getFieldBodyModel();
 100          $count = 0;
 101          if (is_array($body))
 102          {
 103            $bodyReplaced = array();
 104            foreach ($body as $key => $value)
 105            {
 106              $count1 = 0;
 107              $count2 = 0;
 108              $key = is_string($key) ? str_replace($search, $replace, $key, $count1) : $key;
 109              $value = is_string($value) ? str_replace($search, $replace, $value, $count2) : $value;
 110              $bodyReplaced[$key] = $value;
 111  
 112              if (!$count && ($count1 || $count2))
 113              {
 114                $count = 1;
 115              }
 116            }
 117          }
 118          else
 119          {
 120            $bodyReplaced = str_replace($search, $replace, $body, $count);
 121          }
 122  
 123          if ($count)
 124          {
 125            $this->_originalHeaders[$header->getFieldName()] = $body;
 126            $header->setFieldBodyModel($bodyReplaced);
 127          }
 128        }
 129  
 130        $children = (array) $message->getChildren();
 131        foreach ($children as $child)
 132        {
 133          list($type, ) = sscanf($child->getContentType(), '%[^/]/%s');
 134          if ('text' == $type)
 135          {
 136            $body = $child->getBody();
 137            $bodyReplaced = str_replace(
 138              $search, $replace, $body
 139              );
 140            if ($body != $bodyReplaced)
 141            {
 142              $child->setBody($bodyReplaced);
 143              $this->_originalChildBodies[$child->getId()] = $body;
 144            }
 145          }
 146        }
 147        $this->_lastMessage = $message;
 148      }
 149    }

 151    /**

 152     * Find a map of replacements for the address.

 153     * 

 154     * If this plugin was provided with a delegate instance of

 155     * {@link Swift_Plugins_Decorator_Replacements} then the call will be

 156     * delegated to it.  Otherwise, it will attempt to find the replacements

 157     * from the array provided in the constructor.

 158     * 

 159     * If no replacements can be found, an empty value (NULL) is returned.

 160     * 

 161     * @param string $address

 162     * 

 163     * @return array

 164     */
 165    public function getReplacementsFor($address)
 166    {
 167      if ($this->_replacements instanceof Swift_Plugins_Decorator_Replacements)
 168      {
 169        return $this->_replacements->getReplacementsFor($address);
 170      }
 171      else
 172      {
 173        return isset($this->_replacements[$address])
 174          ? $this->_replacements[$address]
 175          : null
 176          ;
 177      }
 178    }

 181     * Invoked immediately after the Message is sent.

 182     * 

 183     * @param Swift_Events_SendEvent $evt

 184     */
 185    public function sendPerformed(Swift_Events_SendEvent $evt)
 186    {
 187      $this->_restoreMessage($evt->getMessage());
 188    }

 192    /** Restore a changed message back to its original state */

 193    private function _restoreMessage(Swift_Mime_Message $message)
 194    {
 195      if ($this->_lastMessage === $message)
 196      {
 197        if (isset($this->_originalBody))
 198        {
 199          $message->setBody($this->_originalBody);
 200          $this->_originalBody = null;
 201        }
 202        if (!empty($this->_originalHeaders))
 203        {
 204          foreach ($message->getHeaders()->getAll() as $header)
 205          {
 206            if (array_key_exists($header->getFieldName(), $this->_originalHeaders))
 207            {
 208              $header->setFieldBodyModel($this->_originalHeaders[$header->getFieldName()]);
 209            }
 210          }
 211          $this->_originalHeaders = array();
 212        }
 213        if (!empty($this->_originalChildBodies))
 214        {
 215          $children = (array) $message->getChildren();
 216          foreach ($children as $child)
 217          {
 218            $id = $child->getId();
 219            if (array_key_exists($id, $this->_originalChildBodies))
 220            {
 221              $child->setBody($this->_originalChildBodies[$id]);
 222            }
 223          }
 224          $this->_originalChildBodies = array();
 225        }
 226        $this->_lastMessage = null;
 227      }
 228    }
