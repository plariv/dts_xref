 106    /**

 107     * Creates a new QpEncoder for the given CharacterStream.

 108     * @param Swift_CharacterStream $charStream to use for reading characters

 109     * @param Swift_StreamFilter $filter if input should be canonicalized

 110     */
 111    public function __construct(Swift_CharacterStream $charStream,
 112      Swift_StreamFilter $filter = null)
 113    {
 114      $this->_charStream = $charStream;
 115      if(!isset(self::$_safeMapShare[$this->getSafeMapShareId()]))
 116      {
 117        $this->initSafeMap();
 118        self::$_safeMapShare[$this->getSafeMapShareId()] = $this->_safeMap;
 119      }
 120      else
 121      {
 122        $this->_safeMap = self::$_safeMapShare[$this->getSafeMapShareId()];
 123      }
 124      $this->_filter = $filter;
 125    }

 150    protected function initSafeMap()
 151    {
 152      foreach (array_merge(
 153        array(0x09, 0x20), range(0x21, 0x3C), range(0x3E, 0x7E)) as $byte)

 155        $this->_safeMap[$byte] = chr($byte);

 159    /**

 160     * Takes an unencoded string and produces a QP encoded string from it.

 161     * QP encoded strings have a maximum line length of 76 characters.

 162     * If the first line needs to be shorter, indicate the difference with

 163     * $firstLineOffset.

 164     * @param string $string to encode

 165     * @param int $firstLineOffset, optional

 166     * @param int $maxLineLength, optional, 0 indicates the default of 76 chars

 167     * @return string

 168     */
 169    public function encodeString($string, $firstLineOffset = 0,
 170      $maxLineLength = 0)
 171    {
 172      if ($maxLineLength > 76 || $maxLineLength <= 0)
 173      {
 174        $maxLineLength = 76;
 175      }
 176  
 177      $thisLineLength = $maxLineLength - $firstLineOffset;
 178  
 179      $lines = array();
 180      $lNo = 0;
 181      $lines[$lNo] = '';
 182      $currentLine =& $lines[$lNo++];
 183      $size=$lineLen=0;
 184  
 185      $this->_charStream->flushContents();
 186      $this->_charStream->importString($string);
 187  
 188      //Fetching more than 4 chars at one is slower, as is fetching fewer bytes

 189      // Conveniently 4 chars is the UTF-8 safe number since UTF-8 has up to 6

 190      // bytes per char and (6 * 4 * 3 = 72 chars per line) * =NN is 3 bytes

 191      while (false !== $bytes = $this->_nextSequence())
 192      {
 193        //If we're filtering the input

 194        if (isset($this->_filter))
 195        {
 196          //If we can't filter because we need more bytes

 197          while ($this->_filter->shouldBuffer($bytes))
 198          {
 199            //Then collect bytes into the buffer

 200            if (false === $moreBytes = $this->_nextSequence(1))
 201            {
 202              break;
 203            }
 204  
 205            foreach ($moreBytes as $b)
 206            {
 207              $bytes[] = $b;
 208            }
 209          }
 210          //And filter them

 211          $bytes = $this->_filter->filter($bytes);
 212        }
 213  
 214        $enc = $this->_encodeByteSequence($bytes, $size);
 215        if ($currentLine && $lineLen+$size >= $thisLineLength)
 216        {
 217          $lines[$lNo] = '';
 218          $currentLine =& $lines[$lNo++];
 219          $thisLineLength = $maxLineLength;
 220          $lineLen=0;
 221        }
 222        $lineLen+=$size;
 223        $currentLine .= $enc;
 224      }
 225  
 226      return $this->_standardize(implode("=\r\n", $lines));
 227    }

 240    /**

 241     * Encode the given byte array into a verbatim QP form.

 242     * @param int[] $bytes

 243     * @return string

 244     * @access protected

 245     */
 246    protected function _encodeByteSequence(array $bytes, &$size)
 247    {
 248      $ret = '';
 249      $size=0;
 250      foreach ($bytes as $b)
 251      {
 252        if (isset($this->_safeMap[$b]))
 253        {
 254          $ret .= $this->_safeMap[$b];
 255          ++$size;
 256        }
 257        else
 258        {
 259          $ret .= self::$_qpMap[$b];
 260          $size+=3;
 261        }
 262      }
 263      return $ret;
 264    }

 277    /**

 278     * Make sure CRLF is correct and HT/SPACE are in valid places.

 279     * @param string $string

 280     * @return string

 281     * @access protected

 282     */
 283    protected function _standardize($string)
 284    {
 285      $string = str_replace(array("\t=0D=0A", " =0D=0A", "=0D=0A"),
 286        array("=09\r\n", "=20\r\n", "\r\n"), $string
 287        );
 288      switch ($end = ord(substr($string, -1)))
 289      {
 290        case 0x09:
 291        case 0x20:
 292          $string = substr_replace($string, self::$_qpMap[$end], -1);
 293      }
 294      return $string;
 295    }
