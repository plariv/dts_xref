 421    /**

 422     * Set the priority of this message.

 423     * The value is an integer where 1 is the highest priority and 5 is the lowest.

 424     * @param int $priority

 425     * @return Swift_Mime_SimpleMessage

 426     */
 427    public function setPriority($priority)
 428    {
 429      $priorityMap = array(
 430        1 => 'Highest',
 431        2 => 'High',
 432        3 => 'Normal',
 433        4 => 'Low',
 434        5 => 'Lowest'
 435        );
 436      $pMapKeys = array_keys($priorityMap);
 437      if ($priority > max($pMapKeys))
 438      {
 439        $priority = max($pMapKeys);
 440      }
 441      elseif ($priority < min($pMapKeys))
 442      {
 443        $priority = min($pMapKeys);
 444      }
 445      if (!$this->_setHeaderFieldModel('X-Priority',
 446        sprintf('%d (%s)', $priority, $priorityMap[$priority])))
 447      {
 448        $this->getHeaders()->addTextHeader('X-Priority',
 449          sprintf('%d (%s)', $priority, $priorityMap[$priority]));
 450      }
 451      return $this;
 452    }

 593    /** Turn the body of this message into a child of itself if needed */

 594    private function _becomeMimePart()
 595    {
 596      $part = new parent($this->getHeaders()->newInstance(), $this->getEncoder(),
 597        $this->_getCache(), $this->_getGrammar(), $this->_userCharset
 598        );
 599      $part->setContentType($this->_userContentType);
 600      $part->setBody($this->getBody());
 601      $part->setFormat($this->_userFormat);
 602      $part->setDelSp($this->_userDelSp);
 603      $part->_setNestingLevel($this->_getTopNestingLevel());
 604      return $part;
 605    }

 607    /** Get the highest nesting level nested inside this message */

 608    private function _getTopNestingLevel()
 609    {
 610      $highestLevel = $this->getNestingLevel();
 611      foreach ($this->getChildren() as $child)
 612      {
 613        $childLevel = $child->getNestingLevel();
 614        if ($highestLevel < $childLevel)
 615        {
 616          $highestLevel = $childLevel;
 617        }
 618      }
 619      return $highestLevel;
