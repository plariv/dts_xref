  21

  27

  34    /**

  35     * Get the level at which this entity shall be nested in final document.

  36     * The lower the value, the more outermost the entity will be nested.

  37     * @return int

  38     * @see LEVEL_TOP, LEVEL_MIXED, LEVEL_RELATED, LEVEL_ALTERNATIVE

  39     */
  40    public function getNestingLevel();

  43     * Get the qualified content-type of this mime entity.

  44     * @return string

  45     */

  47    
  48    /**

  49     * Returns a unique ID for this entity.

  50     * For most entities this will likely be the Content-ID, though it has

  53     * If a Content-ID header is present, this value SHOULD match the value of

  54     * the header.

  55     * @return string

  56     */
  57    public function getId();

  71    public function setChildren(array $children);
  72    
  73    /**

  76     */
  77    public function getHeaders();

  84    public function getBody();

  91    public function setBody($body, $contentType = null);
  92    
  93    /**

  94     * Get this entire entity in its string form.

  97    public function toString();

 100     * Get this entire entity as a ByteStream.

 103    public function toByteStream(Swift_InputByteStream $is);
