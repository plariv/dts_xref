 255    /**

 256     * Set all children of this entity.

 257     * @param array $children Swiift_Mime_Entity instances

 258     * @param int $compoundLevel For internal use only

 259     * @return Swift_Mime_SimpleMimeEntity

 260     */
 261    public function setChildren(array $children, $compoundLevel = null)
 262    {
 263      //TODO: Try to refactor this logic

 264      
 265      $compoundLevel = isset($compoundLevel)
 266        ? $compoundLevel
 267        : $this->_getCompoundLevel($children)
 268        ;
 269      
 270      $immediateChildren = array();
 271      $grandchildren = array();
 272      $newContentType = $this->_userContentType;
 273      
 274      foreach ($children as $child)
 275      {
 276        $level = $this->_getNeededChildLevel($child, $compoundLevel);
 277        if (empty($immediateChildren)) //first iteration
 278        {
 279          $immediateChildren = array($child);
 280        }
 281        else
 282        {
 283          $nextLevel = $this->_getNeededChildLevel($immediateChildren[0], $compoundLevel);
 284          if ($nextLevel == $level)
 285          {
 286            $immediateChildren[] = $child;
 287          }
 288          elseif ($level < $nextLevel)
 289          {
 290            //Re-assign immediateChildren to grandchilden

 291            $grandchildren = array_merge($grandchildren, $immediateChildren);
 292            //Set new children

 293            $immediateChildren = array($child);
 294          }
 295          else
 296          {
 297            $grandchildren[] = $child;
 298          }
 299        }
 300      }
 301      
 302      if (!empty($immediateChildren))
 303      {
 304        $lowestLevel = $this->_getNeededChildLevel($immediateChildren[0], $compoundLevel);
 305        
 306        //Determine which composite media type is needed to accomodate the

 307        // immediate children

 308        foreach ($this->_compositeRanges as $mediaType => $range)
 309        {
 310          if ($lowestLevel > $range[0]
 311            && $lowestLevel <= $range[1])
 312          {
 313            $newContentType = $mediaType;
 314            break;
 315          }
 316        }
 317        
 318        //Put any grandchildren in a subpart

 319        if (!empty($grandchildren))
 320        {
 321          $subentity = $this->_createChild();
 322          $subentity->_setNestingLevel($lowestLevel);
 323          $subentity->setChildren($grandchildren, $compoundLevel);
 324          array_unshift($immediateChildren, $subentity);
 325        }
 326      }
 327      
 328      $this->_immediateChildren = $immediateChildren;
 329      $this->_children = $children;
 330      $this->_setContentTypeInHeaders($newContentType);
 331      $this->_fixHeaders();
 332      $this->_sortChildren();
 333      
 334      return $this;
 335    }

 447    public function toString()
 448    {
 449      $string = $this->_headers->toString();
 450      if (isset($this->_body) && empty($this->_immediateChildren))
 451      {
 452        if ($this->_cache->hasKey($this->_cacheKey, 'body'))
 453        {
 454          $body = $this->_cache->getString($this->_cacheKey, 'body');
 455        }
 456        else
 457        {
 458          $body = "\r\n" . $this->_encoder->encodeString($this->getBody(), 0,
 459            $this->getMaxLineLength()
 460            );
 461          $this->_cache->setString($this->_cacheKey, 'body', $body,
 462            Swift_KeyCache::MODE_WRITE
 463            );
 464        }
 465        $string .= $body;
 466      }
 467      
 468      if (!empty($this->_immediateChildren))
 469      {
 470        foreach ($this->_immediateChildren as $child)
 471        {
 472          $string .= "\r\n\r\n--" . $this->getBoundary() . "\r\n";
 473          $string .= $child->toString();
 474        }
 475        $string .= "\r\n\r\n--" . $this->getBoundary() . "--\r\n";
 476      }
 477      
 478      return $string;
 479    }

 493    /**

 494     * Write this entire entity to a {@link Swift_InputByteStream}.

 495     * @param Swift_InputByteStream

 496     */
 497    public function toByteStream(Swift_InputByteStream $is)
 498    {
 499      $is->write($this->_headers->toString());
 500      $is->commit();
 501      
 502      if (empty($this->_immediateChildren))
 503      {
 504        if (isset($this->_body))
 505        {
 506          if ($this->_cache->hasKey($this->_cacheKey, 'body'))
 507          {
 508            $this->_cache->exportToByteStream($this->_cacheKey, 'body', $is);
 509          }
 510          else
 511          {
 512            $cacheIs = $this->_cache->getInputByteStream($this->_cacheKey, 'body');
 513            if ($cacheIs)
 514            {
 515              $is->bind($cacheIs);
 516            }
 517            
 518            $is->write("\r\n");
 519            
 520            if ($this->_body instanceof Swift_OutputByteStream)
 521            {
 522              $this->_body->setReadPointer(0);
 523              
 524              $this->_encoder->encodeByteStream($this->_body, $is, 0,
 525                $this->getMaxLineLength()
 526                );
 527            }
 528            else
 529            {
 530              $is->write($this->_encoder->encodeString(
 531                $this->getBody(), 0, $this->getMaxLineLength()
 532                ));
 533            }
 534            
 535            if ($cacheIs)
 536            {
 537              $is->unbind($cacheIs);
 538            }
 539          }
 540        }
 541      }
 542      
 543      if (!empty($this->_immediateChildren))
 544      {
 545        foreach ($this->_immediateChildren as $child)
 546        {
 547          $is->write("\r\n\r\n--" . $this->getBoundary() . "\r\n");
 548          $child->toByteStream($is);
 549        }
 550        $is->write("\r\n\r\n--" . $this->getBoundary() . "--\r\n");
 551      }
 552    }

 738    private function _getNeededChildLevel($child, $compoundLevel)
 739    {
 740      $filter = array();
 741      foreach ($this->_compoundLevelFilters as $bitmask => $rules)
 742      {
 743        if (($compoundLevel & $bitmask) === $bitmask)
 744        {
 745          $filter = $rules + $filter;
 746        }
 747      }
 748      
 749      $realLevel = $child->getNestingLevel();
 750      $lowercaseType = strtolower($child->getContentType());
 751      
 752      if (isset($filter[$realLevel])
 753        && isset($filter[$realLevel][$lowercaseType]))
 754      {
 755        return $filter[$realLevel][$lowercaseType];
 756      }
 757      else
 758      {
