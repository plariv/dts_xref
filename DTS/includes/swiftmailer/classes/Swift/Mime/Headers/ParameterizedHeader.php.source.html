 126    public function getFieldBody() //TODO: Check caching here
 127    {
 128      $body = parent::getFieldBody();
 129      foreach ($this->_params as $name => $value)
 130      {
 131        if (!is_null($value))
 132        {
 133          //Add the parameter

 134          $body .= '; ' . $this->_createParameter($name, $value);
 135        }
 136      }
 137      return $body;

 142    /**

 143     * Generate a list of all tokens in the final header.

 144     * This doesn't need to be overridden in theory, but it is for implementation

 145     * reasons to prevent potential breakage of attributes.

 146     * @param string $string The string to tokenize

 147     * @return array An array of tokens as strings

 148     * @access protected

 149     */
 150    protected function toTokens($string = null)
 151    {
 152      $tokens = parent::toTokens(parent::getFieldBody());
 153      
 154      //Try creating any parameters

 155      foreach ($this->_params as $name => $value)
 156      {
 157        if (!is_null($value))
 158        {
 159          //Add the semi-colon separator

 160          $tokens[count($tokens)-1] .= ';';
 161          $tokens = array_merge($tokens, $this->generateTokenLines(
 162            ' ' . $this->_createParameter($name, $value)
 163            ));
 164        }
 165      }
 166      
 167      return $tokens;
 168    }

 172    /**

 173     * Render a RFC 2047 compliant header parameter from the $name and $value.

 174     * @param string $name

 175     * @param string $value

 176     * @return string

 177     * @access private

 178     */
 179    private function _createParameter($name, $value)
 180    {
 181      $origValue = $value;
 182      
 183      $encoded = false;
 184      //Allow room for parameter name, indices, "=" and DQUOTEs

 185      $maxValueLength = $this->getMaxLineLength() - strlen($name . '=*N"";') - 1;
 186      $firstLineOffset = 0;
 187      
 188      //If it's not already a valid parameter value...

 189      if (!preg_match('/^' . self::TOKEN_REGEX . '$/D', $value))
 190      {
 191        //TODO: text, or something else??

 192        //... and it's not ascii

 193        if (!preg_match('/^' . $this->getGrammar()->getDefinition('text') . '*$/D', $value))
 194        {
 195          $encoded = true;
 196          //Allow space for the indices, charset and language

 197          $maxValueLength = $this->getMaxLineLength() - strlen($name . '*N*="";') - 1;
 198          $firstLineOffset = strlen(
 199            $this->getCharset() . "'" . $this->getLanguage() . "'"
 200            );
 201        }
 202      }
 203      
 204      //Encode if we need to

 205      if ($encoded || strlen($value) > $maxValueLength)
 206      {
 207        if (isset($this->_paramEncoder))
 208        {
 209          $value = $this->_paramEncoder->encodeString(
 210            $origValue, $firstLineOffset, $maxValueLength
 211            );
 212        }
 213        else //We have to go against RFC 2183/2231 in some areas for interoperability
 214        {
 215          $value = $this->getTokenAsEncodedWord($origValue);
 216          $encoded = false;
 217        }
 218      }
 219      
 220      $valueLines = isset($this->_paramEncoder) ? explode("\r\n", $value) : array($value);
 221      
 222      //Need to add indices

 223      if (count($valueLines) > 1)
 224      {
 225        $paramLines = array();
 226        foreach ($valueLines as $i => $line)
 227        {
 228          $paramLines[] = $name . '*' . $i .
 229            $this->_getEndOfParameterValue($line, true, $i == 0);
 230        }
 231        return implode(";\r\n ", $paramLines);
 232      }
 233      else
 234      {
 235        return $name . $this->_getEndOfParameterValue(
 236          $valueLines[0], $encoded, true
 237          );
 238      }
 239    }

 241    /**

 242     * Returns the parameter value from the "=" and beyond.

 243     * @param string $value to append

 244     * @param boolean $encoded

 245     * @param boolean $firstLine

 246     * @return string

 247     * @access private

 248     */
 249    private function _getEndOfParameterValue($value, $encoded = false, $firstLine = false)
 250    {
 251      if (!preg_match('/^' . self::TOKEN_REGEX . '$/D', $value))
 252      {
 253        $value = '"' . $value . '"';
 254      }
 255      $prepend = '=';
 256      if ($encoded)
 257      {
 258        $prepend = '*=';
 259        if ($firstLine)
 260        {
 261          $prepend = '*=' . $this->getCharset() . "'" . $this->getLanguage() .
 262            "'";
 263        }
 264      }
 265      return $prepend . $value;
 266    }
