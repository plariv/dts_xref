  56    /**

  57     * Encode stream $in to stream $out.

  58     * QP encoded strings have a maximum line length of 76 characters.

  59     * If the first line needs to be shorter, indicate the difference with

  60     * $firstLineOffset.

  61     * @param Swift_OutputByteStream $os output stream

  62     * @param Swift_InputByteStream $is input stream

  63     * @param int $firstLineOffset

  64     * @param int $maxLineLength

  65     */
  66    public function encodeByteStream(
  67      Swift_OutputByteStream $os, Swift_InputByteStream $is, $firstLineOffset = 0,
  68      $maxLineLength = 0)
  69    {
  70      if ($maxLineLength > 76 || $maxLineLength <= 0)
  71      {
  72        $maxLineLength = 76;
  73      }
  74  
  75      $thisLineLength = $maxLineLength - $firstLineOffset;
  76  
  77      $this->_charStream->flushContents();
  78      $this->_charStream->importByteStream($os);
  79      
  80      $currentLine = '';
  81      $prepend = '';
  82      $size=$lineLen=0;
  83  
  84      while (false !== $bytes = $this->_nextSequence())
  85      {
  86        //If we're filtering the input

  87        if (isset($this->_filter))
  88        {
  89          //If we can't filter because we need more bytes

  90          while ($this->_filter->shouldBuffer($bytes))
  91          {
  92            //Then collect bytes into the buffer

  93            if (false === $moreBytes = $this->_nextSequence(1))
  94            {
  95              break;
  96            }
  97  
  98            foreach ($moreBytes as $b)
  99            {
 100              $bytes[] = $b;
 101            }
 102          }
 103          //And filter them

 104          $bytes = $this->_filter->filter($bytes);
 105        }
 106  
 107        $enc = $this->_encodeByteSequence($bytes, $size);
 108        if ($currentLine && $lineLen+$size >= $thisLineLength)
 109        {
 110          $is->write($prepend . $this->_standardize($currentLine));
 111          $currentLine = '';
 112          $prepend = "=\r\n";
 113          $thisLineLength = $maxLineLength;
 114          $lineLen=0;
 115        }
 116        $lineLen+=$size;
 117        $currentLine .= $enc;
 118      }
 119      if (strlen($currentLine))
 120      {
 121        $is->write($prepend . $this->_standardize($currentLine));
 122      }
 123    }
