 716      do {

 724              $bigString .= " XXX "; // and replace it with an "and" placeholder

 725          }
 726          
 727          $bigString .= $thisString." ||| ";
 728          
 729      } while ($thisText = next($arrayStrings));
 730      
 731      // convert "and" placeholder into an and

 732      $bigString = str_replace(" XXX ", ", and ", $bigString);
