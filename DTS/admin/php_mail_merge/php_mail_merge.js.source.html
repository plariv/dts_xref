 376  function convert2_Color(v) {
 377  	function hex(d) {
 378          return (d < 16) ? ("0" + d.toString(16)) : d.toString(16);
 379      };
 380      if (typeof v == "number") {
 381          var r = v & 0xFF;
 382          var g = (v >> 8) & 0xFF;
 383          var b = (v >> 16) & 0xFF;
 384          return "#" + hex(r) + hex(g) + hex(b);
 385      }
 386  
 387      if (v.substr(0, 3) == "rgb") {
 388          var re = /rgb\s*\(\s*([0-9]+)\s*,\s*([0-9]+)\s*,\s*([0-9]+)\s*\)/;
 389          if (v.match(re)) {
 390              var r = parseInt(RegExp.$1);
 391              var g = parseInt(RegExp.$2);
 392              var b = parseInt(RegExp.$3);
 393              return "#" + hex(r) + hex(g) + hex(b);
 394          }
 395          return null;
 396      }
 397      if (v[0] == "#") {
 398          return v;
 399      }
 400      return null;
 401  };

 569    function insertNodeAtSelection(win, insertNode){
 570    html_box.focus();
 571    if(ie){
 572     sel = win.document.selection.createRange();
 573     sel.pasteHTML(insertNode.outerHTML);
 574  
 575    }
 576    else{
 577        var sel = win.getSelection();
 578        var range = sel.getRangeAt(0);
 579        sel.removeAllRanges();
 580        range.deleteContents();
 581        var container = range.startContainer;
 582        var pos = range.startOffset;
 583        range=document.createRange();
 584        if (container.nodeType==3 && insertNode.nodeType==3) {
 585          container.insertData(pos, insertNode.nodeValue);
 586          range.setEnd(container, pos+insertNode.length);
 587          range.setStart(container, pos+insertNode.length);
 588        } else {
 589          var afterNode;
 590          if (container.nodeType==3) {
 591            var textNode = container;
 592            container = textNode.parentNode;
 593            var text = textNode.nodeValue;
 594            var textBefore = text.substr(0,pos);
 595            var textAfter = text.substr(pos);
 596            var beforeNode = document.createTextNode(textBefore);
 597            afterNode = document.createTextNode(textAfter);
 598            container.insertBefore(afterNode, textNode);
 599            container.insertBefore(insertNode, afterNode);
 600            container.insertBefore(beforeNode, insertNode);
 601            container.removeChild(textNode);
 602          } else {
 603            afterNode = container.childNodes[pos];
 604            container.insertBefore(insertNode, afterNode);
 605          }
 606          range.setEnd(afterNode, 0);
 607          range.setStart(afterNode, 0);
 608        }
 609        sel.addRange(range);
 610        }
 611    };
