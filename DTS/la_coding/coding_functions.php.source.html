 161  function calculate_ldma_scale_averages_for_man_level ($manlevelgeneralkey, $test_time) {
 162      
 163      // clear out previous data for this man level
 164      $sql = sprintf("DELETE FROM man_levels_theme_averages 
 165          WHERE instrument_id = 'LDMA' AND manlevelgeneralkey = %s AND test_time = %s",
 166              SQLstr($manlevelgeneralkey, 'int'),
 167              SQLstr($test_time, 'int'));
 168      $result = mysql_query($sql) or die("ERROR removing previous stored averages for this manlevel.");
 169  
 170      // get averages for all scale scores for this management level
 171      $sql = sprintf("SELECT 
 172              ROUND(AVG(c.code451)) AS avg_ptaken, 
 173              ROUND(AVG(c.code452)) AS avg_psought, 
 174              ROUND(AVG(c.code453), 1) AS avg_pcoord, 
 175              ROUND(AVG(c.code454), 1) AS avg_context, 
 176              ROUND(AVG(c.code455), 1) AS avg_dmprocess, 
 177              ROUND(AVG(c.code456), 1) AS avg_collab, 
 178              COUNT(DISTINCT d.lakey) AS num_assessments
 179          FROM la_data d 
 180              JOIN la_codes c ON d.lakey = c.lakey
 181              JOIN demogs_man_level dm ON d.manlevelkey = dm.manlevelkey
 182              JOIN demogs_man_level_general dmg ON dm.man_level_number = dmg.man_level_number
 183          WHERE d.instrument_id IN('LDMA', 'LDMA_JOURNAL') AND dmg.manlevelgeneralkey = %s
 184              AND d.test_time = %s",
 185              SQLstr($manlevelgeneralkey, 'int'),
 186              SQLstr($test_time, 'int'));
 187      $rs = mysql_query($sql) or die("ERROR fetching average scores for this manlevel.");
 188      $row = mysql_fetch_assoc($rs);
 189  
 190      // only continue if at least 10 records exist for this management level & test time
 191      if ($row['num_assessments'] >= 10) {
 192          // store averages
 193          $k = intval($manlevelgeneralkey);
 194          $t = intval($test_time);
 195          $n = intval($row['num_assessments']);
 196          $sql = sprintf("INSERT INTO man_levels_theme_averages
 197              (manlevelgeneralkey, instrument_id, test_time, theme_name, average, based_on_num_records)

 199              ({$k}, 'LDMA', {$t}, 'perspective_taking', %s, {$n}),
 200              ({$k}, 'LDMA', {$t}, 'perspective_seeking', %s, {$n}),

 202              ({$k}, 'LDMA', {$t}, 'contextual_thinking', %s, {$n}),
 203              ({$k}, 'LDMA', {$t}, 'decision_making_process', %s, {$n}),
 204              ({$k}, 'LDMA', {$t}, 'collaborative_capacity', %s, {$n}) ",
 205                  SQLstr($row['avg_ptaken'], 'int'),
 206                  SQLstr($row['avg_psought'], 'int'),
 207                  SQLstr($row['avg_pcoord'], 'double'),
 208                  SQLstr($row['avg_context'], 'double'),
 209                  SQLstr($row['avg_dmprocess'], 'double'),
 210                  SQLstr($row['avg_collab'], 'double'));
